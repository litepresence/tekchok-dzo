[2684] །བསྒོམ་པས་ཤེས་པ་གསལ་བ་དང༌།
[2685] །སེམས་ནི་སྟོང་པར་འདོད་པའོ།
[2686] །དེས་ནི་གཏི་མུག་སྡུག་བསྔལ་སྤོང༌།
[2687] །དེ་ལྟར་རྣམ་བཞི་བསྒོམས་པ་ཡིས།
[2688] །འབྲས་བུ་རྣམ་པ་བཞི་ཐོབ་བོ།
[2689] །འབྲས་བུ་བཞི་པོ་གང་ཞེ་ན།
[2690] །འགོག་པ་བསྒོམས་པའི་འབྲས་བུ་ནི།
[2691] །རྒྱུན་དུ་ཞུགས་པ་ཐོབ་པ་སྟེ།
[2692] །རབ་ཏུ་དགའ་བ་ཐོབ་པ་ནས།
[2693] །དེ་བཞིན་ས་བདུན་མན་ཆད་དོ།
[2694] །ཀུན་འབྱུང་བསྒོམས་པའི་རང་བཞིན་གྱིས།
[2695] །ལན་ཅིག་ཕྱིར་མི་འོང་བ་ཐོབ།
[2696] །ས་བརྒྱད་མན་ཆད་ཐོབ་པའོ།
[2697] །ལམ་བསྒོམ་པ་ཡི་ཡོན་ཏན་གྱིས།
[2698] །དགྲ་བཅོམ་འབྲས་བུ་ཐོབ་པ་ནི།
[2699] །སུ་བཅུ་མཐར་ཕྱིན་ཐོབ་པའོ།
[2700] །ས་དང་མ་ལ་གནས་པའི་ཚེ།
[2701] །ཉན་ཅིང་ཐོས་པར་བྱེད་པའོ།
[2702] །ས་བདུན་མན་ཆད་སེམས་དཔའ་ནི།
[2703] །རང་རང་དཔོན་ལ་ཉན་པའོ།
[2704] །ས་བརྒྱད་མན་ཆད་སེམས་དཔའ་ནི།
[2705] །སྤྲུལ་པའི་སྐུ་ལ་ཉན་པར་འདོད།
[2706] །དེ་ཡང་དངོས་པོར་མཐོང་མེད་དོ།
[2707] །བྱིན་གྱི་རླབས་ཀྱིས་ཐོས་པའོ།
[2708] །ཉན་ཐོས་ཐེག་པ་དེ་ལྟར་འདོད།
[2709] །ཅེས་སོ།
[2710] །གཉིས་པ་རང་སངས་རྒྱས་ཀྱི་ཐེག་པ་ལའང་ལྔ་ལས།
[2711] ངོ་བོ་ནི།
[2712] རང་དོན་ཕྱོགས་གཅིག་པས་བྱང་ཆུབ་སློབ་དཔོན་ལ་མ་བརྟེན་པར་སྲིད་པ་ཐ་མའི་ཚེ་མངོན་དུ་བྱེད་པའི་ཐབས་བདག་མེད་ཕྱེད་དང་གཉིས་རྟོགས་པའི་རང་བཞིན་ནོ།
[2713] །ངེས་ཚིག་ནི་རང་ཉིད་གཅིག་པུ་སངས་རྒྱས་སུ་འདོད་པའོ
[2714] །དགོས་པ་ནི་འཁོར་བ་ལས་ཐར་བའི་བྱང་ཆུབ་འབྲིང་ཐོབ་པའོ།
[2715] །འབྲས་བུ་ནི་དགྲ་བཅོམ་གཉིས་སོ།
[2716] །དབྱེ་ན།
[2717] ནེ་ཙོ་ལྟར་ཁྱུས་ལྡིང་པ་དང༌།
[2718] བསེ་རུ་ལྟར་གཅིག་པུར་གནས་པ་གཉིས་ལས།
[2719] བསེ་རུ་ནི་དམིགས་ཡུལ་བདེན་པ་བཞིས་བསྡུས་པའི་ཆོས་རྣམས་ལ།
[2720] རྟེན་འབྲེལ་བཅུ་གཉིས་ལུགས་འབྱུང་དུ་བསྒོམས་པས།
[2721] ཁམས་གསུམ་པའི་ཕྲ་རྒྱས་ཐམས་ཅད་སྤོང་བར་འདོད་དོ།
[2722] །དེ་ཡང་མ་རིག་པའི་རྒྱུ་ལས་འདུ་བྱེད།
[2723] དེ་ལས་རྣམ་པ་ཤེས་པ་ནས་རྒ་ཤིའི་བར་ལ་སྔ་ཕྱི་རྒྱུ་འབྲས་སུ་ཤེས་ཤིང༌།
[2724] ལྟ་བ་གང་ཟག་གི་བདག་དང༌།
[2725] ཆོས་བདག་གི་ཕྱེད་ཕྱི་གཟུང་བ་རྡུལ་ཕྲན་ཆ་བཅས་དང་ཆ་མེད་གཉིས་ཀ་མེད་པར།
