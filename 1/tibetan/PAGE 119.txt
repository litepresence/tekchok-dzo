[4417] སངས་རྒྱས་རྣམས་ཀྱི་བཅུད་བསྡུས་ནས།
[4418] །འགྲོ་བ་ཀུན་གྱི་སྙིང་ལ་བཅུག
[4419] །དེ་ཡི་སྙིང་པོ་ལུ་གུ་རྒྱུད།
[4420] །དངས་མ་བསྡུས་བས་བཅུད་ཅེས་བྱ།
[4421] །ཞེས་པ་དང༌།
[4422] བརྩོན་པ་དོན་ལྡན་གྱི་རྒྱུད་ལས།
[4423] བསྟན་པ་ཀུན་གྱི་ཡང་བཅུད་ནི།
[4424] །ཆོས་ཉིད་མངོན་སུམ་ལམ་ཞེས་བྱ།
[4425] །ཞེས་པ་དང༌།
[4426] རྡོ་རྗེ་ནམ་མཁའ་དང་མཉམ་པའི་རྒྱུད་ལས།
[4427] གཞི་དང་སྙིང་པོ་མེ་ཏོག་ནི།
[4428] །སངས་རྒྱས་ཀུན་གྱི་རང་བཞིན་རྒྱུད།
[4429] །ཁྱད་པར་གསང་མཆོག་བཅུད་དུ་དྲིལ།
[4430] །འདིས་ནི་འཁོར་བའི་ཕ་རོལ་དུ།
[4431] །སྐྱེས་བུ་མདའ་འཕངས་ཇི་བཞིན་ནོ།
[4432] །ཞེས་པ་དང༌།
[4433] གསེར་འོད་དམ་པ་ལས།
[4434] གཞི་དང་རང་བཞིན་ཐབས་དག་ལ།
[4435] །བརྟེན་ནས་སངས་རྒྱས་ཞིང་དུ་ཡང༌།
[4436] །ཚེགས་མེད་སླ་ཕྱིར་རྒྱུད་དུ་གཞག
[4437] །ཅེས་སོ།
[4438] །དེ་ལྟར་ན་འདི་དག་གི་དོན་ངེས་པ་ནི།
[4439] མུ་ཏིག་ཕྲེང་བ་ལས།
[4440] རང་བཞིན་རྒྱུད་ནི་འདི་ལྟ་སྟེ།
[4441] གཞི་དང་སྙིང་པོ་བཅུད་ཉིད་དོ།
[4442] །གཞི་ནི་ངོ་བོ་རང་བཞིན་དང༌།
[4443] །ཐུགས་རྗེ་ཀུན་ལ་ཁྱབ་ཅིང་འཆར།
[4444] །ཡེ་ཤེས་གསུམ་གྱི་བདག་ཉིད་ཅན།
[4445] །ཡེ་ནས་དྲི་མ་ཀ་ནས་དག
[4446] །འདུ་འབྲལ་མེད་པའི་ཉག་གཅིག་ཉིད།
[4447] །སངས་རྒྱས་རྣམས་ཀྱི་གསང་ཆེན་ཡིན།
[4448] །སྐུ་ནི་སྙིང་པོ་རྫོགས་པའི་རིག
[4449] །སྐུ་ཡི་རྒྱུད་ནི་སྟོང་པས་ཁྱབ།
[4450] །གསལ་བ་ཉིད་ཀྱང་རང་སྣང་ངོ༌།
[4451] །མི་སྣང་གསལ་བའི་རང་བཞིན་ཅན།
[4452] །སྙིང་པོ་རྫོགས་པའི་རང་བཞིན་རྒྱུད།
[4453] །སྣང་བ་ཐམས་ཅད་རྒྱུད་ཡིན་ཏེ།
[4454] །ཐམས་ཅད་ཐམས་ཅད་ཅད་འབྲེལ་ཞིང་ཆགས།
[4455] །རང་བཞིན་ཙིཏྟའི་དཀྱིལ་འཁོར་དུ།
[4456] །མ་བསྐྱེད་རྫོགས་པས་སྙིང་པོ་ཆེ།
[4457] །དེ་ཕྱིར་དྲིལ་བས་སྙིང་པོ་སྟེ།
[4458] །ཀུན་གྱི་སྲོག་ཏུ་གྱུར་པའི་ཕྱིར།
[4459] །སྙིང་པོའི་རྒྱུད་ཅེས་རྣམ་པར་བཤད།
[4460] །སངས་རྒྱས་ཐམས་ཅད་གཅིག་བསྡུས་པའི།
[4461] །ངོ་བོ་ཐམས་ཅད་བཅུད་ཕྱུང་བའི།
[4462] །སྙིང་པོ་བསྡུས་པས་བཅུད་ཅེས་ཏེ།
[4463] །སྟོང་གསལ་རིག་པའི་བཅུད་དུ་དྲིལ།
[4464] །སངས་རྒྱས་ཀུན་གྱི་བདེ་གསང་ཆེ།
[4465] །གཞི་དང་སྙིང་པོ་མེ་ཏོག་ཅན།
[4466] །རང་རིག་རྣམས་ཀྱི་ངོ་བོ་ཉིད།
[4467] །རྫོགས་པ་ཆེན་པོའི་བཅུད་ཅེས་ནི།
[4468] །རྒྱུད་གཞན་ལས་ཀྱང་རབ་ཏུ་གྲགས།
[4469] །གསང་བའི་རྒྱུད་ནི་རྣམ་བཤད་པ།
[4470] །ངེས་པས་གཟུང་དུ་ཡོད་པའི་ལུང༌།
[4471] །རང་བཞིན་སྟོང་པས་ཀུན་ཁྱབ་བོ།
[4472] །དེ་ཡང་བཅུད་ཀྱི་རང་བཞིན་ནི།
[4473] །སྙིང་པོས་བསྡུས་པས་དངས་མ་གསུམ།
[4474] །གསལ་བ་གསུམ་དང་ཤེས་པས་ཁྱབ།
[4475] །གདེང་གསུམ་མི་ལྡོག་གསུམ་གྱིས་ནི།
[4476] །བཅུད་མི་འགྱུར་བའི་ངེས་པ་ཐོབ།
[4477] །སྣང་བ་གསུམ་དང་སྟོངས་པ་གསུམ།
[4478] །བཅུད་ནི་རང་བཞིན་དག་པ་ཡི།
[4479] །རྒྱུད་དུ་གཏན་ལ་ཕབ་པའོ།
[4480] །གཞི་དང་ལམ་དང་འབྲས་བུ་ཡིས།
[4481] །གཟུང་བར་ངེས་པས་རང་བཞིན་རྒྱུད།
[4482] །དེ་ཕྱིར་རྒྱུད་ཅེས་འགྲེལ་པས་གྲུབ།
[4483] །རང་བཞིན་བབས་ཀྱིས་གྲུབ་པས་ན།
[4484] །རང་བཞིན་ཉིད་ཀྱི་རྒྱུད་ཅེས་བྱའོ།
[4485] །ཞེས་སོ།
[4486] །དེ་ལྟར་གཞི་རིག་པ་གཅིག་གི་རང་བཞིན་ལས།
[4487] །མཚོན་བྱ་དང་མཚོན་བྱེད་ཐབས་ཀྱི་རྒྱུད་གཉིས།
