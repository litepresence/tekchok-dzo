[7075] །བསྒོམས་པས་ཆོས་ཀྱི་སྐུ་མཐོང་ན།
[7076] །རང་སྣང་ཆོས་ཉིད་བརྫུན་པར་འགྱུར།
[7077] །བལྟས་པ་རིག་པའི་དོན་མི་རྟོགས།

[7085] སྣང་ཚུལ་ཅིར་སྣང་ཡང་རོལ་པ་འགག་མེད་དུ་ཤར་བས།

[7464] ཀློང་སྡེ་ལྟར་དེ་ཉིད་ཆོས་ཉིད་དུ་བྱེད་པས་དག་པ་ལྟ་བུ་མ་ཡིན་ནོ།
[7465] །ཅིའི་ཕྱིར་ཞེ་ན།
[7466] ཉོན་མོངས་པ་དེ་ཡེ་ཇི་བཞིན་ཡོད་མ་མྱོང་དུ་མ་ཤེས་ན།
[7467] སྤངས་པས་མི་སྤོང་སྟེ།
[7468] ཉོན་མོངས་དེ་སེམས་ཀྱིས་སྤོངས་ན།
[7469] སྤང་བྱ་སྤོང་བྱེད་གཉིས་རྫས་གཅིག་པས་དག་པར་མི་རུང་ལ།
[7470] ཡེ་ཤེས་ཀྱིས་སྤོང་ན་སེམས་ཀྱིས་ཡེ་ཤེས་མ་མཐོང༌།
[7471] ཡེ་ཤེས་ཀྱིས་སེམས་མ་མཐོང་བས་ལྷན་ཅིག་མི་གནས་པའི་དངོས་འགལ་ལ་སྤང་བྱ་སྤོང་བྱེད་ཀྱི་ཐ་སྙད་མི་རུང་ལ།
[7472] སྔ་མ་ཕྱི་མས་སྤངས་པ་མེད་དེ་འདས་མ་འོངས་རྫས་འགལ་བས།
[7473] གཅིག་ཡོད་དུས་གཅིག་འགགས་ལ།
[7474] དུས་མཉམ་ན་ཕན་ཚུན་འཆོལ་བས་སྤང་བྱ་གཉེན་པོར་ཐལ་ལ།
[7475] གཉེན་པོ་སྤང་བྱར་ཐལ་ཏེ།
[7476] སེམས་རྒྱུན་གཅིག་དུས་མཉམ་པའི་ཕྱིར་རོ།
[7477] །ཡང་སེམས་དེ་ཉིད་ལས་ཉོན་མོངས་འབྱུང་བ་ལུས་དང་གྲིབ་མ་ལྟ་བུ་ལ་སྤངས་པས་སྤོང་བའི་དུས་མེད་དོ།
[7478] །དེ་བཞིན་དུ་སྦྱངས་པ་ལ་སོགས་པ་ལ་ཡང་སྦྱོར་ཏེ།
[7479] ཉི་ཟླ་ཁ་སྦྱོར་ལས།
[7480] དེ་བཞིན་ཉིད་དུ་མ་རྟོགས་ན།
[7481] །རྣལ་འབྱོར་ལུས་དང་གྲིབ་མ་བཞིན།
[7482] །དེ་ལས་དེ་ཉིད་རང་འབྱུང་བས།
[7483] །སྤངས་པས་སྤོང་བ་མ་ཡིན་ནོ།
[7484] །དེ་བཞིན་སྦྱངས་པས་ཀྱང་ནི་མིན།
[7485] །བསྒྱུར་བས་ཡང་ནི་མི་འགྱུར་རོ།
[7486] །འགྱུར་མེད་་སྙིང་པོ་མ་རྟོགས་པར།
[7487] །ཤེལ་གྱི་བྲག་གམ་གཡུ་རྡོ་བཞིན།
[7488] །སྦྱང་ཞིང་བཞུ་བར་ནུས་པ་མིན།
[7489] །རང་བཞིན་གྱིས་ནི་གནས་དག་ཏུ།
[7490] །རྣལ་འབྱོར་ལྡན་པ་ཡིས་ནི་བརྟག
