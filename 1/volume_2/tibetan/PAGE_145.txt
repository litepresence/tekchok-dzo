[6139] སྣང་བ་འཁྲུལ་པའི་རང་བཞིན་འདི།
[6140] །ཡེ་ཤེས་ཡིན་པར་སྔར་མ་མཐོང༌།
[6141] །རྟོག་བཅས་འཁྲུལ་པའི་གཞི་མེད་པས།
[6142] །ཀུན་གཞི་ཆོས་སྐུ་ཆེན་པོར་རྟོགས།
[6143] །འཁྲུལ་པའི་རང་རྟོག་རྒྱུན་ཆད་དོ།
[6144] །དེ་ལྟར་འཁྲུལ་མེད་སྣང་བ་ནི།
[6145] །ཡེ་ནས་རལ་བའི་གནད་ལས་བྱུང༌།
[6146] །འགྲོ་བའི་ཕུང་པོ་སངས་རྒྱས་སྐུ།
[6147] །ཡེ་ནས་ཡིན་པར་སུས་མ་རིག
[6148] །ཕུང་པོ་སངས་རྒྱས་མངའ་བདག་ཏུ།
[6149] །རྩ་གནས་ཡི་གེར་གྲུབ་ལས་བྱུང༌།
[6150] །ཆུ་བུར་མིག་ལ་ཡེ་ཤེས་ནི།
[6151] །རང་སྣང་ཡིན་པར་སུས་མ་མཐོང༌།
[6152] །ཞེས་པ་དང༌།
[6153] མུ་ཏིག་ཕྲེང་བ་ལས།
[6154] རང་བཞིན་སྣང་བའི་ཆོས་རྣམས་ལ།
[6155] །དྲི་མ་མེད་པས་ཕྲ་རགས་ལོག
[6156] །གཟུང་འཛིན་མེད་པས་འཁྲུལ་པ་ལོག
[6157] །འགྱུ་བ་མེད་པས་ལས་རླུང་ལོག
[6158] །ཟང་ཐལ་ཡིན་པས་དངོས་པོ་ལོག
[6159] །གསལ་བ་ཡིན་ཕྱིར་སྟོང་པ་ལོག
[6160] །ཡེ་ནས་རྟོགས་པས་བསྐྱེད་རིམ་ལོག
[6161] །རྩོལ་བ་མེད་པས་སྤྱོད་པ་ལོག
[6162] །མ་བཙལ་བཞག་པས་བསྒོམ་པ་ལོག
[6163] །རང་གྲོལ་ཡིན་པས་སྣང་བ་ལོག
[6164] །མངོན་དུ་སྣང་བས་ལྟ་བ་ལོག
[6165] །སྒྲ་ཚིག་མེད་པས་ཐ་སྙད་ལོག
[6166] །ཡོད་མ་མྱོང་བས་ཞེན་པ་ལོག
[6167] །མེད་མ་མྱོང་བས་གཞན་ལྟོས་ལོག
[6168] །གཟུང་དུ་མེད་པས་བློས་བརྟགས་ལོག
[6169] །ཉག་གཅིག་ཡིན་པས་གྲངས་མཐའ་ལོག
[6170] །ཡོངས་ལ་ཁྱབ་པས་གཞན་བྱུང་འགགས།
[6171] །རང་ཤར་ཡི་པས་མུན་པ་སངས།
[6172] །སྙིང་པོ་ཡིན་པས་ཆོས་སྐུར་ཁྱབ།
[6173] །གསལ་བར་སྣང་བས་ལོངས་སྐུར་རྒྱས།
[6174] །མ་བུ་ཕྲད་པས་སྤྲུལ་སྐུར་བྱུང༌།
[6175] །རྟོགས་སུ་ཡོད་པས་ཐབས་སུ་བྱུང༌།
[6176] །བསམ་འདས་ཡིན་པས་དྲན་པ་སངས།
[6177] །རྟག་ཆད་མེད་པས་གྲུབ་མཐའ་རྫོགས།
[6178] །ཞེས་པ་དང༌།
[6179] ཡང༌།
[6180] ས་དང་ཕྱོགས་དང་རིས་ལས་འདས།
[6181] །ཡོད་དང་མེད་དང་ལྟར་སྣང་མེད།
[6182] །སྒྲ་ཚད་རྣམས་ཀྱི་སྤྱོད་ཡུལ་མིན།
[6183] །གཅིག་དང་གཉིས་དང་བགྲང་ཡུལ་མེད།
[6184] །སྟོང་པ་ཆེན་པོར་བབས་ཀྱིས་གྲུབ།
[6185] །ལུང་དང་རིག་པའི་སྤྱོད་ཡུལ་མིན།
[6186] །རྒྱུད་དང་མན་ངག་གིས་མི་མཚོན།
[6187] །ལྟ་དང་སྒོམ་དང་སྤྱོད་པས་ནི།
[6188] །དེ་ཡི་དོན་རྣམས་ཤེས་མི་འགྱུར།
[6189] །དེ་དོན་འབྲས་བུར་གྲུབ་པ་མེད།
[6190] །ཆོས་ཀྱིས་དེ་ཉིད་བསྒྲིབས་པར་འགྱུར།
[6191] །བསམ་པས་འགྲུབ་པ་མ་ཡིན་ཏེ།
[6192] །ཤེས་རབ་ཀྱིས་ཀྱང་དཔྱད་དུ་མེད།
[6193] །སྦྱིན་དང་ཚུལ་ཁྲིམས་བཟོད་པ་ཡིས།
[6194] །དེ་ཉིད་གནོད་དང་ཕན་འབྲས་བྲལ།
[6195] །ཐོས་དང་བསམ་དང་སྒོམ་པས་ནི།
[6196] །མཆོག་ཏུ་བརྟགས་ཀྱང་དེ་མི་མཐོང༌།
[6197] །དེ་ལ་དཀྱིལ་འཁོར་ཡོད་པ་མིན།
[6198] །ལྷ་མེད་བསྐྱེད་པའི་ས་བོན་མེད།
[6199] །སྔགས་དང་ཕྱག་རྒྱས་ཅི་ཞིག་བྱ།
[6200] །མཆོད་པ་ལ་སོགས་སྤྲོ་མི་དགོས།
[6201] །དབང་དང་དམ་ཚིག་ག་ལ་ཡོད།
[6202] །བསྙེན་པ་དང་ནི་སྒྲུབ་པ་མེད།
[6203] །འཁོར་ལོ་སྤྲོ་བ་ཡོད་པ་མིན།
[6204] །བསྲུང་བ་མེད་ཅིང་བསྲུང་ཡུལ་མེད།
[6205] །གདོན་དང་གནོད་པའི་ཡུལ་ལས་འདས།
[6206] །ཞེས་པ་དང༌།
