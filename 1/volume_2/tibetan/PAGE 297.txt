[12814] ད་ལྟ་མ་ཤི་བ་ལ་བརྩོན་འགྲུས་བསྐྱེད་ནས་སྒོམ་པ་གལ་ཆེའོ།
[12815] །དྲུག་པ་སྐྱེ་གནས་ཀྱི་རྟགས་བསྟན་པ་ལ་དྲོད་སྤྱི་བོར་ཡལ་ན་རབ་ཀ་དག་འབྲིང་བར་དོར་ཐ་མ་སྤྲུལ་སྐུར་དབུགས་འབྱིན་པར་ངེས་ལ།
[12816] སྐྱེ་བོ་ཐ་མལ་པ་རྣམས་མཐོ་རིས་སུ་སྐྱེ་བས་དབུགས་འབྱིན་ནོ།
[12817] །དེའང་འཆི་དུས་མིག་རབ་རིབ་བྱེད་ཅིང་མཚན་མར་དྲོད་འདུས་ན་བྱོལ་སོང་དུ་སྐྱེའོ།
[12818] །ལག་པ་གཡས་གཡོབ་ཅིང་དྲོད་སྤྱི་བོར་འདུས་ན་ལྷའོ།
[12819] །ཚིག་འཆོལ་ཞིང་མཆན་ཁུང་གཡས་སུ་དྲོད་འདུས་ན་ལྷ་མིན་ནོ།
[12820] །རྐང་པ་གཡོས་ས་ལ་རྡེབ་ཅིང་དྲོད་རྐང་མཐིལ་དུ་འདུས་ན་དམྱལ་བའོ།
[12821] །ལུས་སེར་ཞིང་བཀྲག་མེད་ལ་དྲོད་ཁར་འདུས་ན་ཡི་དགས་སོ།
[12822] །ཚིག་གསལ་ཞིང་དྲན་པ་མ་ཉམས་པར་དྲོད་མིག་ཏུ་འདུས་ན་མིའི་སྐྱེ་གནས་སུ་སྐྱེའོ།
[12823] །ཡང་དེ་ཉིད་ལས།
[12824] ཕྱི་ཡི་སྣང་བ་འདི་ལྟ་བུ།
[12825] །གལ་ཏེ་དྲོད་ནི་གྱེན་ཡལ་ན།
[12826] །དབུགས་ཆེན་འབྱིན་པར་འགྱུར་བ་ཡིན།
[12827] །མིག་ནི་མི་གསལ་རབ་རིབ་ན།
[12828] །ངན་སོང་གནས་རྣམས་དག་ཏུ་སྐྱེ།
[12829] །ལག་པ་དག་ནི་གཡས་གཡོབ་ན།
[12830] །མི་དེ་ལྷ་ཡི་གནས་སུའོ།
[12831] །ཁ་ནས་ཚིག་ནི་འཆོལ་བ་ནི།
[12832] །ལྷ་མིན་གནས་སུ་དེ་བཞིན་སྐྱེ།
[12833] །རྐང་པ་གཡོན་པ་ས་ལ་རྡེབ།
[12834] །དེ་ནི་དམྱལ་བའི་གནས་སུའོ།
[12835] །ལུས་ཀྱི་མདོག་ནི་སེར་བཀྲག་མེད།
[12836] །དེ་ནི་ཡི་དགས་ཡུལ་དུའོ།
[12837] །ཁ་ནས་སེམས་ཅན་སྐད་འདོན་ན།
[12838] །དེ་ནི་བྱོལ་སོང་གནས་སུའོ།
[12839] །ཡང་ན་ཁ་ལྐུགས་སོ་འཐམས་ན།
[12840] །དེ་བཞིན་བྱོལ་སོང་སྐྱེ་གནས་སོ།
[12841] །ཚིག་གསལ་དྲན་པ་མ་ཉམས་ན།
[12842] །དེ་ནི་མི་ཡི་གནས་སུའོ།
[12843] །ཕྱི་ཡི་རྟགས་ནི་དེ་དག་དང༌།
[12844] །ནང་གི་རྟགས་ནི་འདི་དག་སྦྱར།
[12845] །སྤྱི་བོར་དྲོད་རྣམས་འདུས་པ་ན།
[12846] །དེ་ནི་ལྷ་ཡི་གནས་སུའོ།
[12847] །དེ་བཞིན་རྐང་མཐིལ་དྲོད་འདུས་ན།
[12848] །དེ་ནི་དམྱལ་བའི་གནས་སུའོ།
[12849] །མཆན་ཁུང་གཡས་སུ་དྲོད་འདུས་ན།
[12850] །དེ་ནི་ལྷ་མིན་ཡུལ་དུའོ།
[12851] །དེ་བཞིན་ཁར་ནི་དྲོད་འདུས་ན།
[12852] །མི་དེ་ཡི་དགས་གནས་སུ་སྐྱེ།
[12853] །དེ་བཞིན་མཚན་མར་དྲོད་འདུས་ན།
[12854] །དེ་ནི་བྱོལ་སོང་སྐྱེ་གནས་སོ།
[12855] །མིག་ཏུ་དྲོད་རྣམས་འདུས་པ་ན།
[12856] །མི་ཡི་གནས་སུ་སྐྱེ་བ་ཡིན།
[12857] །དེ་ལྟར་རྟགས་དང་སུ་ལྡན་པ།
[12858] །འདི་ནི་ངེས་པ་ཁོ་ནའོ།
[12859] །ཞེས་སོ།
[12860] །དེ་དག་ལས་ཁྱད་པར་དུ་བྱོལ་སོང་དུ་སྐྱེ་ན་མཚན་མར་དྲོད་འདུས་པ་དང༌།
