[4193] །དེ་ཡང་ཐོག་མ་ལུས་ཆགས་པའི་དུས་ན་ངག་དང་སེམས་གཉིས་དེ་ལ་བརྟེན་ནས་ཆགས་པས།
[4194] ཐོག་མར་ལུས་ཀྱི་སྤྱོད་པ་བྱ་དགོས་སོ།
[4195] །ལུས་ཀྱི་སྤྱོད་པ་བྱས་པའི་ཕྱིས་ལུས་ཀྱི་ལས་ཟད་ནས་བྱེད་པ་ལ་མི་སྤྲོ་བ་དང་ལུས་ཀྱི་སྡིག་པ་དག་པའི་དགོས་པ་ཡོད་དོ།
[4196] །འཁོར་ལོ་རྣམ་པར་བརྩེགས་པ་ལས།
[4197] ལུས་ཀྱི་ལུ་ཤན་སུས་ཕྱེད་པ།
[4198] །འདིས་ནི་འཁོར་བའི་བག་ཆགས་འབྱོངས།
[4199] །ཞེས་སོ།
[4200] །ལུས་རུ་ཤན་འབྱེད་པ་ལ་ཉི་མ་བདུན་ནམ།
[4201] བཅུ་གཅིག་གམ།
[4202] ཟླ་བ་ཕྱེད་དུ་བྱས་ན།
[4203] ཕྱི་ལུས་ཀྱི་ཞེན་པ་རང་ལོག་པ་དང༌།
[4204] ནང་ལུས་ཞི་དུལ་དང་ལྡན་པ་དང༌།
[4205] གསང་བ་སྒྱུ་ལུས་འོད་གསལ་དུ་གྲོལ་ངེས་སོ།
[4206] །དེ་ནས་རྣལ་དུ་འབེབས་པ་ལ་ཞག་གསུམ་མམ།
[4207] །བཅུ་གཅིག་གམ།
[4208] བདུན་དུ་བྱས་པས་ཕུང་པོ་ལ་སོགས་པ་རྒྱས་ཤིང་ལུས་སྤྲུལ་པའི་སྐུ་ལ་སྦྱོར་ཏེ།
[4209] གསེར་གྱི་སྙིང་པོའི་རྒྱུད་ལས།
[4210] རྣལ་དུ་འབེབས་པར་འདོད་པ་ཡིས།
[4211] །ལུས་ཀྱི་བྱ་བྱེད་དལ་བ་བསྟེན།
[4212] །འདིས་ནི་སངས་རྒྱས་སྐུ་དང་རེག
[4213] །ཁྱད་པར་མཆོག་ཏུ་གསང་བས་འགྲུབ།
[4214] །ཅེས་སོ།
[4215] །གཉིས་པ་ངག་རུ་ཤན་འབྱེད་པའི་སྤྱོད་པ་ནི།
[4216] གསང་བ་སྤྱོད་པ་ས་བོན་གྱི་རྒྱུད་ལས།
[4217] དེ་འདས་ངག་གི་སྤྱོད་པ་བརྩམ།
[4218] །སྔགས་དང་ཁ་དོན་ཆོས་ཀྱི་སྒྲ།
[4219] །ཡོད་དང་མེད་དང་སྣང་བ་དང༌།

[4221] །བྱོལ་སོང་སྣ་ཚོགས་བརྡ་སྐད་དང༌།

[4224] །ངག་གི་མངོན་ཞེན་རང་ཆོད་ནས།
[4225] །འཁོར་འདས་རུ་ཤན་འདིས་ཕྱེད་དོ།

[4230] །བརྗོད་པ་རྟོགས་པའི་རང་བཞིན་ཕྱིར།
[4231] །སྒྲ་སྐད་རྣམ་པ་སྣ་ཚོགས་པ།
