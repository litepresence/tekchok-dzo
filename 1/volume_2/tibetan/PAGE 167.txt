[7085] སྣང་ཚུལ་ཅིར་སྣང་ཡང་རོལ་པ་འགག་མེད་དུ་ཤར་བས།
[7086] རང་བྱུང་དོན་གྱི་དཀྱིལ་འཁོར་དུ་ཤར་ཏེ།
[7087] བཟའ་བཏུང་ཆོས་ཉིད་ཀྱི་བསྙེན་པ།
[7088] ཉལ་འདུག་བྱང་ཆུབ་ཀྱི་བསྒྲུབ་པ།
[7089] འདུ་བྱེད་ལྷག་གནས་ཀྱི་བསམ་གཏན།
[7090] གླལ་ལུད་དཀྱིལ་འཁོར་གྱི་སྒྲུབ་པ།
[7091] སྟོང་གསུམ་ལྷུན་གྲུབ་ཀྱི་དཀྱིལ་འཁོར།
[7092] །ཆར་ཆུ་རང་བབས་ཀྱི་ཆག་ཆག
[7093] །ལམ་སྲང་རང་བཞིན་གྱི་ཐིག་ཆེན།
[7094] རྐང་རྗེས་ཁ་དོག་གི་རྡུལ་ཚོན།
[7095] སྒུལ་སྐྱོད་ཕྱག་རྒྱའི་རང་བཞིན།
[7096] སྨྲ་བརྗོད་རྡོ་རྗེའི་བཟླས་པ།
[7097] དྲན་བསམ་བསྐྱེད་རིམ་གྱི་འཁོར་ལོ།
[7098] གང་སྣང་ལྷའི་རོལ་པ།
[7099] རང་ལུས་དབང་གི་བུམ་པ།
[7100] རབ་བབས་རྒྱུན་གྱི་དམ་ཚིག
[7101] ཤེས་རིག་རང་གསལ་གྱི་གདམས་ངག
[7102] སྐྱེ་ཤི་ཆོས་ཉིད་ཀྱི་གོམས་ཚད།
[7103] ཚོགས་དྲུག་ཡུལ་སྣང་གི་རྟོགས་པ་རང་ཤར་བས།
[7104] ཁམས་གསུམ་ཡོངས་གྲོལ་ཆེན་པོའི་དགོངས་པར་ལྷུན་གྱིས་གྲུབ་པ་ནི།
[7105] མུ་ཏིག་ཕྲེང་བ་ལས།
[7106] ཨེ་མ་དེ་ལྟར་སྣང་བ་རྣམས།
[7107] །འཁྲུལ་པ་ཉིད་ཀྱང་རང་གྲོལ་ན།
[7108] །འཁྲུལ་པ་མེད་པ་ཅིས་མི་གྲོལ།
[7109] །ཇི་ལྟར་སྨྲ་དང་བྱ་བྱེད་རྣམས།
[7110] །སྟོང་རིག་གསལ་བའི་སྤྱོད་པ་ཡིན།
[7111] །བཟང་དང་ངན་པའི་རྟོག་པ་རྣམས།
[7112] །སྒོམ་པ་ཆུ་བོའི་ཀློང་ཆེན་ཡིན།
[7113] །ལོག་ལྟ་ཡང་དག་འདོད་པ་རྣམས།
[7114] །རྣལ་འབྱོར་རིས་མེད་ལྟ་བ་ཡིན།
[7115] །རེ་དང་དོགས་པར་འཛིན་པ་ཀུན།
[7116] །ཟང་མ་ཐལ་བྱུང་འབྲས་བུ་ཡིན།
[7117] །ཕྱག་རྒྱ་ཆེན་པོ་མངོན་འདོད་པས།
[7118] །བཟའ་དང་བཏུང་བ་བསྙེན་པ་ལ།
[7119] །ཉལ་དང་འདུག་པ་བསྒྲུབ་པའོ།
[7120] །འདུ་བྱེད་རྣམས་ནི་ལྷག་གནས་ཡིན།
[7121] །གླལ་དང་ལུད་པ་དཀྱིལ་འཁོར་བསྒྲུབ།
[7122] །སྟོང་གསུམ་ཐམས་ཅད་དཀྱིལ་འཁོར་ལ།
[7123] །ཆར་དང་ཁུ་རླངས་ཆག་ཆག་བཏབ།
[7124] །འགྲོ་བའི་ལམ་དམིགས་ཐིག་ཆེན་ཏེ།
[7125] །རྐང་པའི་རྗེས་ནི་རྡུལ་ཚོན་ནི།
[7126] །འགྲོ་བར་འདོད་པ་སྟངས་སྟབས་ཉིད།
[7127] །ཡན་ལག་བསྐྱོད་པ་ཕྱག་རྒྱའོ།
[7128] །ཇི་ལྟར་སྨྲས་པ་སྔགས་ཀྱི་ཚིག
[7129] །བསམ་པ་རྣམས་ནི་བསྐྱེད་རིམ་ཡིན།
[7130] །ཡིད་ལ་འགྱུ་བ་མཆོད་པ་ཉིད།
[7131] །གཟུགས་སུ་སྣང་བ་ལྷ་ཡི་སྐུ།
[7132] །སྒྲ་ཆེན་བརྗོད་པ་རོལ་མོ་ཉིད།
[7133] །རང་གི་ལུས་ནི་བུམ་པ་ལ།
[7134] །སྐྲ་ནི་ལོ་འདབ་ཁ་རྒྱན་མཛེས།
[7135] །ཁྲག་དང་ཆུ་སེར་ཆུ་རྣམས་ཉིད།
[7136] །དོན་སྙིང་ལ་སོགས་རྫས་ཀྱིས་གཏམས།
[7137] །རང་ལ་སྣང་བའི་དབང་རྣམས་བྱིན།
[7138] །མ་བསྐུར་ཉིད་ལ་རྫོགས་པའོ།
[7139] །འདའ་བར་འདོད་པ་དམ་ཚིག་ལ།
[7140] །བསྲུང་བར་འདོད་པ་བཅིངས་པ་ཉིད།
[7141] །གྲོལ་བར་འདོད་པ་ཉམས་པ་སྟེ།
[7142] །མེད་པར་འདོད་པ་ཐུབ་མཆོག་ཡིན།
[7143] །མཐོང་བ་རྣམས་ནི་ཕྱག་རྒྱའི་གདའ།
[7144] །དེ་ལ་ཆགས་པ་ཉམས་མྱོང་ཡིན།
[7145] །ཤེས་རིག་གསལ་བ་གདམས་ངག་ལ།
[7146] །ཡུལ་སེམས་གཉིས་སུ་གདབ་པའི་ཡུལ།
[7147] །སྐྱེ་རྒ་ན་འཆི་གོམས་པའི་གནད།
[7148] །ཚོགས་དྲུག་མ་འགགས་རྟོགས་པའོ།
[7149] །ཞེས་སོ།
[7150] །དེ་ལྟར་རྣལ་འབྱོར་པས་ཅི་བྱས་ཆོས་ཉིད་ཀྱི་ངང་དུ་བསྐྱེད་པ་ལྷའི་འཁོར་ལོ་དང་དབང་རྫོགས་གདམས་ངག་དང་བཅས་པ།
