[1546] མ་བྲིས་པར་ཡེ་ནས་བཀྲ་བ་རང་རྫོགས་རང་ལ་ཡོད་པའོ།
[1547] །ཕྱག་རྒྱ་ནི།
[1548] མ་བསྒྱུར་གཞི་ནས་རྫོགས་པ་བྱང་ཆུབ་མཆོག་ལ་སོགས་པ་སྐུ་ལྔ་ལ་སྣང་བའོ།
[1549] །ལྷ་ཚོགས་ནི།
[1550] མ་བསྐྱེད་ཡེ་རྫོགས་ཀྱི་ཞི་ཁྲོ་རང་ལ་གནས་པའོ།
[1551] །སྔགས་ཀྱི་སྙིང་པོ་ནི།
[1552] རྩ་གནས་ཡི་གེའི་འཁོར་ལོ་དང་དེའི་དངས་མ་སྒྲོན་མ་བཞི་ཡུལ་ལ་ཤར་བའོ།
[1553] །དེ་ཡང་རང་ཤར་ལས།
[1554] འཕྲིན་ལས་ཀུན་གྱི་དཀྱིལ་འཁོར་ནི།
[1555] །མ་བཙལ་རང་ལ་ཡོད་པའོ།
[1556] །འཕྲིན་ལས་ཀུན་གྱི་ཕྱག་རྒྱ་ནི།
[1557] །མ་བསྒྱུར་བཞི་ལ་རྫོགས་པའོ།
[1558] །འཕྲིན་ལས་ཀུན་གྱི་ལྷ་ཚོགས་ནི།
[1559] །རང་གི་ལུས་ལ་རྫོགས་པའོ།
[1560] །འཕྲིན་ལས་ཀུན་གྱི་སྙིང་པོ་ནི།
[1561] །སྒྲོན་མའི་ཡེ་ཤེས་ཆེན་པོར་གསུངས།
[1562] །ཞེས་སོ།
[1563] །གཉིས་པ་རང་བཞིན་རྫོགས་ཚུལ་གྱི་རིམ་པ་རྒྱས་པར་བཤད་པ་ལ་གཉིས་ཏེ།
[1564] གདོད་མའི་གཞི་ལ་རྫོགས་པའི་ཚུལ་དང༌།
[1565] སེམས་ཅན་གྱི་དུས་ན་རྫོགས་པའི་ཚུལ་ལོ།
[1566] །དང་པོ་ལ་གཉིས་ལས།
[1567] སྤྱིར་བསྟན་པ་ནི།
[1568] གཞི་གདོད་མའི་ཀ་དག་ལ་ནང་ཡེ་ཤེས་ཀྱི་འོད་ལྔ་དབྱིངས་རུམ་དུ་གསལ་བའི་མཐིང་ག་ཐམས་ཅད་ཀྱི་དབྱིངས་ཡིན་པས།
[1569] འཕྲིན་ལས་ཀྱི་འོད་བཞིའི་འཆར་གཞིར་གནས་པས་མི་བགྲང་ཞིང༌།
[1570] དཀར་སེར་དམར་ལྗང་བཞི་དེ་ལས་མཆེད་པའི་འཕྲིན་ལས་སུ་བཤད་དེ།
[1571] དེ་ལས།
[1572] དེ་ཡང་འཕྲིན་ལས་བཞི་ཡི་དོན།
[1573] མཐིང་ག་ཉིད་ལ་རྫོགས་པ་སྟེ།
[1574] །འཕྲིན་ལས་རྣམ་བཞི་གཞི་ལ་གྲུབ།
[1575] །དེ་ཕྱིར་ཁ་དོག་མཐིང་གར་བཤད།
[1576] །ཅེས་པ་དང༌།
[1577] འཕྲིན་ལས་རྣམ་པ་བཞི་ཡི་དོན།
[1578] །གཞི་ལ་ལེགས་པར་ཚང་བའོ།
[1579] །དེ་ཡང་བསྟན་པ་འདི་ལྟ་སྟེ།
[1580] །ཁ་དོག་དཀར་དམར་སྣ་ཚོགས་འོད།
[1581] །གཞི་ལ་བབས་ཀྱིས་གྲུབ་པའོ།
[1582] །དཀར་པོ་ཞི་བའི་དཀྱིལ་འཁོར་ལྡན།
[1583] །ཁ་དོག་སེར་པོ་རྒྱས་པའི་འོད།
[1584] །དམར་པོ་དབང་གི་འོད་དུ་འདོད།
[1585] །ལྗང་གུ་དྲག་པོའི་འོད་དུ་གསུངས།
[1586] །མཐིང་ག་འཕྲིན་ལས་རྫོགས་པའི་འོད།
[1587] །ཅེས་སོ།
[1588] །གཉིས་པ་བྱེ་བྲག་ཏུ་བཤད་པ་ལ་བཞི་ལས།
[1589] དང་པོ་ཞི་བ་ནི།
[1590] དེ་ལས།
[1591] དཀར་པོ་ཆོས་ཉིད་ཞི་བའོ།
[1592] །ཉོན་མོངས་དྲི་མ་ག་ལ་ཡོད།
[1593] །གོས་པ་མེད་པའི་དཀྱིལ་འཁོར་ནི།
[1594] །དཔེར་ན་ཤེལ་གོང་དག་པ་བཞིན།
[1595] །དྲི་མའི་སྐྱོན་གྱིས་མ་གོས་པའོ།
[1596] །དེ་བཞིན་ཞི་བའི་ཆོས་ཉིད་ལ།
[1597] །ཉོན་མོངས་གོས་པ་མེད་པའོ།
[1598] །ཞེས་སོ།
[1599] །གཉིས་པ་རྒྱས་པ་ནི།
