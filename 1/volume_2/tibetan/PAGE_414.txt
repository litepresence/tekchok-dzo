[17263] །ཇི་ལྟར་མཚན་མོ་དངས་པའི་ནམ་མཁའ་ལ།
[17264] །རི་བོང་འཛིན་ཟེར་ཅན་ཡང་འོད་འགྱེད་ཅིང༌།
[17265] །སྐར་ཚོགས་དབུས་ན་དཀྱིལ་འཁོར་གང་བ་ཡིས།
[17266] །མེ་ཏོག་ཀནྡའི་ཚལ་རྣམས་རྒྱས་པ་བཞིན།
[17267] །བློ་གྲོས་མཁའ་ལ་སྙིང་པོའི་འོད་བརྒྱ་པ།
[17268] །རྣམ་བཀྲ་ཡེ་ཤེས་གསལ་བའི་མདངས་འཕྲོ་བས།
[17269] །རྡོ་རྗེ་སྙིང་པོའི་ལམ་བཟངས་སྣང་བྱེད་ཅིང༌།
[17270] །སྐལ་བཟངས་བློ་ཡི་པདྨ་རྒྱས་བྱེད་ཡིན།
[17271] །ཚུལ་འདི་ཐེག་མཆོག་སྙིང་པོའི་པད་ཚལ་ལས།
[17272] །རབ་འཁྲུངས་སྣ་ཚོགས་རང་གྲོལ་རྣལ་འབྱོར་པས།
[17273] །འོད་གསལ་སྙིང་པོའི་དོན་རྣམས་བསྟན་ནས་སོང༌།
[17274] །དྲི་མེད་འོད་ཟེར་སྟོང་ལྡན་དེ་སྐད་དོ།
[17275] །རང་བཞིན་གསང་བའི་གནད་རྣམས་མ་ལུས་པ།
[17276] །སྐལ་ལྡན་རྣམས་ལ་མ་ནོར་མངོན་སུམ་དུ།
[17277] །ཀློང་ཆེན་རབ་འབྱམས་བཟང་པོས་བསྟན་ནས་སོང༌།
[17278] །ཚུལ་ཁྲིམས་བློ་གྲོས་བཟང་པོ་དེ་སྐད་ལོ།
[17279] །ད་ལྟར་བཞུགས་དང་སླད་ཀྱིས་འབྱོན་པ་ཡི།
[17280] །ཕྱི་རབས་རྗེས་སུ་འཇུག་པའི་སྐལ་ལྡན་ལ།
[17281] །མཐར་ཐུག་དོན་གྱི་གནད་རྣམས་གདམས་ཟིན་གྱིས།
[17282] །བརྩོན་པས་ཉམས་སུ་ལེན་ལ་འབད་པར་མཛོད།
[17283] །གསང་བའི་གནས་ཀྱང་ཤིན་ཏུ་རྒྱ་ཆེ་ལ།
[17284] །རྒྱུད་ལུང་མན་ངག་ཡི་གེ་གྲངས་མང་ཞིང༌།
[17285] །རྡོ་རྗེ་གནས་ཀྱང་བརླིང་ཞིང་རྟོགས་དཀའ་བས།
[17286] །དེ་དག་དོན་འདུས་ཐེག་མཆོག་མཛོད་འདི་གཅེས།
[17287] །དཔལ་མགོན་བླ་མ་དམ་པ་མཉེས་བྱས་ཤིང༌།
[17288] །ཟབ་མོའི་གནས་རྣམས་མ་ནོར་འདིར་ཚང་ལ།
[17289] །ལུང་རིགས་མན་ངག་རྒྱ་མཚོའི་ཀློང་ཡངས་པས།
[17290] །ཟབ་ཅིང་རྒྱ་ཆེའི་གནས་འདི་མཆོག་ཏུ་འཕགས།
[17291] །ཕྱིན་ཆད་སྐལ་མེད་རྣམས་ལ་མི་བསྟན་ཅིང༌།
[17292] །ཐུན་མོང་མ་ཡིན་གྲུབ་མཐའ་གང་ཡིན་པ།
[17293] །དེ་ཀུན་ཚོགས་སུ་མ་གླེང་དམ་དུ་སྦོས།
[17294] །ཡི་གེ་འདི་ལ་བཅོས་བསླད་མ་བྱེད་ཅིག
[17295] །མཛོད་འདི་དཔལ་ལྡན་སྔགས་ཀྱི་བདག་མོ་དང༌།
[17296] །གཟའ་ཡི་དགེ་བསྙེན་ཆེན་པོ་རཱ་ཧུ་ལ།
[17297] །དམ་ཅན་རྡོ་རྗེ་ལེགས་པ་སྐྱོང་ལ་གཏད།
[17298] །ཐུབ་པར་སྲུངས་ཤིག་བཀའ་ཡི་བྱ་ར་གྱིས།
[17299] །འདི་ལ་རྩོམ་གཞིའི་ཡེ་གེ་ཐུ་བ་དང༌།
[17300] །མང་ཉུང་འདྲ་མིན་བསླེ་བསླད་བྱུང་སྲིད་ན།
[17301] །དེ་ཡི་སྙིང་ཁྲག་རོལ་པའི་དགའ་སྟོན་མཛོད།
[17302] །དབུགས་ཆོད་མདངས་ཕྲོགས་མིག་འབྲས་ཐང་ལ་སྦྲུལ།
[17303] །རིགས་འཛིན་བཀའ་སྡོད་རྒྱ་མཚོའི་ཚོགས་རྣམས་ཀྱིས།
[17304] །འདི་ལ་སྤྱོད་པའི་སྐལ་ལྡན་བྱིན་གྱིས་རློབས།
[17305] །དེ་ཡི་འདོད་དགུ་ཡིད་བཞིན་རྫོགས་མཛོད་ལ།
[17306] །གདོད་མའི་གནས་སུ་མངོན་པར་བྱང་ཆུབ་ཤོག
