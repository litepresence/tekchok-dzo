[57] ཕྱི་འབྱུང་རྒྱས་པར་བཤད་པ་དང༌།
[58] ནང་འབྱུང་རྒྱས་པར་བཤད་པའོ།
[59] །དང་པོ་ལ།
[60] ངོ་བོ།
[61] ངེས་ཚིག
[62] དགོས་ཆེད།
[63] མཚན་ཉིད།
[64] ལས་རིམ།
[65] ཆོས་ཉིད།
[66] དོན་སྦྱར།
[67] གྲོལ་ཚུལ།
[68] དབྱེ་བ།
[69] ཚང་ཚུལ་དང་བཅུ་ལས།
[70] དང་པོ་ངོ་བོ་ནི།
[71] ཕྱི་འབྱུང་གཟུགས་སུ་སྣང་བ་ནང་འབྱུང་དངོས་ཀྱི་རོལ་པ་ཡིན་པས།
[72] ནང་གི་ཁོངས་སུ་འདུ་བས་ན་འབྱུང་བ་ལྔ་སྟེ།
[73] རང་ཤར་ལས།
[74] ཕྱི་ཡི་འབྱུང་བ་འདི་ལྟ་སྟེ།
[75] །ས་དང་ཆུ་དང་མེ་དང་ནི།
[76] །དེ་བཞིན་རླུང་དང་ནམ་མཁའ་འོ།
[77] །ཞེས་པས།
[78] ཕྱི་འབྱུང་གཟུགས་ལྔ་ནི་སེམས་ཅན་གྱི་རྟེན་སྣོད་ཀྱི་སྣང་བར་ཆགས་ལ།
[79] ནང་འབྱུང་དངོས་ལྔ་ནི་སེམས་དང་རིག་པའི་རྟེན་དུ་ཆགས་པ་སྟེ།
[80] ཐལ་འགྱུར་ལས།
[81] འབྱུང་བའི་འདོད་པ་འདི་ལྟ་སྟེ།
[82] །སྤྱི་མཐུན་ལས་ཀྱི་བྱེ་བྲག་ལས།
[83] །སེམས་ཅན་ལས་ཀྱི་རྟེན་སོ་ཡིས།
[84] །ཕྱི་ཡི་འབྱུང་བ་ལ་བརྟེན་ནས།
[85] །ལས་དང་གསོ་དང་དབང་པོ་དང༌།
[86] །འཁོར་དང་སྐྱེ་བ་རྫོགས་པར་འདོད།
[87] །ཅེས་སོ།
[88] །གཉིས་པ་ངེས་ཚིག་ནི།
[89] རང་ཤར་ལས།
[90] འབྱུང་བའི་ངེས་ཚིག་འདི་ལྟ་སྟེ།
[91] །བྱས་པས་མ་བྱུང་ལྷུན་གྱིས་གྲུབ།
[92] །སྐྱེད་པར་བྱེད་པས་འབྱུང་བ་ཡིན།
[93] །འགྲོ་བ་ཡོངས་ལ་ཁྱབ་པས་འབྱུང་བ་ཆེན་པོ་ཡིན།
[94] །ཅིག་ཅར་སྐྱེ་བས་འབྱུང་བ་ཆེན་པོ་ཡིན།
[95] །རང་ལ་ཡོད་པས་འབྱུང་བ་ཆེན་པོ་ཡིན།
[96] །འགྱུར་བ་མེད་པས་འབྱུང་བ་ཆེན་པོ་ཡིན།
[97] །མཐུན་པར་སྣང་བས་འབྱུང་བ་ཆེན་པོ་ཡིན།
[98] །འབྱུང་བ་ཆེན་པོའི་ངེས་ཚིག་དེ་བཞིན་ནོ།
[99] །ཞེས་སོ།
[100] །གསུམ་པ་དགོས་ཆེད་ནི།
[101] ཐལ་འགྱུར་ལས།
[102] འབྱུང་བའི་དོགས་པ་འདི་ལྟ་སྟེ།
[103] །ཆུས་ནི་དྭངས་མ་སྡུད་པ་དང༌།
[104] །སྙིགས་མ་རྣམས་ནི་འབྱེད་པའི་ལས།
[105] །སོ་སོའི་ལུས་ལ་བྱེད་པས་ན།
[106] །དགོས་པ་ཚ་དང་གྲང་ངོ་ཤེས།
[107] །སས་ནི་གཞི་མ་འཛིན་པ་དང༌།
[108] །གྲུབ་ཅིང་ཆགས་པའི་ལས་རྣམས་བྱེད།
[109] །སྨིན་པས་རྩ་ལ་སོགས་པ་སྐྱེད།
[110] །དགོས་པ་ཀུན་རྫོབ་ཐིག་ལེ་ཤེས།
[111] །མེས་ནི་འབྱུང་བ་སྤེལ་བ་དང༌།
[112] །སྨིན་ཅིང་གསལ་བའི་ལས་རྣམས་བྱེད།
[113] །འབར་ཞིང་མཆེད་པས་མིག་སོགས་སྐྱེད།
[114] །དགོས་པ་དྲོད་རྣམས་ཤེས་པར་བྱེད།
[115] །རླུང་གིས་གཏོར་དང་འདེགས་པ་དང༌།
[116] །བསྐྱོད་ཅིང་གཡོ་བའི་ལས་རྣམས་བྱེད།
[117] །འགུལ་ཞིང་འཕྲིག་པས་ལུས་འཛིན་ཅིང༌།
[118] །དགོས་པ་སེམས་ཀྱི་གནས་པ་ཤེས།
[119] །འབྱུང་བ་བཞི་བདག་དཀྱིལ་འཁོར་ཆེ།
[120] །ལུས་སོགས་འགྲུབ་པའི་རྒྱུ་བྱས་ཏེ།
[121] །བར་དུ་གནས་ཤིང་འཛིན་པ་བྱས།
[122] །ཐ་མ་འཇིག་པའི་ལས་ཀུན་བྱེད།
[123] །འབྱུང་བ་ཆེན་པོ་རྒྱུ་གཅིག་པས།
[124] །སྣ་ཚོགས་བྱེད་པའི་བདག་པོར་སྣང༌།
[125] །དེས་ན་དགོངས་པ་རང་དང་གཞན།
[126] །བྱ་བ་རྫོགས་པའི་དོན་ལ་མཁས།
[127] །ཞེས་སོ།
[128] །བཞི་པ་མཚན་ཉིད་ནི་ལྔ་ལས།
[129] །སའི་མཚན་ཉིད་སྲ་བ།
