[15378] །ཁ་དོག་ལུས་རྫོགས་ཟས་གོས་དུས་གཅིག་རྫོགས།
[15379] །ལུས་ཀྱི་རྩོལ་བྲལ་རླུང་བསྐྱོད་འགྲོ་དང་འདུག

[15381] །ཡན་ལག་བརྒྱད་ལྡན་རོ་བརྒྱ་འདུས་པའི་ཆུ།
[15382] །འཐུངས་པས་ཚིམ་བྱེད་རིག་ན་ལུས་ཀྱང་འཇམ།
[15383] །ས་ཆུ་མེ་རླུང་འབྱུང་བས་ཕན་གནོད་བྲལ།
[15384] །བཀྲེས་དང་སྐོམ་དང་ཚ་གྲང་སྡུག་བསྔལ་སེལ།
[15385] །འདོད་ཆགས་ཞེ་སྡང་གཏི་མུག་ཉོན་མོངས་རྣམས།
[15386] །སྐྱེ་བའི་རྒྱུ་མེད་ཤེས་པ་བག་ཡངས་སོ།
[15387] །བལྟས་དགོད་ལག་བཅངས་གཉིས་འཁྱུད་འདོད་ཆགས་མེད།
[15388] །ཆོག་ཤེས་སྲེད་བྲལ་རྟག་ཏུ་ཆོས་ལ་ཉན།
[15389] །སྒྲིབ་གཉིས་བྲལ་ནས་སྐྱེ་བ་གཅིག་གིས་ཐོགས།
[15390] །མཆོད་པའི་ལྷ་མོས་ཡིད་འོང་བཀྲ་ཤིས་བརྗོད།
[15391] །གཞལ་ཡས་ཁང་མཛེས་སྒོ་བཞི་རྟ་བབས་ལྡན།
[15392] །ཤེལ་དང་གསེར་དང་མར་གད་ཨིནྡྲི་ནཱིལ།
[15393] །བཀྲ་ཤིས་གདུགས་དང་འཁོར་ལོ་རྒྱལ་མཚན་མཛེས།
[15394] །དེ་ལྟར་རང་བཞིན་ཕྱོགས་བཅུ་སྤྲུལ་པའི་ཞིང༌།
[15395] །རང་རང་སྟོན་པས་རང་རང་འཁོར་རྒྱུད་འདུལ།
[15396] །མདུན་གྱི་ནམ་མཁའ་མི་མཇེད་བརྩེགས་པ་ན།
[15397] །འཇིགས་པའི་ཁྲོ་གནས་དུར་ཁྲོད་མེ་རི་འབར།
[15398] །འཇིགས་པའི་སྒྲ་དང་ཁྲོ་གཟུགས་ཨ་ལ་ལ།
[15399] །འདི་ཉིད་མཐོང་བ་ཙམ་གྱིས་དེ་གྲོལ་འགྱུར།
[15400] །སྐལ་ལྡན་སྟོབས་མེད་སྙིང་པོའི་དོན་འཛིན་པ།
[15401] །དེར་ནི་དབུགས་ཕྱུང་སྐྱེ་བ་གཅིག་པ་སྟེ།
[15402] །ལྔ་བརྒྱ་དག་ན་ངེས་པའི་སར་གྲོལ་ལོ།

[15405] རིག་པ་རང་ཤར་ཆེན་པོའི་རྒྱུད་ལས།
