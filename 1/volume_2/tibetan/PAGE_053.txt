[2573] །གཉིས་འཇུག་བདེ་བ་སུ་བསྒོམ་པ།
[2574] །འདིས་ནི་འདོད་ཡོན་ལམ་དུ་བྱེད།
[2575] །བར་སྣང་ལམ་དུ་སུ་བྱེད་པ།
[2576] །འདིས་ནི་སེམས་ལ་དབང་ཡང་ཐོབ།
[2577] །ཅེས་སོ།
[2578] །འདི་དག་གི་ཞར་ལ་སྣང་བའི་ཡུལ་གཟུགས་སྒྲ་དྲི་རོ་་རེག་ལྔ་ལ་སེམས་གཟུང་བའི་ཐབས་ཀྱང༌།
[2579] ཐལ་འགྱུར་ལས།
[2580] མིག་གཟུགས་སྒྲ་དྲི་རོ་དང་རེག
[2581] །འདོད་ཡོན་ལྔ་ཡི་གནས་སྦྱངས་ཏེ།
[2582] །གཟུགས་ནི་ཁ་དགོ་དབྱིབས་རྣམས་དང༌།
[2583] །བཟོ་དང་སྒྱུ་རྩལ་ཐ་དད་དང༌།
[2584] །མིག་འཕྲུལ་ལ་སོགས་སྣ་ཚོགས་པའོ།
[2585] །སྒྲ་ནི་པི་ཝང་བུམ་ལྡིར་དང༌།
[2586] །རྒྱུད་མངས་རྫ་རྔ་གླིང་བུ་དང༌།
[2587] །ཕེག་དང་ཧར་ལ་སོགས་པའོ།
[2588] །དྲི་ནི་ངད་དང་སྦྱར་མ་དང་
[2589] །ཡིད་མཐུན་རྟེན་འབྲེལ་ཟབ་མོས་བྱ།
[2590] །རོ་ནི་བུ་རམ་དྲུག་་ལྡན་དང༌།
[2591] །རྒུན་ཆང་སྦྱར་མ་སྣ་ཚོགས་བསྒྱུར།
[2592] །རྟེན་འབྲེལ་ཡན་ལག་སྦྱོར་བའོ།
[2593] །རེག་ནི་མཁས་པས་ཁྲི་སྟེངས་འདུག
[2594] །གཞུང་བསྣོལ་སྦྱོར་བའི་འབྱུང་བ་དང༌།
[2595] །ཤིང་གི་ལོ་མ་འགའ་ཞིག་གིས།
[2596] །རེག་པ་ཙམ་གྱིས་ཆོས་ཉིད་ལ།
[2597] །འབྲལ་མེད་སེམས་ཀྱིས་གནས་པའོ།
[2598] །ཞེས་སོ།
[2599] །ཁྱད་པར་དུ་ར་བཞི་རྣལ་འབྱོར་དུ་བྱ་བའི་རིམ་པ་ཡང༌།
[2600] །དེ་ལས་གསུངས་པ།
[2601] སྐུ་གསུམ་བསླབ་པའི་རིམ་པ་ཉིད།
[2602] །འབྱུང་བའི་ཡོན་ཏན་གཙོར་བྱས་ཏེ།
[2603] །མཆོག་ཏུ་ས་ཆུ་མེ་རླུང་གི
[2604] །སྒྲ་ལ་བསླབས་པས་ངེས་པར་འགྲུབ།
[2605] །ཆུ་ཡི་སྒྲ་ནི་བཤད་པ་ལ།
[2606] །མཁའ་འགྲོ་མ་ཡི་སྒྲ་དབྱངས་འཛིན།
[2607] །འདི་ལ་རྟག་ཏུ་གོམས་བྱས་ན།
[2608] །སྤྲུལ་པའི་སྐུ་ཡང་ངེས་པར་འགྲུབ།
[2609] །ས་ཡི་སྒྲ་ནི་བསིལ་ཞིང་ལྕི།
[2610] །ཚངས་པ་ཆེན་པོའི་སྒྲ་སྐད་ལྡན།
[2611] །འདི་ལ་རྟག་ཏུ་ཉན་སྦྱོང་ན།
[2612] །ལོངས་སྤྱོད་རྫོགས་སྐུ་ངེས་འགྲུབ་བོ།
[2613] །མེ་ཡི་སྒྲ་ནི་རང་བྱེད་བསླབ།
[2614] །ཁྱབ་འཇུག་ཆེན་པོའི་གསུང་དབྱངས་སྟོན།
[2615] །འདི་ལ་ཉན་པ་སུས་སྤྱོད་པ།
[2616] །ཆོས་སྐུའི་ཡོན་ཏན་ངེས་པར་ཐོབ།
[2617] །རླུང་གི་སྒྲ་ནི་གཟིར་ཞིང་གྲག
[2618] །མཁའ་ལྡིང་རྒྱལ་པོ་སྦྱོར་བའི་གསུང༌།
[2619] །འདི་ནི་རྟག་ཏུ་ཟློས་ཤེས་ན།
[2620] །སྐུ་གསུམ་ཐུན་མོང་བསླབ་བྱའོ།
[2621] །དེ་ལྟར་འབྱུང་བཞིའི་སྒྲ་དོན་ནི།
[2622] །ཕྱི་ཡི་དུས་ལ་ངེས་པར་སྦྱར།
[2623] །དགུན་ཏེ་དཔྱིད་དང་དབྱར་དུས་སྟོན།
[2624] །ཆུ་དང་ས་སྟེ་མེ་རླུང་གིས།
[2625] །རིམ་པར་དུས་དང་ངེས་སྦྱར་ཏེ།
[2626] །རྣལ་འབྱོར་ལུས་དང་བསྟུན་བྱས་ན།
[2627] །འགྲུབ་འགྱུར་འདི་ལ་ཐེ་ཚོམ་མེད།
[2628] །ཅེས་སོ།
[2629] །དེ་ལྟར་ལྷ་དང་མིའི་སེམས་འཛིན་གྱི་དོན་བསྡུས་ན།
[2630] །རང་ཤར་ལས།
