[10608] །གཉིས་པ་ཐུན་མོང་མ་ཡིན་པ་གསང་བ་མཆོག་གི་རྟགས་མངོན་དུ་བྱ་བ་ལ་གསུམ་སྟེ།
[10609] སྔོན་འགྲོ་འབྱོངས་པ་ལུས་ངག་སེམས་ཀྱི་རྟགས་སྤྱིར་བསྟན་པ།
[10610] དངོས་གཞི་འབྱོངས་པ་འོད་གསལ་གྱི་རྟགས་བྱེ་བྲག་ཏུ་བཤད་པ།
[10611] རྟགས་ཀྱི་ཁྱད་པར་གཞན་བརྗོད་པས་ལམ་གྱི་ཚད་བཟུང་བའོ།

[10613] སྐུ་གསུམ་གྱི་སྣ་ཁྲིད་པ་འབྱུང་བཞི་སྒྲ་དོན་གྱི་སྔོན་འགྲོ་འབྱོངས་པའི་རྟགས་ནི།
[10614] སྐུ་གདུང་འབར་བ་ལས།
[10615] གང་གིས་འབྱུང་བའི་སྒྲ་རྣམས་དང༌།
[10616] །རིགས་དྲུག་སྒྲ་ལ་སུས་གོམས་པ།
[10617] །འདི་ལ་འབྱུང་བས་མི་གནོད་ཅིང༌།
[10618] །དེ་བཞིན་མེ་ཡིས་ཚིག་པ་མེད།
[10619] །ཆུ་ཡིས་བྱིང་ཞིང་ཁྱེར་བ་མེད།
[10620] །རྐང་པ་ས་ལ་མ་རེག་པར།
[10621] །བར་སྣང་ལ་ཡང་འགྲོ་བར་ནུས།
[10622] །རླུང་གིས་གྲང་ཞིང་གཡོ་བ་མེད།
[10623] །འདི་ལྟར་སུ་ལ་བྱུང་བ་ནི།
[10624] །འབྱུང་བཞིའི་སྒྲ་ལ་གོམས་པ་ཡིན།
[10625] །ཅེས་སོ།
[10626] །སེམས་ཀྱི་སྣ་ཁྲིད་པ་ལུས་ངག་ཡིད་ཀྱི་སྔོན་འགྲོ་འབྱོངས་པའི་ཚད་ནི།
[10627] དེ་ལས།
[10628] སེམས་ཅན་སྣང་བ་དང་མི་སྣང་བའི།
[10629] སྐད་ཅིག་གང་གིས་ཤེས་པ་དེ།
[10630] །རིགས་དྲུག་གཟུགས་སུ་མི་ལེན་ཕྱིར།
[10631] །འགྲོ་བ་དྲུག་གི་སྐད་རྣམས་ཤེས།
[10632] །འགྲོ་དྲུག་སེམས་རྒྱུན་རང་ཆོད་ཕྱིར།
[10633] །དེ་ཀུན་སེམས་ཀྱི་བསམ་པ་ཤེས།
[10634] །འཁྲུལ་རྟོག་བསམ་པ་ཟད་པའི་ཕྱིར།
[10635] །མདོར་ན་མངོན་ཤེས་དྲུག་བྱུང་སྟེ།
[10636] །འགྲོ་བ་དྲུག་གི་སྣང་བ་ནི།
[10637] །རང་གི་རང་བཞིན་ལོ་གནས་ནི།
[10638] །ཕྱིས་ཀྱི་སྐྱེ་རྒྱུན་བཅད་པའི་དོན།
[10639] །འདི་ལྟར་སྐྱེས་བུ་འདིས་ཆོད་པས།
[10640] །རང་རྟགས་འདི་ཀུན་འབྱུང་བའོ།
[10641] །ཞེས་སོ།
[10642] །རིག་པའི་སྣ་ཁྲིད་པ་འཁོར་འདས་རུ་ཤན་ཕྱེད་པའི་རྟགས་ནི།
[10643] ཡང་དེ་ལས།
[10644] འཁོར་འདས་རུ་ཤན་འབྱེད་པ་ཡི།
[10645] །སྔོན་དུ་འགྲོ་བ་ལུས་ངག་ཡིད།
[10646] །སྤྱོད་པ་ལ་ནི་གོམས་པ་ལ།
[10647] །འདི་ལྟར་ལུང་ལ་རྟགས་རྣམས་འབྱུང༌།
[10648] །ལུས་ཀྱི་ཞེན་པ་རང་ལོག་པས།
[10649] །འཁྲུལ་པའི་ཞེན་པ་རྒྱུན་བཅད་ཕྱིར།
[10650] །རང་གི་ལུས་ལ་ཆགས་པ་མེད།
[10651] །ནམ་མཁའ་ལ་ནི་སྤྱོད་པ་དང༌།
[10652] །དེ་བཞིན་ས་ལ་འཛུལ་བར་ནུས།
[10653] །བསམ་དང་མི་བསམ་དོན་རྣམས་ཤེས།
[10654] །ཁ་ནས་ཆོས་ཀྱང་སྨྲ་བ་འབྱུང༌།
[10655] །འདི་ལྟར་སུ་ལ་བྱུང་བའི་མི།
[10656] །ལུས་དང་ངག་དང་སེམས་ཀྱིས་ནི།
[10657] །འཁོར་བའི་རྟོགས་པའི་འཆིང་བ་ནི།
[10658] །གང་གིས་བཅིངས་པ་ཀུན་ལས་གྲོལ།
[10659] །ཕྱིས་ནི་ལུས་ངག་སེམས་འབྲེལ་ཆོད།

[10661] །དོན་གཉིས་པ་དངོས་གཞིའི་རྟགས་བྱེ་བྲག་ཏུ་བཤད་པ་ལ་གཉིས་ཏེ།
