[1799] །དེ་ཉིད་གྲོལ་བར་ངས་མ་བཤད།
[1800] །ངང་ཉིད་ངེས་པའི་རྣལ་འབྱོར་ནི།
[1801] །རྣམ་པར་མི་རྟོག་ཆིག་ཚུགས་བསྒོམ།
[1802] །མི་རྟོག་པ་ཡི་ངང་ཉིད་ཅན།
[1803] །དེ་ཡང་གྲོལ་བར་ངས་མ་བཤད།
[1804] །སྣང་བ་སེམས་ཀྱི་རྣལ་འབྱོར་ནི།
[1805] །ཐམས་ཅད་ང་ཡི་སེམས་ལས་བྱུང༌།
[1806] །སེམས་ཉིད་རྟོགས་ནས་ཅི་བྱར་ཡོད།
[1807] །དེ་སྐད་ཀུན་ཏུ་རྗོད་བྱེད་པ།
[1808] །ང་ཡིས་གྲོལ་བར་མ་བཤད་དོ།
[1809] །འཇུག་པ་ལས་ཀྱི་རྣལ་འབྱོར་ནི།
[1810] །དོན་ཉིད་བྱ་བས་འགྲུབ་པར་འདོད།
[1811] །བྱ་བ་ཆེན་པོས་བཅིངས་པ་རྣམས།
[1812] །དང་པོ་སེམས་བསྐྱེད་བྱ་བ་བྱེད།
[1813] །དེ་ནས་བཟུང་ནས་བྱ་བ་རྣམས།
[1814] །ཐམས་ཅད་བྱེད་ཅིང་བསྒྲུབ་པའོ།
[1815] །དེ་ཡང་གྲོལ་བར་ངས་མ་བཤད།
[1816] །བྱ་བ་རྒྱུ་ཡི་རྣལ་འབྱོར་ནི།
[1817] །རྒྱུད་ཉིད་ལམ་དུ་བསླང་བ་སྟེ།
[1818] །རྒྱུ་ལ་རེ་ཞིང་རྒྱུ་ལ་ལྟོས།
[1819] །རྒྱུ་ཉིད་མངོན་པར་སྟོན་པ་སྟེ།
[1820] །དེ་ཡང་གྲོལ་བར་ངས་མ་བཤད།
[1821] །བྱེད་པ་རྐྱེན་གྱི་རྣལ་འབྱོར་ནི།
[1822] །སྤྱོད་པས་ཡང་དག་དོན་ཚོལ་བ།
[1823] །རིགས་མི་མཐུན་པའི་ཕྱོགས་འཛིན་པ།
[1824] །དེ་ཡང་གྲོལ་བར་ངས་མ་བཤད།
[1825] །མཐར་ཕྱིན་འབྲས་བུའི་རྣལ་འབྱོར་ནི།
[1826] །རྟོགས་པ་ལམ་དུ་ཁྱེར་བ་སྟེ།
[1827] །ས་དང་ལམ་ལ་མི་གནས་པར།
[1828] །རང་གི་དེ་ཉིད་མཐོང་ནས་ནི།
[1829] །ཡང་དག་གྲོལ་བ་ཆེན་པོར་འདོད།
[1830] །ཡང་དག་དོན་གྱི་རྣལ་འབྱོར་ནི།
[1831] །རང་གི་རྟོགས་པ་རྙེད་ནས་ཀྱང༌།
[1832] །བར་དོའི་སྣང་བ་དག་པ་ལ།
[1833] །རང་གི་ཡང་དག་ཉིད་མཐོང་ནས།
[1834] །དེ་ནས་མཐར་ཕྱིན་གྲོལ་བར་འདོད།
[1835] །མཐར་ཕྱིན་གྲོལ་བའི་རྣལ་འབྱོར་ནི།
[1836] །ངེས་པ་རྙེད་པའི་རྣལ་འབྱོར་ནི།
[1837] །ངེས་པ་རྙེད་པའི་ས་ཉིད་ལ།
[1838] །རང་གི་དེ་ཉིད་རྙེད་མཐོང་ནས།
[1839] །ཡང་དག་ས་ལས་འདའ་བར་འདོད།
[1840] །བྱས་པ་རྫོགས་པའི་རྣལ་འབྱོར་ནི།
[1841] །ཀ་དག་སངས་རྒྱས་ཆེན་པོ་ཡིན།
[1842] །དེ་ལས་གཤེགས་སུ་མེད་པར་གསུངས།
[1843] །རྣལ་འབྱོར་ཆེན་པོ་དེ་བཞི་ལ།
[1844] །ས་ལ་སྦྱང་དུ་མེད་པར་གསུངས།
[1845] །ལམ་ལ་བགྲོད་དུ་མེད་པར་གསུངས།
[1846] །ཞེས་སོ།
[1847] །དེའང་ལམ་བསྒོམ་ཚུལ་ཐ་དད་པས།
[1848] །འདིར་འོད་གསལ་མངོན་སུམ་བྱེད་པས་འོག་མ་ནས་གདོད་ཞུགས་ཏེ་ས་ལམ་སྦྱང་བགྲོད་མི་དགོས་པར་གསུངས་ཀྱང༌།
