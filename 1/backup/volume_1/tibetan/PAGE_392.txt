[16293] མ་བརྟགས་ན་བློ་དང་སྣང་ཡུལ་འཁྲུལ་པ་རང་རིག་འཆིང་བྱེད་ལྟར་སྣང་བའོ།
[16294] །བཅིངས་པར་སྣང་དུས་ནས་བཅིངས་པ་མེད་ལ།
[16295] གྲོལ་བར་སྣང་དུས་ནས་འཆིང་བྱེད་རང་མཚན་པ་མེད་པས་གྲོལ་བ་མེད་དེ།
[16296] སྒྱུ་འཕྲུལ་ལས།
[16297] སུས་ཡང་མ་བཅིངས་བཅིངས་མེད་དེ།
[16298] འཆིང་བར་བྱ་བ་ཡོད་མ་ཡིན།
[16299] །བཅིངས་མེད་རྣམ་པར་བཀྲོལ་མེད་པ།
[16300] །ཡེ་ནས་ལྷུན་རྫོགས་སངས་རྒྱས་ཆོས།
[16301] །བསྟན་ཕྱིར་སྤྲོ་བ་སྣ་ཚོགས་མཛད།
[16302] །ཅེས་གསུངས་པ་བཞིན་ནོ།
[16303] །ཡང་རང་སྣང་ཡས་སུ་མི་འཇུག་པ་འཆིང་བས་ཀྱང་ཞགས་པའོ།
[16304] །སྣང་བ་ལ་ཞེན་པ་མེད་པར་ནང་ཏིང་ངེ་འཛིན་གྱི་དགའ་བདེས་རླན་པར་བྱེད་པས་ཆུ་ཞེས་བྱའོ།
[16305] །ཆུའི་དངས་མ་འདུས་པ་ལ་བརྟེན་པས་ཀྱང་ཆུ་ཞེས་བྱའོ།
[16306] །གཟུགས་སྣང་དབང་ཤེས་ལ་གསལ་བ་དང༌།
[16307] འོད་སྣང་གོང་འཕེལ་དུ་ཤར་བས་སྒྲོན་ཞེས་པའོ།
[16308] །དབྱིངས་་རིག་གི་སྣང་བ་སྐྱེད་པའི་གཞི་མ་ཡིན་པས་མ་ཞེས་བྱའོ།
[16309] །དངས་མའི་དབང་པོ་གཟུགས་ཅན་ལྟ་བའི་རྟེན་གཞི་བྱེད་པས་ཀྱང་མ་ཞེས་བྱའོ།
[16310] །དེ་སྐད་དུ་ཡང་སྒྲོན་མ་འབར་བ་ལས།
[16311] ངེས་པའི་ཚིག་ནི་འདི་ལྟ་བུ།
[16312] །རྒྱང་ནི་རིང་པའི་རྒྱུད་འཛིན་པས།
[16313] གཟུགས་སུ་་སྣང་བ་འཛིན་པ་དང༌།
[16314] །ཁ་དོག་ལ་སོགས་སྡུད་པ་དང༌།
[16315] །རིག་པའི་སྐུ་ཡང་རྒྱང་རིང་ནས།
[16316] །མཐོང་བྱེད་འཛིན་པར་བྱེད་པ་དང༌།
[16317] །འཁོར་བ་རྟོག་པའི་ཕྱོགས་ནས་ནི།
[16318] །རྒྱང་དུ་འཕེན་པས་རྒྱང་ཞེས་བྱ།
[16319] །དེ་བཞིན་མ་རྟོགས་ཕྱོགས་ནས་ཀྱང༌།
[16320] །མྱང་འདས་འཕེན་པས་རྒྱང་ཡིན་ནོ།
[16321] །དེ་བཞིན་འཁོར་བའི་སྣང་ཆ་ནས།
[16322] །གཟུགས་དང་འདོད་ཡོན་ལྔ་རྣམས་ནི།
[16323] །འདི་ཡིས་འཆིང་བས་ཞགས་པ་ཡིན།
[16324] །དབང་པོའི་སྣང་ལ་སྣང་བ་ཡི།
[16325] །དྲན་བསམ་བདག་ཏུ་འདིས་འཛིན་པས།
[16326] །ཞགས་པ་ཞེས་ནི་རྒྱང་ནས་འཛིན།
[16327] །དེ་བཞིན་ནམ་མཁའ་སྤྲིན་བྲལ་ལ།
[16328] །རིག་པ་བདག་མེད་སྐུར་སྣང་བ།
[16329] །སྣང་བ་ཡས་སུ་མི་འཇུག་པར།
[16330] །འདི་ཡིས་འཛིན་པས་ཞགས་ཞེས་བྱ།
[16331] །ཆུ་ནི་སྣང་བ་དེ་དག་ལ།
[16332] །ཞེན་པ་རང་སར་ལྡོག་ནས་ན།
[16333] །ཀུན་ལ་ཆགས་པའི་སེམས་མེད་པར།
[16334] །སྣང་བ་རང་སར་བཞག་པས་ནི།
[16335] །བདག་འཛིན་བྲལ་བས་ཆུ་ཞེས་བྱ།
[16336] །སྒྲོན་ནི་རང་གི་སྣང་བ་ཀུན།
[16337] །གོང་ནས་གོང་དུ་འཕེལ་བ་ཡིས།
[16338] །རིག་པའི་ཡེ་ཤེས་གསལ་བས་སྒྲོན།
[16339] །འབྱུང་བ་རགས་པའི་སྣང་བ་ཀུན།
[16340] །རང་ཤེས་སྤྱད་པ་འགགས་མེད་ཅིང༌།
[16341] །དབང་པོའི་རང་སྒོར་གསལ་བས་སྒྲོན།
[16342] །རིག་པ་ཕྱི་ཡི་དབྱིངས་བཞག་པས།
[16343] །ནང་དབྱིངས་རང་སར་དག་སྟེ།
