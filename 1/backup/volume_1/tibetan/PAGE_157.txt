[6139] །གདོལ་པའི་རིགས་ཀྱིས་ལོ་སྟོང་པར།
[6140] །འདི་ཉིད་གནས་ཤིང་འཕེལ་བར་བྱས།
[6141] །དེ་འདས་འོག་ཏུ་མུན་པ་ཉིད།
[6142] །ལོ་ནི་སྟོང་ཕྲག་ལྔར་གནས་སོ།
[6143] །དེ་ནས་ཡང་ནི་འདིར་བབས་ཏེ།
[6144] །རྡོ་རྗེ་གདན་ཞེས་འབྱུང་བའི་གནས།
[6145] །འདི་ཉིད་འཕེལ་ནས་འོད་འབར་བ།
[6146] །ལོ་ནི་ལྔ་བརྒྱ་ལྔ་བཅུའོ།
[6147] །འདི་ཉིད་བློ་གྲོས་བཟང་པོས་འཛིན།
[6148] །དེ་ཡི་ཡོན་བདག་མཁའ་འགྲོ་མ།
[6149] །བསིལ་བྱེད་གཙང་མས་བྱས་ནས་ནི།
[6150] །དེ་ཡི་འོག་ཏུ་འཇིག་རྟེན་འདི།
[6151] །ས་ཆུ་མེ་རླུང་སྡེབས་འཁྲུགས་ནས།
[6152] །མི་གནས་སྟོང་པ་ཆེན་པོར་འགྱུར།
[6153] །བསྐལ་པ་ཞེས་པ་དང་པོ་ལས།
[6154] །ཐ་མ་འཇིག་པའི་དུས་འབྱུང་ངོ༌།
[6155] །ཞེས་སོ།
[6156] །བློ་གྲོས་བཟང་པོ་ནི་འདི་འཛིན་པ་ཐམས་ཅད་ཡིན་ལ།
[6157] ལོ་ལྔ་བརྒྱ་ནི་ལོ་བོ་ཆེ་སྟེ།
[6158] ལྔ་བརྒྱ་ལྔ་བརྒྱར་ལོངས་པའོ།
[6159] །དེ་ཙམ་ཞིག་མི་རྣམས་ལ་བསྐལ་པ་དང་པོ་ནས་ཚེ་ལོ་བཅུ་པ་ཡ་ཐོག་འདིའི་བར་སྣང་ལ།
[6160] གཞན་རྣམས་ནི་མ་ཡིན་པ་མཁའ་འགྲོ་མའི་གནས་སུ་དོན་བྱེད་པའོ།
[6161] །གཉིས་པ་དར་ནུབ་བྱེ་བྲག་ཏུ་བཤད་པ་ནི།
[6162] འཛམ་བུའི་གླིང་འདིར་ཡང་ད་རེས་ཡ་ཐོག་གི་ཚེ་ལོ་བཅུ་པའི་བར་འདི་ལ་འཕེལ་འགྲིབ་རྣམ་པ་བདུན་འབྱུང་བ་ནི།
[6163] ཨ་ཏི་བཀོད་པ་ཆེན་པོ་ལས།
[6164] ཀྱེ་ཉོན་ཅིག་རིགས་ལྡན་ཤཱ་རིའི་བུ།
[6165] །བསྟན་པའི་དམ་པ་རྒྱ་ཆེ་རབ་བཀོད་པ།
[6166] །མཆོག་ཏུ་སྙིང་པོ་ཡང་ནས་ཡང་བསྡུས་པ།
[6167] །གསེར་གྱི་སྙིང་པོ་ཉི་མ་སྣང་བྱེད་འདི།
[6168] །ཐུན་མོང་འཆོལ་བར་དར་བ་མ་ཡིན་ཏེ།
[6169] །གསང་ཆེན་རིགས་དང་ལྡན་པ་འགའ་ལ་སྣང༌།
[6170] །སྙིགས་མ་རྩོད་ལྡན་ཉོན་མོངས་རགས་པའི་དུས།
[6171] །སྡེ་པ་ནང་འཐབ་དབེན་གནས་ཀུན་ཀྱང་འཇིག
[6172] །རྒྱལ་ཁྲིམས་མི་གནས་རང་དགའ་འདོད་རྒྱལ་སྤྱོད།
[6173] །ཕ་དང་བུ་འཐབ་མཚོན་གྱིས་ལུས་རྣམས་བསྲུང༌།
[6174] །བསགས་པའི་ཚོགས་ཡལ་ས་ཡི་བཅུད་རྣམས་ཟད།
[6175] །མི་མཐུན་ལྟ་སྤྱོད་དག་ཀྱང་གྲངས་མང་འབྱུང༌།
[6176] །སྙིགས་མ་ལྔ་བརྒྱ་ཚེ་ལོ་རིམ་གྱིས་བྲི།
[6177] །བསོད་ནམས་སྐལ་ཆུང་ཚེ་ལོ་དྲུག་ཅུའི་ཚེ།
[6178] །སྙིང་པོ་འདི་ནི་ལོ་ནི་དྲུག་ཅུར་གནས།
[6179] །དེ་འདས་འོག་ཏུ་ལོ་བརྒྱའི་བར་དུ་ནི།
[6180] །རྟོག་གེ་སྤྱོད་ཅིང་གསང་བ་འདི་མི་གནས།
[6181] །དེ་འདས་འོག་ཏུ་ལོ་སྟོང་ཉིད་དུ་ཡང༌།
[6182] །ཐོས་བསམ་བྲལ་བའི་གང་ཟག་ཡལ་ནས་ནི།
[6183] །གསང་བ་མཆོག་འདི་ཉི་མ་བཞིན་དུ་འགྱུར།
[6184] །དེ་འདས་འོག་ཏུ་ཉིས་བརྒྱ་མུན་པར་འགྱུར།
[6185] །དེ་ནས་ཚེ་ལོ་དག་ནི་བཞི་བཅུ་ན།
[6186] །སྙིང་པོ་འདི་ནི་ལོ་བརྒྱའི་བར་དུ་གནས།
[6187] །དེ་འདས་སུམ་ཅུའི་བར་དུ་མུན་པར་འགྱུར།
[6188] །དེ་འདས་ལོ་བརྒྱའི་བར་དུ་གནས་པ་ལ།
[6189] །སྐྱེ་དགུའི་ཚེ་ལོ་སུམ་ཅུའི་དུས་ན་ནི།
[6190] །གསང་མཆོག་འདི་ནུབ་རྒྱང་ཕན་ལོ་བརྒྱད་དར།
[6191] །ཚེ་ལོ་ཉི་ཤུ་པ་ལ་འདི་མི་འབྱུང༌།
[6192] །ཚེ་ལོ་བཅུ་པའི་དུས་སུ་སྙིང་པོ་འདི་བྱུང་ནས།
[6193] །འདི་དང་གང་ཕྲད་འབྱུང་བ་རང་དེངས་འགྱུར།
[6194] །ནམ་མཁར་ཡལ་ཏེ་མི་སྣང་བསྟན་པའི་བཙས།
[6195] །ཡང་ནི་བསྟན་པའི་མེ་རོ་གཞན་དུ་འགྱུར།
[6196] །ཅེས་པ་དང༌།
