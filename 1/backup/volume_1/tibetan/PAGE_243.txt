[9820] དམ་ཚིག་ཉམས་པའི་སྔ་ལྟས་སུ།
[9821] །མི་བདེ་རྣམ་པ་སྣ་ཚོགས་འབྱུང༌།
[9822] །ཡམས་དང་འཚེ་བའི་ནད་རྣམས་དང༌།
[9823] །རིམས་ནད་སྣ་ཚོགས་འབྱུང་བར་འགྱུར།
[9824] །གདོན་དང་ལོག་འདྲེན་རྣམས་ཀྱིས་ཀྱང༌།
[9825] །
[9826] བསད་ནས་སེམས་ཅན་དམྱལ་བར་འགྲོ།
[9827] །མིག་གིས་གཟུགས་མི་མཐོང་བར་འགྱུར།
[9828] །རྣ་བ་འོན་ཅིང་ལས་མི་འགྲུབ།
[9829] །མཛེ་དང་ཤུ་བའི་ནད་རྣམས་འབྱུང༌།
[9830] ཆོམ་རྐུན་རྒྱལ་པོའི་ཆད་པ་འབྱུང༌།
[9831] །གཞན་ལ་མེད་པའི་རིམས་ནད་འབྱུང༌།
[9832] །བུ་དང་ཆུང་མ་འཆི་བར་སྟོན།
[9833] །ཡུལ་རྣམས་ཐམས་ཅད་དགྲར་ལྡང་ཞིང༌།
[9834] །བྱས་པ་དོན་མེད་ཆེན་པོར་འགྱུར།
[9835] །ཅེས་སོ།
[9836] །ཉམས་ན་བསྐང་ཐབས་ལ་གསུམ་ལས།
[9837] དང་པོ་སྐུའི་ཉམས་ན་བསྐང་བའི་ཐབས་ནི།
[9838] ཡང་དེ་ཉིད་ལས།
[9839] གསུང་རབ་རྒྱ་ཆེན་བཀླག་པ་དང༌།
[9840] །ཚོགས་ཀྱི་འཁོར་ལོ་རྒྱ་ཆེན་བསྐོར།
[9841] །འཁོར་གསུམ་ཡང་དག་བྱ་བར་གསུངས།
[9842] །ཞེས་སོ།
[9843] །གསུང་ཡང་དེ་ལས།
[9844] དེ་ཡང་སྦྱང་ཐབས་འདི་ལྟ་སྟེ།
[9845] མར་མེ་བརྒྱ་རྩའི་ཆོ་ག་དང༌།
[9846] །སྤང་སྐོང་བརྒྱ་རྩ་གདོན་པ་དང༌།
[9847] །བཤགས་པ་རྣམ་པ་སྣ་ཚོགས་བྱ།
[9848] །དེས་ནི་བྱང་བའི་ཡོན་ཏན་མཐོང༌།
[9849] །ཞེས་སོ།
[9850] །སྤང་སྐོང་ནི་ན་རག་རྒྱུན་བཤགས་སོ།
[9851] །ཐུགས་ཀྱང་དེ་ཉིད་ལས།
[9852] དེ་ཡང་སྦྱང་ཐབས་འདི་ལྟ་སྟེ།
[9853] སློབ་དཔོན་མཉེས་པའི་རྫས་ཀྱིས་མཆོད།
[9854] །འབྱོར་ཚོགས་ཅི་ཡོད་འཕགས་པ་མཆོད།
[9855] །དེ་ལྟར་བཤགས་པ་བྱས་གྱུར་ན།
[9856] །བསམ་པ་ཐམས་ཅད་འགྲུབ་པར་འགྱུར།
[9857] །འདོད་པ་ཐམས་ཅད་སྐོང་བར་འགྱུར།
[9858] །བྱ་བ་ཐམས་ཅད་རྫོགས་པར་འགྱུར།
[9859] །ཅེས་སོ།
[9860] །བསྲུང་ཐབས་ནི།
[9861] སྐྱོན་ཡོན་མཐོང་ནས་དྲན་པ་དང་ཤེས་བཞིན་དག་པས་རྒྱུད་སྡོམ་པ་སྟེ།
[9862] དེ་ཉིད་ལས།
[9863] དམ་ཚིག་ལ་ནི་ཇི་བཞིན་གནས།
[9864] །བསྐལ་པ་ཞིག་ཀྱང་མི་གཏོང་ན།
[9865] །དམ་ཚིག་ཆེན་པོའི་བསྲུང་ཐབས་སོ།
[9866] །ཞེས་སོ།
[9867] །དེའང་སློབ་དཔོན་གྱིས་རྒྱུད་དང་དམ་ཚིག་གི་རིམ་པ་ཤེས་པས་ཞེས་བཅུས་བསྲུངས་ལ།
[9868] སློབ་མས་མི་ཤེས་པས་སློབ་དཔོན་གྱི་བཀའ་བཞིན་བསྲུང་བའོ།
[9869] །ཤེས་པ་བཅུ་ནི།

[9871] དྲིས་པའི་ལན་ལྡོན་པར་ཤེས་པ་དང༌།
