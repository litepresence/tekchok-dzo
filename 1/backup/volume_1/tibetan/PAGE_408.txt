[17096] རང་སྣང་རང་ངོར་རིག་པས་ཁྱབ་ཅིང་ཤེས་པས་མ་ཞེས་བྱའོ།
[17097] །ཡང་དེ་ཉིད་ལས།
[17098] ཤེས་པས་དངོས་དང་དངོས་པོ་མེད་པའི།
[17099] །སྣང་བ་ཀུན་ལ་རབ་ཞུགས་ནས།
[17100] །སྤྱི་ཡི་དོན་རྣམས་ཤེས་པ་དང༌།
[17101] །མ་བཙལ་བཞག་པའི་གདེང་རྙེད་དེ།
[17102] །རང་གི་ཡེ་ཤེས་ཚད་རྣམས་ཤེས།
[17103] །རང་བཞིན་དག་པའི་ཆོས་སྐུ་ནི།
[17104] །སུམ་མ་རྙེད་པ་རང་རིག་པས།
[17105] །ཀུན་ལས་ཁྱད་འཕགས་རབ་ཏུ་སྣང༌།
[17106] །གཞན་གྱིས་དཔྱད་པའི་གནས་མེད་པས།
[17107] །རང་ཤར་ཡེ་ནས་དག་པའི་ཡུལ།
[17108] །གཞན་གྱི་ཡུལ་མིན་རང་ཞེས་བྱ།
[17109] །བྱས་པས་འདི་དོན་རྙེད་པ་མེད།
[17110] །སུ་ལ་རེ་བ་མེད་པའི་ཕྱིར།
[17111] །ཡེ་ནས་རང་སྣང་བྱུང་ཞེས་བྱ།
[17112] །སྒྲོན་མས་ཆོས་ཉིད་ཐམས་ཅད་ཀུན།
[17113] །གཉིས་སུ་མེད་པར་སྣང་བར་བྱེད།
[17114] །མས་ནི་རང་ངོ་ཁྱབ་པས་སོ།
[17115] །ཞེས་པ་དང༌།
[17116] མུ་ཏིག་ཕྲེང་བ་ལས།
[17117] དཔེར་ན་སྲེག་བྱེད་སྒྲོན་མེ་ལྟར།
[17118] །ཤེས་པས་ཆོས་རྣམས་ཅིག་ཅར་རྟོགས།
[17119] །རབ་ནི་ཀུན་གྱི་རྩེ་མོར་གྱུར།
[17120] །རང་ནི་གཞན་ལྟོས་ཆོས་དང་བྲལ།
[17121] །བྱུང་བས་མ་བཀག་ལྷུག་པར་སྣང༌།
[17122] །ཞེས་གསུངས་སོ།
[17123] །གསུམ་པ་དབྱེ་ན་ལྔ་ལས།
[17124] གཞི་གནས་རང་བྱུང་གི་ཤེས་རབ་ནི།
[17125] ནང་གནས་རིག་པའི་མཁྱེན་ཆ་སྟེ་འཆར་གཞི་དང་རྟེན་དུ་གནས་པའོ།
[17126] །སྣང་བ་ཅེར་འཛིན་གྱི་ཤེས་རབ་ནི།
[17127] དབྱིངས་རིག་གཉིས་མེད་ཟུང་འཇུག་གི་རང་དུས་རང་གསལ་དྲི་མེད་དག་པས་དགོངས་པ་ཆོས་ཉིད་དུ་སྐྱེལ་བར་བྱེད་པའོ།
[17128] །དྲན་པ་འགག་མེད་ཀྱི་ཤེས་རབ་ནི།
[17129] ཤར་རིས་མེད་དུ་མར་ཤར་ཡང༌།
[17130] གྲོལ་རང་ངོ་ཆོས་ཉིད་གཅིག་ཏུ་སྡུད་པའོ།
[17131] །སོ་སོར་རྟོག་པའི་ཤེས་རབ་ནི།
[17132] ཆོས་དང་ཆོས་ཅན་རང་དང་སྤྱིའི་རྣམ་པ་དང༌།
[17133] མཚན་ཉིད་ལ་སོགས་པ་ཇི་ལྟར་སྣང་བ་མ་འདྲེས་པར་ཤེས་པའོ།
[17134] །འཇུག་བྱེད་རྣམ་གསུམ་གྱི་ཤེས་རབ་ནི།
[17135] རང་བཞིན་གྱི་བར་དོ་འདིར་ཐོས་པས་གཏན་ལ་འབེབས།
[17136] བསམ་པས་ངེས་པར་བྱེད།
[17137] སྒོམ་པས་རོ་མྱོང་བས་འཁོར་བ་ལས་ངེས་པར་འབྱིན་པ་སྟེ།
[17138] ཚིག་དོན་རྟོགས་པར་བྱེད་པའོ།
[17139] །ཡང་དེ་ཉིད་ལས།
[17140] དབྱེ་བ་རྣམ་པ་ལྔ་ཡིན་ཏེ།
[17141] །གཞི་གནས་རང་བྱུང་ཤེས་རབ་ཀྱིས།
[17142] །ལུས་སོགས་ཆོས་ཀྱི་རྟེན་རྣམས་བྱེད།
[17143] །སྣང་བ་ཅེར་འཛིན་ཤེས་བར་ཀྱིས།
[17144] །དབྱིངས་དང་རིག་པའི་འདུན་ས་འཛིན།
[17145] །དྲན་པ་འགག་མེད་ཤེས་རབ་ཀྱིས།
[17146] །ཤེས་པ་རྩེ་གཅིག་ཡུལ་རྣམས་སྡུད།
[17147] །སོ་སོར་རྟོག་པའི་ཤེས་རབ་ཀྱིས།
[17148] །སྤྱི་དང་རང་གི་མཚན་ཉིད་འབྱེད།
[17149] །འཇུག་བྱེད་རྣམ་གསུམ་ཤེས་རབ་ཀྱིས།
[17150] །རང་བཞིན་བར་དོར་གཏན་ལ་འབེབས།
[17151] །ཞེས་སོ།
[17152] །བཞི་པ་མཚན་ཉིད་ནི།
