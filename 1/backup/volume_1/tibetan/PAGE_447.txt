[18984] དོན་དམ་ཆོས་ཉིད་འབྱུང་བ་ལས།
[18985] །སྒྲོན་མ་རྣམས་པ་བཞི་ཞེས་ཏེ།
[18986] །སྨིན་ཅིང་འཇུག་ལ་ཁྱབ་སྟོང་པས།
[18987] །ཆོས་ཉིད་ཡུལ་སྣང་སྡུད་པའོ།
[18988] །རྒྱང་ཞགས་ལས་ནི་གཟུགས་བསྡུས་ཏེ།
[18989] །རླུང་གིས་གཅོད་ཅིང་འཕེན་པར་བྱེད།
[18990] །འོང་དང་འགྲོ་བའི་འཕེལ་འགྲིབ་ཀྱིས།
[18991] །ཡེ་ཤེས་དག་ནི་སྡུད་པར་བྱེད།
[18992] །ཐིག་ལེ་སྟོང་པས་ལས་ཀུན་བྱེད།
[18993] །འདས་པ་ལམ་གྱི་སྣ་བཟུང་ནས།
[18994] །འདས་པ་ལམ་གྱི་སྣ་བཟུང་ནས།
[18995] །རླུང་གིས་བཀྲག་དང་གཟི་མདངས་བསྐྱེད།
[18996] །བསྒྱུར་ཞིང་ཡེ་ཤེས་སྣང་བ་སྟོན།
[18997] །ཤེས་རབ་རང་བྱུང་སྒྲོན་མ་ཡིས།
[18998] །ཆོས་ཀུན་མཉམ་པ་རོ་གཅིག་ཏུ།
[18999] །བསྡུས་ཏེ་རང་སྣང་འཛིན་པར་བྱེད།
[19000] །འདིས་ནི་བག་ཆགས་འབྲེལ་རྒྱུན་གཅོད།
[19001] །དབྱིངས་ཀྱིས་དངས་མ་སྡུད་པ་དང༌།
[19002] རིག་པའི་སྐུ་རྣམས་འཛིན་པ་དང༌།
[19003] །གནད་གསུམ་བཅུད་དུ་སྨིན་པ་ཡིས།
[19004] །འཁོར་བ་ཉིན་ནི་སྤོང་བར་བྱེད།
[19005] །དེ་ལྟར་མིག་ལས་སྒྲོན་ཤར་བས།
[19006] །སངས་རྒྱས་དགོངས་པའི་གནད་འདུས་པར།
[19007] །རང་སྣང་ཡུལ་རྣམས་འཛིན་པའོ།
[19008] །སྒྲོན་མའི་གནད་ནི་ངས་བཤད་ཀྱིས།
[19009] །ལྷའི་བུ་སྐལ་ལྡན་ཁྱོད་ཉོན་ཅིག
[19010] །ཡུལ་དང་རིག་པ་རླུང་དག་གིས།
[19011] །ཆོས་ཉིད་ལམ་དུ་གཟུང་བའི་ཕྱིར།
[19012] །ཡེ་ཤེས་རྫོགས་པས་ཆོས་ཉིད་དག
[19013] །འདི་ཡང་སྒྲོན་མའི་གནད་ཡིན་ནོ།
[19014] །རྒྱང་ཞགས་འགུལ་བ་མེད་པ་གནད།
[19015] །འདི་དུས་ལྟ་སྟངས་གསུམ་དང་སྦྱར།
[19016] །ཤེས་རབ་སྒྲོན་མ་སྦྱང་བའི་གནད།
[19017] །ཆོ་ག་གསུམ་གྱིས་སྤེལ་བ་དང༌།
[19018] །བསྒྲུབ་པའི་གནད་ཀྱིས་རྒྱ་ཉིད་བསྐྱེད།
[19019] །ཐིག་ལེའི་སྒྲོན་མ་གཏེམས་པ་གནད།
[19020] །རྩ་དང་མཐེབ་མཛུབ་མ་བྲལ་བས།
[19021] །ཡེ་ཤེས་གོང་དུ་འཕེལ་བའོ།
[19022] །དབྱིངས་ཀྱི་གནད་ནི་འཁྲིད་པ་སྟེ།
[19023] །ཉི་མའི་བསླབ་ཐབས་རྣལ་འབྱོར་པས།
[19024] །ལེགས་པར་སྦྱར་ཏེ་ཁྲིད་པར་བྱ།
[19025] །ཟླ་བ་དག་ལའང་བསླབ་པར་བྱའོ།
[19026] །འདི་ལྟར་རྒྱུན་དུ་སུས་བྱས་པ།
[19027] །འཁྲུལ་པའི་སྣང་བ་རྒྱུན་ཆད་དེ།
[19028] །དག་པ་ཡེ་ཤེས་ལྔ་ཡི་འོད།
[19029] །ས་རྡོ་རི་བྲག་བགས་་སྟོངས་ནས།
[19030] །གྲུ་ཆད་དམ་ནི་ཡུལ་གྲུ་ཙམ།
[19031] །འཁྲིད་ལ་མཁས་པས་སྣང་བར་འགྱུར།
[19032] །འདི་དུས་ཐབས་ཀྱིས་བཅོས་པ་གནད།
[19033] །རྫོགས་པས་ཆོས་ཉིད་ཐོབ་པའོ།
[19034] །འདི་ལྟར་གོམས་པས་ཆོས་ཉིད་སྟོབས།
[19035] །ལུས་འབྱུང་རང་སོར་རྫོགས་པའོ།
[19036] །དེ་ལྟར་ལུས་འབྱུང་རྒྱུན་ཆད་པས།
[19037] །ཟག་བཅས་ཕུང་པོ་མི་སྣང་བར།
[19038] །འདི་ལ་མངོན་པར་འཚང་རྒྱའོ།
[19039] །ཞེས་སོ།
[19040] །གསུམ་པ་སྣང་བའི་གདངས་འོད་གསལ་བསྟན་པ་ལ་ལྔ་སྟེ་སྐུ།
