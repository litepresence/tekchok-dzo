[8177] རིགས་ཀྱི་སྒོ་ནས་བརྟག་པ་ལ་གསུམ་ལས།
[8178] ཕྱིའི་རིགས་ནི་རྒྱལ་རིགས་ཀྱིས་བདེ་བར་གཤེགས་པའི་རིགས་ཀྱི་སངས་རྒྱས་འགྲུབ་ལ།
[8179] རྗེའུའི་རིགས་ཀྱིས་རྡོ་རྗེ་དང༌།
[8180] དམངས་རིགས་ཀྱིས་རིན་པོ་ཆེ་དང༌།
[8181] བྲམ་ཟེས་པདྨ་དང༌།
[8182] གདོལ་པའི་རིགས་ཀྱིས་ལས་ཀྱི་རིགས་ཀྱི་སངས་རྒྱས་འགྲུབ་པའོ།
[8183] །སངས་རྒྱས་གང་རུང་གཅིག་གིས་འགྲུབ་བོ་སྙམ་ན་མ་ཡིན་ཏེ།
[8184] རིགས་སོ་སོ་ལ་རིག་མ་ཡང་སོ་སོ་མེད་ན་མི་འགྲུབ་བོ།
[8185] །སངས་རྒྱས་རིགས་ལྔར་བྱུང་བ་ལྟར་རིགས་ལྔར་ངེས་སོ།
[8186] །དེ་ལྟར་ཡང༌།
[8187] དབང་བཀོད་ཆེན་པོ་ལས།
[8188] སངས་རྒྱས་རིགས་ཀྱང་བྱེ་བྲག་ལས།
[8189] རིག་མའང་དེ་བཞིན་བྱེ་བྲག་འགྱུར།
[8190] །ཞེས་སོ།
[8191] །ནང་གི་རིགས་ནི་གནོད་སྦྱིན་གྱི་བུ་མོ་དང༌།
[8192] ལྷའི་བུ་མོ་དང༌།
[8193] ཀླུའི་བུ་བོ་དང༌།
[8194] ཚངས་པའི་བུ་མོ་དང༌།
[8195] བརྒྱ་བྱིན་གྱིས་བུ་མོ་ལྔ་ནི་ལས་རྒྱའི་རིགས་ཏེ།
[8196] རྡོ་རྗེ་མཁའ་འགྲོ་སྡོམ་པ་ལས།
[8197] ལྷ་དང་ལྷ་མིན་གནོད་སྦྱིན་ནམ།
[8198] །ཀླུ་དང་བརྒྱ་བྱིན་ཚངས་པ་དང༌།
[8199] རིགས་ལས་ལྷག་པར་ཕྱག་རྒྱས་ཤིས།
[8200] །བཀུག་ལ་རང་གིས་ལེགས་པར་སྦྱར།
[8201] ཅེས་སོ།
[8202] །སྔོན་ཀོ་ས་ལའི་སློབ་དཔོན་དཔལ་ལྡན་བདེ་བ་ཆེན་པོས་ཚངས་པ་དང་བརྒྱ་བྱིན་ལ་སོགས་པའི་བུ་མོ་བཀུག་ནས་དབང་བསྐུར་བས་དངོས་གྲུབ་བརྙེས་པ་བཞིན་ནོ།
[8203] །གསང་བའི་རིགས་ནི་བཞི་སྟེ།
[8204] རིགས་ཅན་བཞིའི་རྩ་ལས་བུད་མེད་རྣམས་མི་འདའོ།
[8205] །དེའང་དུང་ཅན་མ་དང༌།
[8206] གླང་སྣ་ཅན་མ་དང༌།
[8207] པདྨ་ཅན་དང༌།
[8208] རི་དགས་ཅན་དང༌།
[8209] བ་གླང་ཅན་མའོ།
[8210] །དུང་ཅན་མ་ནི།
[8211] སྐྲ་ནག་ལ་སོ་ལེགས་ཤིང་ཐགས་བཟང༌།
[8212] གླང་སྣ་མ་སྐྲ་ངན་པ་མདུན་ན་སོ་ལྷག་ཡོད།
[8213] པདྨ་ཅན་ནི།
[8214] སྒྲ་སེར་ལ་གཡས་སུ་ལི།
[8215] རི་དགས་ཅན།
[8216] སྐྲ་ངན་ལ་དཔྲལ་གདེངས་ཆེ།
[8217] བ་གླང་ཅན།
[8218] སྐྲ་བཟང་ལ་དཔྲལ་བ་ན་གཉེར་མ་གྱེན་དུ་འགྲེང་བ་ཡོད་པའོ།
[8219] །དེའང་པདྨ་ཅན་གྱིས་བདེ་བ་ལྷག་པར་སྐྱེ་བར་བྱེད་ཅིང་ཆོས་ཉིད་གསལ་བའི་གྲོགས་ཏེ།
[8220] དཔེར་ན་པདྨ་རྒྱས་པ་དང་མ་རྒྱས་པ་ལྟ་བུའི་རྩ་ཡོད་པས།
[8221] རྩའི་ཁ་འབྱེད་པའི་གདམས་ངག་གཙོ་ཆེའོ།
[8222] །རི་དགས་ཅན་ནི།
