[130] །དེ་དག་མཚན་ཉིད་མི་འདྲ་བསྐལ་མི་མཉམ།
[131] །སྐྲ་ཡི་ཁྱོན་ལ་འགྲོ་བ་ལྔ་རྣམས་ཏེ།
[132] །སེམས་ཅན་དམྱལ་དང་དུད་འགྲོའི་འགྲོ་བ་དང༌།
[133] །གཤིན་རྗེའི་འཇིག་རྟེན་ལྷ་དང་མི་རྣམས་བཞུགས།
[134] །འདྲེས་པ་མེད་ཅིང་གནོད་པར་གྱུར་པ་མེད།
[135] །དེ་དག་གནས་སུ་མཚོ་དང་རྒྱ་མཚོར་བཅས།
[136] །ཆུ་ཀླུང་དེ་བཞིན་མཚོ་དང་ལྟེང་ཀ་རྣམས།
[137] །འདྲེས་པ་མེད་ཅིང་གནོད་པར་འགྱུར་མེད་དེ།
[138] །དེ་ལྟར་རྒྱལ་བའི་ཆོས་ནི་བསམ་མི་ཁྱབ།
[139] །དེ་ཡི་གནས་སུ་རི་བོ་དུ་མ་སྟེ།
[140] །ཁོ་ར་ཁོར་ཡུག་ལྷུན་པོ་རབ་ལྷུན་དང༌།
[141] །བཏང་ཟུང་དང་ནི་བཏང་ཟུང་ཆེན་པོ་དང༌།
[142] །འབིགས་བྱེད་བྱ་རྒོད་ཕུང་པོ་གངས་ཀྱི་རི།
[143] །གནས་དེར་སེམས་ཅན་དམྱལ་བ་མི་བཟད་པ།
[144] །ཚ་དང་རབ་ཏུ་ཚ་བ་ཉམས་མི་དགའ།
[145] །དེ་ནས་སེམས་ཅན་དམྱལ་བའི་ཚོར་བ་ནི།
[146] །སྡུག་བསྔལ་དེ་དག་རྗེས་སུ་མྱོང་བར་བྱེད།
[147] །དེ་ཡི་གནས་ན་ལྷ་ཡི་གཞལ་མེད་ཁང༌།
[148] །དེ་དག་དཔག་ཚད་བཅུ་གཉིས་ཉམས་དགའ་བ།
[149] །ལྷ་རྣམས་སྟོང་ཕྲག་མང་པོ་དག་གིས་ཀྱང༌།
[150] །ལྷ་ཡི་དགའ་བས་ཤིན་ཏུ་བདེ་བ་མྱོང༌།
[151] །དེ་ཡི་གནས་སུ་སངས་རྒྱས་རྣམས་བཀྱེ་ཞིང༌།
[152] །འཇིག་རྟེན་རིག་པའི་བསྟན་པ་འབར་བར་སྣང༌།
[153] །དེ་ཡི་གནས་ཉིད་ན་ཡང་ཆོས་ནུབ་ཅིང༌།
[154] །འདྲེན་པ་མྱ་ངན་འདས་པའི་སྒྲ་ཡང་ཐོས།
[155] །མི་དག་བདག་གི་ཁྱིམ་ན་རྨི་ལམ་བཞིན།
[156] །འདོད་པའི་ཡོན་ཏན་དག་ལ་དགའ་མྱོས་ནས།
[157] །ཕྱིར་སད་ནས་ཀྱང་དེ་ཡི་འདོད་མ་མཐོང༌།
[158] །དེ་ནི་རྨི་ལམ་ཡིན་པ་དེ་ཡིས་ཤེས།
[159] །དེ་བཞིན་མཐོང་ཐོས་རིག་དང་བྱེ་བྲག་ཕྱེད།
[160] །འདི་དག་ཐམས་ཅད་མི་བདེན་རྨི་ལམ་བཞིན།
[161] །ཅེས་སོ།
[162] །དེ་ལྟར་རྡུལ་གཅིག་གི་གོ་སའི་འཇིག་རྟེན་དེ་ན་སྣང་བའི་རྡུལ་ཕྲ་བ་ལའང་སེམས་ཅན་གྱི་ཞིང་དང་བཀོད་པ་ཚད་མེད་དེ།
[163] སངས་རྒྱས་ཀྱི་ཡུལ་དང་ཡེ་ཤེས་བསམ་གྱིས་མི་ཁྱབ་པ་ལ་འཇུག་པའི་མདོ་ལས།
[164] སྟོང་གསུམ་གྱི་ལྷ་དང་མི་ནི་ཆེས་ཉུང་གིས།
[165] ཕྱི་རོལ་པའི་དྲང་སྲོང་མངོན་པར་ཤེས་པ་དང་ལྡན་པ་གཅིག་གིས་ཤིང་རྟའི་འཕང་ལོ་ཙམ་གྱི་སའི་ཁྱོན་ལ་སེམས་ཅན་མཐོང་བ་ནི་དེ་བས་ཆེས་མང་ངོ༌།
[166] །དེ་དག་གིས་སྟོང་གསུམ་དུ་མཐོང་བ་བས་ནི་ཉན་ཐོས་རྣམ་པར་ཐར་པ་བརྒྱད་ལས།
[167] བསམ་གཏན་པའི་གང་ཟག་གཅིག་གིས་ཤིང་རྟའི་འཕང་ལོ་ཙམ་ལ་མཐོང་བ་ཆེས་མང་ངོ༌།
[168] །དེ་དག་གིས་སྟོང་གསུམ་དུ་མཐོང་བ་བས་ནི་བྱང་ཆུབ་སེམས་དཔའ་རྣམ་པར་ཐར་པ་བརྒྱད་ལས་བསམ་གཏན་པ་གཅིག་གིས་ཤིང་རྟའི་འཕང་ལོ་ཙམ་ལ་མཐོང་བ་ཆེས་མང་ངོ༌།
[169] །དེ་དག་གིས་སྟོང་གསུམ་དུ་མཐོང་བ་བས་ནི།
[170] དེ་བཞིན་གཤེགས་པའི་སྤྱན་གྱིས་ཤིང་རྟའི་འཕང་ལོ་ཙམ་གྱི་སའི་ཁྱོན་ལ་གཟིགས་པ་ཆེས་མང་ངོ་ཞེས་གསུངས་སོ།
[171] །དེ་ལྟར་ན་ནམ་མཁས་གར་ཁྱབ་ཀྱི་གནས་ན་སེམས་ཅན་གྱིས་ཁྱབ་ལ།
[172] དེ་དག་ལ་སངས་རྒྱས་ཀྱི་སྤྲུལ་པས་ཁྱབ་ནས་དོན་མཛད་པ་ཐམས་ཅད་ཀྱང༌།
[173] ལ་ལ་ནི་སྟོན་པ་ཀུན་ཏུ་བཟང་པོའི་སྤྲུལ་པས་དངོས་སུ་དོན་མཛད་ལ།
[174] ལ་ལར་རྒྱུད་གཞན་གྱིས་བསྡུས་པའི་སངས་རྒྱས་ཀྱིས་དོན་མཛད་པ་དེ་དག་ཀྱང༌།
[175] དང་པོ་ཀུན་ཏུ་བཟང་པོས་ལམ་བསྟན་ནས་སེམས་བསྐྱེད་དུ་བཅུག་པ་ལས་བྱུང་བས།
[176] ཀུན་ཏུ་བཟང་པོས་རྒྱུད་ནས་དོན་མཛད་པ་ཞེས་བྱའོ།
[177] །འདི་དག་བསྟན་བཅོས་འགའ་ཞིག་ཏུ།
[178] སངས་རྒྱས་ལས་ཆོས་ཆོས་ལས་འཕགས་པའི་ཚོགས།
[179] །ཞེས་འབྱུང་བའི་གོ་རིམ་གྱི་དགོངས་པ་ཡིན་ནོ།
[180] །གཉིས་པ་ཞིང་འདིར་རྡོ་རྗེ་འཆང་གིས་ཞི་བ་བརྙེས་པའི་ཚུལ་ལ།
[181] ཁ་ཅིག་ཐོག་མའི་མགོན་པོ་ལས།
[182] རྒྱུད་ཐ་དད་པའི་སེམས་ཅན་གྱིས་སེམས་བསྐྱེད་པ་ཡིན་ནོ་ཞེས་འདོད་པ་ནི་མི་རིགས་ཏེ།
[183] སྒྱུ་འཕྲུལ་རྡོ་རྗེ་ལས།
[184] དང་པོའི་རྒྱལ་བ་ས་རྒྱལ་བ།
[185] །སེམས་དཔའ་ཆེ་དང་འཇོམས་པ་ཆེ།
[186] །ཞེས་དང༌།
[187] འདི་ནི་སྟོན་པ་རྡོ་རྗེ་འཆང༌།
[188] །སངས་རྒྱས་ཀུན་བས་སྔར་སངས་རྒྱས།
[189] །ཞེས་གསུངས་པ་དང་འགལ་བའི་ཕྱིར་རོ།
[190] །དེས་ན་ཞིང་འདིའི་འགྲོ་བའི་དོན་དུ་ཐོག་མའི་དུས་སུ་ཀུན་ཏུ་བཟང་པོ་ཉིད་ཀྱི་རྣམ་འཕྲུལ་གྱིས་སོ་སོ་སྐྱེ་བོས་སེམས་བསྐྱེད་པ་ལྟར་བསྟན་ནས་སངས་རྒྱས་པའི་ཚུལ་དང་མཛད་པ་ཆེན་པོའི་རོལ་པ་བཀྱེ་བ་ཡིན་ནོ།
[191] །འདི་ལྟ་བུའི་ཚུལ་ནི་ཐུན་མོང་གི་མདོ་སྡེ་དག་ལས་ཀྱང་གྲགས་ཏེ།
