[15246] རླུང་ལ་ཤིས་པའི་ཁ་ཟས་དང་ཆང་བཟང་པོ་བག་རེ་བསྟེན་ལ།།མིག་་གི་ཕུགས་ཀྱི་གཡས་གཡོན་ན།
[15247] །ཉ་མོ་ལྟ་བུའི་རྩ་རེ་སྟེ།
[15248] །ཁ་དོག་དམར་པོ་འོད་ཟེར་འབར།
[15249] །དེ་གཉིས་ལ་ཡང་བཅུ་བཅུའོ།
[15250] །ཙཀྲ་དེ་ཡི་མདུན་རྒྱབ་ནས།
[15251] །ཟླ་བ་ལྟ་བུའི་རྩ་རེ་སྟེ།
[15252] །རྩེ་ལྟག་གཉིས་སུ་རྣམ་པར་འཐོན།
[15253] །འོད་ཟེར་དཀར་པོ་དྲི་མེད་ཅིང༌།
[15254] །དེ་གཉིས་ལ་ཡང་བཅུ་བཅུའོ།
[15255] །དེ་ནས་འཁོར་ལོའི་མཚམས་ནས་ནི།
[15256] །ཉ་མིག་ལྟ་བུའི་རྩ་ཆེན་ནི།
[15257] །རྣམ་པར་འཁྱིལ་བ་རེ་རེ་སྟེ།
[15258] །སྣ་ཡི་གཡས་གཡོན་ཙཀྵུའི་ཕུགས།
[15259] །རྣམ་པར་གནས་པ་རེ་རེའོ།
[15260] །དེ་གཉིས་ལ་ནི་ལྔ་ལྔའོ།
[15261] །འཁོར་ལོ་རིམ་པ་ལྔ་པ་ལས།
[15262] །སྒྱུ་མ་ལྟ་བུའི་རྩ་ཆེན་ནི།
[15263] །རྣམ་པར་འཁྱིལ་བའི་སྡོང་པོ་གཅིག
[15264] །ཁ་ནི་སྡོང་པོ་གཉིས་སུ་ལྡན།
[15265] །གཅིག་ནི་པདྨ་ལྟ་བུ་སྟེ།
[15266] །ལྕེ་ཡི་སྟེང་ན་རྣམ་པར་གནས།
[15267] །དེ་ལ་ཡང་ནི་གཉིས་གཉིས་སོ།
[15268] །བྱེ་རུའི་མདོག་དང་ལྡོང་རོས་མདོག
[15269] །རྣམ་པ་གཉིས་ནི་གནས་པའོ།
[15270] །དེ་དག་ལི་ནི་བཅུ་བཅུའོ།
[15271] །དེ་ཡི་རྩ་ཡི་སྟོང་ཟླ་ནི།
[15272] །འདབ་མ་ལྡན་པ་གསུམ་ལྡན་ཏེ།
[15273] །ཁ་དོག་ནག་ཅིང་ལྕགས་ཀྱུའི་དབྱིབས།
[15274] །རྣམ་པ་གསུམ་ནི་སྒྲིལ་མའོ།
[15275] །དེ་དག་ལ་ཡང་བཅུ་བཅུའོ།
[15276] །རྩ་ཡི་རྣམ་པ་དེ་དག་ནི།
[15277] །འབྱུང་བའི་ཐབས་དང་གནད་ཀྱི་རྒྱུ།
[15278] །སྒྲུབ་པ་པོས་ནི་ཤེས་པར་བྱ།
[15279] །རླུང་དང་ཐིག་ལེ་འདུ་གནད་དང༌།
[15280] །ཤེས་པའི་རྟ་དང་དངས་མ་ནི།
[15281] །རྣམ་པ་ལྔ་ལྔར་ཤེས་པར་བྱ།
[15282] །རླུང་དང་མཁྲིས་པ་བད་ཀན་དང༌།
[15283] །འདུས་པ་དང་ནི་རྣམ་པ་བཞི།
[15284] །འདོད་ཆགས་སྡེ་དང་གཏི་མུག་དང༌།
[15285] །ཁོང་ཁྲོ་ལས་ནི་འགྱུར་བ་སྟེ།
[15286] །དེ་དག་རྣམས་ཀྱི་མཚན་ཉིད་ནི།
[15287] །རེ་རེ་ལ་ཡང་གསུམ་གསུམ་མོ།
[15288] །ཡུལ་དང་དུད་ཚོད་གནས་རྣམས་དང༌།
[15289] འཇུག་པར་བྱེད་པ་རྣམས་ཡིན་ནོ།
[15290] །རླུང་ལ་རྣམ་པ་གཉིས་ཡིན་ཏེ།
[15291] །ཡེ་ཤེས་རླུང་དང་ལས་རླུང་ངོ༌།
[15292] །ཚ་བ་དང་ནི་གྲང་བའོ།
[15293] །ཡེ་ཤེས་རླུང་ནི་བཞི་ཡིན་ནོ།
[15294] །རིགས་དང་ཡེ་ཤེས་ཁ་དོག་དང༌།
[15295] །ཐིག་ལེ་དངོས་ཀྱི་རླུང་ཡིན་ནོ།
[15296] །ཐིག་ལེ་ལ་ཡང་གཉིས་ཡིན་ནོ།
[15297] །ལས་ཀྱི་རྒྱུ་དང་ཡེ་ཤེས་སོ།
[15298] །ལས་ལ་རྣམ་པ་གཉིས་ཡིན་ནོ།
[15299] །འཛིན་པའི་རྒྱུ་དང་སྐྱེད་པའི་རྒྱུ།
[15300] །ཡེ་ཤེས་ལ་ནི་གཉིས་ཡིན་ནོ།
[15301] །རླུང་གི་གདངས་དང་འོད་ཀྱི་རྒྱུ།
[15302] །འདུ་བ་རྣམས་ནི་གཉིས་ཡིན་ཏེ།
[15303] །ཚང་དང་གྲང་བ་གཉིས་ཡིན་ནོ།
[15304] །ཤེས་པའི་རྟ་ཡང་དེ་དང་འདྲ།
[15305] །སྡུད་བྱེད་ཡིད་དང་ཤེས་རབ་བོ།
[15306] །དངས་མ་ཡང་ནི་རྣམ་གཉིས་སོ།
[15307] །ཁྲག་དང་ཆུ་སེར་ཤ་རྣམས་དང༌།
[15308] །འོད་དང་རླུང་གི་ཁ་དོག་གོ
[15309] །དེ་ལྟར་ཀུན་ལ་ཤེས་པར་བྱ།
[15310] །ཞེས་སོ།
[15311] །གསུམ་པ་གནད་བྱེ་བྲག་ཏུ་བསྟན་པའང༌།
[15312] དེ་ཉིད་ལས།
[15313] སྤྱི་གཙུག་བདེ་བའི་ཙཀྲ་ལ།
[15314] །ལྷག་མ་ཁྱབ་ཀྱིས་འདབ་མ་རྣམས།
[15315] །ཀུན་ལ་སྣང་བ་ཡིན་པ་ལ།
[15316] །རྩ་ཡི་རྣམ་པ་དེ་དག་ལ།
[15317] །གནས་པའི་རླུང་དང་ཐིག་ལེ་དང༌།
[15318] །ཤེས་པའི་རྟ་ནི་འདི་ལྟར་ཡོད།
[15319] །གཏེམ་དུ་རུང་བའི་རྩ་བཞི་ལ།
[15320] །ཤར་དུ་རུང་བའི་རྩ་བཞི་ཡོད།
[15321] །སྣང་བའི་ཐིག་ལེ་བཞི་གནས་པའོ།
[15322] །གཏེམ་དུ་རུང་བའི་རླུང་སྒོ་ནས།
[15323] །འཇུག་པའི་རྩ་ནི་ལྔ་བཅུའོ།
[15324] །ཐོབ་པའི་འབྲས་བུ་དེ་ཉིད་དོ།
[15325] །རབ་ཏུ་བརྟན་པའི་རྩ་ལྔ་ལས།
[15326] །ཤེས་པའི་དངས་མ་རེ་རེའོ།
[15327] །དེ་གཏེམས་དྲོད་ཀྱི་གདམས་ངག་རྫོགས།
[15328] །གནས་པའི་རྩ་ནི་སུམ་ཅུ་གཉིས།
[15329] །གསལ་བའི་རླུང་ནི་སུམ་ཅུ་གཉིས།
[15330] །ཆ་དང་ཆ་ཡི་སྣང་བ་ལས།
[15331] །འདི་དག་རྣམས་ནི་རྫོགས་པའོ།
[15332] །གཏེམས་ན་ཟློག་པར་འགྱུར་བ་ཡི།
[15333] །རྩ་ནི་ལྔ་བཅུ་འདི་དག་ལས།
[15334] །རྨུགས་པའི་རླུང་ནི་རེ་རེའོ།
[15335] །འཇུག་པའི་རླུང་ནི་དེ་དག་ལས།
[15336] །ལྷག་མ་རྣམས་ནི་གཞི་ལ་གནས།
[15337] །རྩ་ཡི་རྣམ་པ་དེ་དག་ལ།
[15338] །འཕར་བའི་རྩ་ནི་སུམ་གཉིས།
[15339] །རླུང་གི་གདངས་ཀྱང་དེ་ཙམ་མོ།
[15340] །རྩ་དང་རླུང་གི་ཆ་དང་ནི།
[15341] །འཁྲུལ་པའི་རླུང་ཡང་དེ་ཙམ་མོ།
[15342] །གནས་པའི་རླུང་ནི་བདུན་ཕྲག་གཉིས།
[15343] །ལུས་ལ་གནས་ཤིང་རྫོགས་པའོ།
[15344] །བདེ་བའི་རྩ་ནི་དེ་དག་ལས།
[15345] །སྣང་བ་དང་ནི་མི་སྣང་བའི།
[15346] །འཁྲུལ་པའི་རྟེན་ཡང་དེ་ཙམ་མོ།
[15347] །གསེང་ན་སྐྱོན་འགྱུར་ཉི་ཤུ་བརྒྱད།
[15348] །མ་གསེང་སྐྱོན་འགྱུར་བཅུ་དགུའོ།
[15349] །བསྲེགས་ན་སྐྱོན་འགྱུར་བདུན་ཡོད་དོ།
[15350] །མ་བསྲེགས་སྐྱོན་འགྱུར་ཉི་ཤུའོ།
[15351] །ཞེས་སོ།
[15352] །བཞི་པ་རྩའི་འཁོར་ལོ་འདིར་འདུ་བའི་ནད་ཇི་ལྟར་བཅོས་པའི་མན་ངག་ནི།
[15353] །གནད་ཀྱི་ཡི་གེ་ལས།
[15354] སྤྱི་བོའི་གཙུག་ནི་དྲང་པོ་ན།
[15355] །ཡི་གེ་ཨ་ནི་གནས་པ་སྟེ།
[15356] །སྨན་སྐད་གཉན་གྱི་རེ་ཐག་ཡིན།
[15357] །འདི་གཏར་སྙིང་གི་རིམས་ནད་ཕན།
[15358] །འདི་ནས་གཡས་ཀྱི་ཚོན་གང་ནས།
[15359] །ཡི་གེ་ཀཱ་ནི་གནས་པ་སྟེ།
[15360] །འདི་མིང་བོ་དྷི་གཡས་པ་ཞེས།
[15361] །འདི་གཏར་གཞོགས་གཡས་ན་ལ་ཕན།
[15362] །དེ་གཡོན་ཚོན་གང་ཀཱི་ཡིག་སྟེ།
[15363] །འདི་མིང་བོ་དྷི་གཡོན་པ་ཞེས།
[15364] །འདི་གཏར་གཞོགས་གཡོན་ན་ལ་ཕན།
[15365] །དེ་ནས་སོར་བཞི་མཚོག་མའི་སྟེང༌།
[15366] །ཀི་གནས་རལ་མིག་ཆེན་མོ་ཞེས།
[15367] །འདི་གཏར་མཚོག་མ་འཐིབ་ལ་ཕན།
[15368] །དེ་གཡས་ཚོན་གང་བཅལ་བ་ན།
[15369] །ཀུ་གནས་དུང་རྩ་འཁོར་མ་ཞེས།
[15370] །འདི་གཏར་ལྡན་གཡས་ནད་ལ་ཕན།
[15371] །དེ་འོག་ཚོན་གང་ཀླུ་ཡིག་སྟེ།
[15372] །དུང་རྩ་འགྲེང་དུ་ཞེས་བྱ་བ།
[15373] །འདི་གཏར་ལྡན་གཡོན་ནད་ལ་ཕན།
[15374] །དེ་ནས་ཚོན་གང་མདུན་ངོས་ན།
[15375] །ཀྱེཧྱ་གནས་རལ་མིག་ཆུང་ངུ་ཞེས།
[15376] །འདི་གཏར་མིག་གི་ནད་ལ་ཕན།
[15377] །དེ་ཡི་གཡས་ཀྱི་ཚོན་གང་ན།
[15378] །ཀྱེཧྱ་གནས་མཁྲིས་རྩ་སེར་པོ་ཞེས།
[15379] །འདི་གཏར་དང་ག་འགྲིབ་ལ་ཕན།
[15380] །དེ་ཡི་གཡོན་ནས་ཚོན་གང་ན།
[15381] །ཀེ་གནས་མཁྲིས་པའི་དོང་བུ་ཞེས།
[15382] །འདི་གཏར་ཁ་ཁའི་ནད་ལ་ཕན།
[15383] །དེ་ནས་དཔྲལ་བའི་དབུས་ན་ནི།
[15384] །ཀཱཻ་གནས་མཁྲིས་རྩ་སེར་མདུང་ཡིན།
[15385] །འདི་གཏར་ཆང་གི་ནད་ལ་ཕན
[15386] །དེ་གཡས་མིག་གི་ཕུགས་ནས་གྱེན།
[15387] །མཚོན་བྱེད་ཀཱོ་གནས་མིག་རྩ་འགྲེང༌།
[15388] །འདི་གཏར་ལིང་ཐོག་ནད་ལ་ཕན།
[15389] །གཡོན་པའི་ཕུག་ན་ཀོ་ཡིག་གནས།
[15390] །མིག་རྩ་འཕར་མ་ཞེས་བྱ་བ།
[15391] །འདི་གཏར་ཚད་པའི་ནད་ལ་ཕན།
[15392] །དེ་ལས་གྱེན་དུ་ཚོན་གང་གཡས།
[15393] །ཀཱ་གནས་མིག་རྩ་འཕྲེད་ཉལ་ཞེས།
[15394] །གཡོན་འཕྲེང་ཀཾ་གནས་མི་གཏར་རོ།
[15395] །དེ་ནས་ཕར་བཅལ་རྩེ་འདང་གཡས།
[15396] །ཐ་གནས་གཏར་བས་ཀླད་ནད་ཕན།
[15397] །རྩེ་འདང་གཡོན་ན་ཡི་གེ་ཐཱ།
[15398] །གཏར་བས་མིག་ནི་ལུག་ལ་ཕན།
[15399] །མིག་ཕུགས་ན་ཐི་གནས་ཏེ།
[15400] །འདི་མིང་མིག་རྩ་ལྡུམ་བུ་ཡིན།
[15401] །འདི་གཏར་མིག་རྩའི་ནད་ལ་ཕན།
[15402] །གཡོན་གྱི་ཕུགས་ན་ཡི་གེ་ཐཱི།
[15403] །འདི་མིང་མིག་རྩ་གང་པོ་ཞེས།
[15404] །འདི་གཏར་མིག་ནི་རྣོ་བར་བྱེད།
[15405] ལྕེ་སྟེང་གཡས་ཕྱོགས་ཡ་རྐན་ཕྱོགས།
[15406] །དཀར་བའི་་རྩ་ཆེན་གཉིས་ན་ནི།
[15407] །ཐུ་དང་ཐཱུ་གཉིས་འཁྲིལ་བའི་ཚུལ།
[15408] །འདི་མིང་ལྕེ་རྩ་ནག་པོ་ཞེས།
[15409] །འདི་གཏར་ལྕེ་ཕུགས་བྲོས་ལ་ཕན།
[15410] །རྣལ་འབྱོར་པ་ཡིས་གདོན་ཞིང་བཟླས།
[15411] །འདིས་ནི་འབྱུང་བ་འཁྲུགས་པ་སེལ།
[15412] །ཅེས་སོ།
[15413] །གསུམ་པ་ཉམས་སུ་བླང་ཚུལ་ལ།

[15415] དང་པོ་ལ་གནད་བཞི་ལས་འདུག་སྟངས་ཀྱི་གནད་ནི།
[15416] སྐུ་གསུམ་གྱི་འདུག་སྟངས་ལ་གནས་པས་ཆོས་ཉིད་ཀྱི་བདེན་པ་ཁོང་དུ་ཆུད་ནས་འཁོར་བའི་ཆོས་ལས་གྲོལ་བའོ།
[15417] །ཞིབ་ཏུ་འོང་ཏུ་འཆད་དོ།
[15418] །རྩའི་གནད་ནི།
[15419] གཅུ་བ་དང་གཏེམས་པ་དང་དཀྲུགས་པ་ལ་སོགས་པ་མཐོང་བ་བརྒྱུད་པ་བཞིན་ནོ།
[15420] །སྒོའི་གནས་ནི་གཞིགས་སྟངས་གསུམ་སྟེ།
[15421] འཁྲུལ་སྣང་འོད་གསལ་དུ་བསྒྱུར་བའོ།
[15422] །གཉིད་དང་རྨི་ལམ་རྒྱས་འདེབས་པ་ནི་ཉལ་ལུགས་ཀྱི་གནད་ཅེས་པས་འགྲུབ་སྟེ།
[15423] སེང་གེ་ལྟ་བུར་ཉལ་བས་གཉིད་འོད་གསལ་དུ་འཆར་བའི་གཞི་བྱེད།
[15424] །གླང་པོ་ལྟ་བུར་ཉལ་བས་རྨི་ལམ་གྱི་རྩ་བ་སྟོངས་པའི་ལམ་སེལ།
[15425] དེ་ཡང་གཏི་མུག་གི་ཉལ་ལུགས་སྤངས་པས་འོད་གསལ་མཉམ་པའི་དོན་སྐྱེའོ།
[15426] །འདོད་ཆགས་ཀྱི་ཉལ་ལུགས་སྤངས་པས་འདོད་ཆགས་སེལ་བའི་གཞི་རྙེད།
[15427] ཞེ་སྡང་གི་ཉལ་ལུགས་སྤངས་པས་ཆོས་ཉིད་མཉམ་པའི་སྙོམས་པའོ།
[15428] །དེའང་རིམ་པ་ལྟར་ཁ་སྦུབ་པ་དང༌།
[15429] གན་རྐྱལ་དང༌།
[15430] གཡས་སྟེང་དུ་གདོན་པ་སྤངས་པའོ།
[15431] །དེ་ལྟར་སྤྱིའི་དོན་བསྟན་ནས།
[15432] ད་ནི་བྱེ་བྲག་ཏུ་ཉམས་ལེན་དངོས་ནི།
[15433] སྔོན་འགྲོ་དང་དངོས་གཞི་གཉིས་ལས།
[15434] སྔོན་འགྲོ་ནི།
[15435] ཉི་མ་གསུམ་བཞི་ཙམ་དུ་ལུས་དལ་བར་བྱའོ།
[15436] །དེ་ནས་ཉི་མ་གསུམ་བཞིར་ལུས་ལ་མར་སྣ་ཚོགས་པའི་བསྐུ་མཉེ་བྱ་ཞིང་ཉི་མ་ལ་བསྲེག
[15437] །རྐང་མཐིལ་ལ་ཆང་བཟང་པོ་བྱུགས་ནས་དྲག་ཏུ་མཉེ།
[15438] དེ་ནས་ཞག་གསུམ་དུ་ཡན་ལག་ཐང་བཅད་ལ་གན་རྐྱང་ལ་དུ་ཉལ་ལོ།
[15439] །དེས་རྩ་ཐམས་ཅད་ལངས་ནས་འོང་སྟེ།
[15440] འཕར་བ་དང༌།
[15441] རྒལ་བར་འགྱུར་རོ།
[15442] །སྐྱེས་པའི་གཡས་འཕར་དྲག་པ་ལ།
[15443] ཙནྡན་དང་སེང་ལྡེང་གི་ཐང་ཆུ་བྱུགས་ལ་གྱོ་མོ་འཇམ་པོས་ཕྱི་ནས་བསྲོའོ།
[15444] །དེ་ལྟར་བྱ་བ་གནད་ཆེའོ།
[15445] །གཉིས་པ་དངོས་གཞི་ནི།
[15446] རླུང་ལ་ཤིས་པའི་ཁ་ཟས་དང་ཆང་བཟང་པོ་བག་རེ་བསྟེན་ལ།
[15447] སྐྱེས་པ་རྣམས་ཀྱིས་གཞོགས་གཡོན་འོག་ཏུ་བཅུག་ནས་ཉམ་ལ།
[15448] བུད་མེད་རྣམས་ཀྱིས་གཡས་འོག་ཏུ་བཅུག་ནས་ཉལ་བས་རྩ་སོ་སོའི་ལས་འགྱུར་བར་བྱེད་པའི་གནད་དོ།
[15449] །དེས་རྩ་འཕར་ཞིང་ལངས་ནས་འོང་སྟེ།
[15450] གང་འབར་བསྐྱེན་ཅིང་དྲག་པ་རྣམས་ལ་རུ་རྟའི་མེ་རྩ་འཇམ་ཙམ་བྱའོ།
[15451] །ལག་པའི་སྲིན་ལག་གིས།
[15452] གཡས་ཀྱི་འཕར་རྩ་གང་དྲག་པ་ཀུན་མནན་ཏེ།
[15453] ཐོག་མ་ནེམ་ཙམ་པ་ལས་ཇེ་དྲག་ཇེ་དྲག་ལ་མནན་དུས།
[15454] རླུང་ནང་དུ་བསྟིམས་ལ་རེས་ཕྱིར་ལ་དྲག་ཏུ་ཕུལ།
[15455] རེས་ནང་དུ་རྔུབ་པ་དྲག་ཏུ་བྱའོ།
[15456] །དེ་ལྟར་བྱས་པས་ཤེས་པ་ཉོགས་ཀྱིས་སོང་དུས་ལངས་ལ་མགྲིན་པ་གཡས་སུ་གཅུས་ལ།
[15457] ལྐོག་མ་དཔུང་པ་གཡས་པའི་སྟེང་དུ་བཞག་སྟེ།
[15458] ལག་པ་གཡས་པས་མགྲིན་པའི་ཕྱི་རོལ་ནས།
[15459] རྩ་འཕར་ཞན་པ་གྱེན་དུ་རྒྱུ་བ་ཅིག་ཡོད་དེ།
[15460] དེ་ལ་ཐོག་མ་སྲན་མ་བདུན་གྱི་ལྗིད་ཙམ་དུ་མནན་ནོ།
[15461] །དེ་ནས་བགས་ཀྱིས་ཇེ་དྲག་ལ་བཅུ་བཞི་དང༌།
[15462] ཉེར་གཅིག་གི་ལྗིད་ཀྱིས་མནན་པས།
[15463] །དང་པོ་བན་བུན་དུ་འགྱུར་ཏེ།
[15464] ལས་དང་ཉོན་མོངས་པའི་རྩ་སྤོང་བར་བྱེད་པའི་ཉམས་སོ།
[15465] །དེ་ནས་སང་སེང་དུ་སྣང་བ་དག་འོང་བ་ནི་ཡེ་ཤེས་ཀྱི་མཐོང་བ་རང་བཞིན་གྱི་ཡུལ་ལོ།
[15466] །དེ་ནས་ཁྲལ་ཁྲོལ་དུ་སྣང་བ་ནི།
[15467] ཡེ་ཤེས་རྣམ་པར་དག་པའི་ལམ་མངོན་དུ་གྱུར་པའི་ཉམས་སོ།
[15468] །དེ་ལྟར་འགྱུར་བའི་དུས་ན་སྐབས་སུ་བླ་མ་ལ་གསོལ་བ་གདབ་པ་དང་ལམ་ཅི་རིགས་སུ་བྱའོ།
[15469] །དེ་ལྟར་མ་བྱས་ན་རླུང་སྲོག་པ་རྩར་ཤོར་ནས་ཤེས་པ་འུར་བ་དག་འོང་སྲིད་དོ།
[15470] །དེ་ནས་ཐུན་དུ་བཅད་ལ་ཡུན་རིང་དུ་བྱས་པས།
[15471] འོད་གསལ་ཆེན་པོའི་ཉམས་རེས་འཇོག་དང་བཅས་པ་འཆར་རོ།
[15472] །དེ་ལྟར་རྟག་ཏུ་བྱས་པས་ལུས་སྦྲིད་ཅིང་རེག་པ་ཉམས་པར་འགྱུར་བ་ནི་རྩའི་ཡུལ་སྟོངས་པ་ལས་བྱུང་བས།
[15473] མར་ནག་དང་གུ་གུལ་སྦྱར་བའི་བསྐུ་མཉེ་བྱའོ།
[15474] །སེམས་ཀྱི་རྣམ་རྟོག་སྣ་ཚོགས་ཀུན་འགགས་ནས་འོད་གསལ་ཆེན་པོའི་ཉམས་ལ་གནས་པ་ནི།
[15475] ཡེ་ཤེས་ཆེན་པོའི་ཐིག་ལེ་གནད་ལ་ཕེབས་པ་ལས་བྱུང་སྟེ།
[15476] འདིའི་དུས་སེམས་འབྱུང་བ་བཞིའི་དྲ་བར་གཞུག་པར་བྱ་ཞིང༌།
[15477] རླུང་ནང་དུ་དགུག་པ་ནི་མན་ངག་གོ
[15478] །འདི་ལ་བདེ་བ་སྐྱེ་བ་དང༌།
[15479] སྐྱེས་པ་དང༌།
[15480] བརྟན་པ་དང༌།
[15481] ཁྱད་པར་གྱི་ཡེ་ཤེས་ལམ་དུ་འབྱོར་པ་ལ་སོགས་པར་འབྱུང་བ་ནི།
[15482] བརྩོན་འགྲུས་ཀྱི་རིམ་པ་ལས་བྱུང་བའོ།
[15483] །བུད་མེད་རྣམས་ལ་གཡོན་ནས་བྱའོ།
[15484] །དེ་ཡང་རྩའི་ཡུལ་ལ་བརྟེན་ནས་ཉམས་སུ་བླང་བ་ནི།
[15485] གཏེམ་པ་དང༌།
[15486] མནན་པ་དང༌།
[15487] བསྐྱིལ་བ་དང༌།
[15488] གཟིར་བ་དང༌།
[15489] གཅུན་པ་དང༌།
[15490] གཞིལ་བ་དང༌།
[15491] སྦ་བ་རྣམས་སུ་ཤེས་པར་བྱ་སྟེ།
[15492] དེ་ལ་སྦ་བ་ནི་རོ་མའི་གནད།
[15493] གཅུན་པ་ནི་རྐྱང་མའི་གནད།
[15494] ཀུན་འདར་མ་གཞིལ་བའི་གནད།
[15495] རྩ་ཕྲན་རྣམས་ལ་གཏེམ་པ་ལུས་ཀྱི་གནད་དུ་བྱ་བ་དང༌།
[15496] མནན་པ་ངག་གི་གནད་དུ་བྱ་བ་དང༌།
[15497] བསྐྱིལ་པ་སེམས་ཀྱི་གནད་དུ་བྱ་བ་དང༌།
[15498] གཟིར་བ་སྣང་བའི་གནད་དུ་བྱ་བ་ཡིན་ནོ།
[15499] །འདིར་ཡང་ལུས་ཀྱི་གནད་ཀྱིས་རྩའི་ཡུལ་བཙལ་བ་དང༌།
[15500] རླུང་གི་གནད་ཀྱིས་རྩའི་ལམ་བསལ་བ་དང༌།
[15501] རྩའི་འགྱུ་ཚད་ཀྱིས་སྐྱེ་ཤིའི་ཁུངས་བརྟག་པ་དང༌།
[15502] རྩའི་ངོ་བོས་ཆོས་ཉིད་སྐྱེ་བའི་དུས་བཟུང་བ་རྣམས་སུ་ཤེས་པར་བྱའོ།
[15503] །རྩའི་འཕེལ་བ་ནི།
[15504] སྐྱེས་བུ་ལུས་ལ་འབྱུང་བའི་ནད་མེད་པ་ན་ཚོད་དར་ལ་བབ་པ་ཅིག་ལ།
[15505] ཉིན་ཞག་ཕྲུགས་གཅིག་ལ།
[15506] རྩའི་འགྱུ་ཚད་དྲུག་ཁྲི་བཞི་སྟོང་དང་བརྒྱད་བརྒྱ་འགྱུ་ལ།
[15507] དེ་ཡང་འཕར་བ་དྲག་ཞན་གྱི་སྟོབས་ལ་ནད་དང་གདོན་བརྟག
[15508] །རྩའི་གྲངས་ལས་ཚེའི་དུས་བརྟག
[15509] །རྩའི་འགྱུར་བྱེད་ཀྱི་འཕེལ་འགྲིབ་ལས་ནད་དང་སྡུག་བསྔལ་གྱི་བྱེ་བྲག་བརྟག
[15510] །རྩའི་སྦོམ་ཕྲའི་བྱེ་བྲག་ལས་ལས་དང་ཉོན་མོངས་པའི་དུས་བརྟག
[15511] རྩའི་དལ་པ་དང་མྱུར་བ་ལས་ཡེ་ཤེས་སྐྱེ་བའི་དུས་ངོས་བཟུང༌།
[15512] རྩ་སོ་སོའི་ཆག་གན་གྱི་མལ་ལས་ལུས་ངག་ཡིད་ཀྱི་ལས་བརྟག་པའོ།
[15513] །དེ་དག་ཀྱང་ལུས་དང༌།
[15514] དུས་དང༌།
[15515] ཡུལ་གསུམ་དུ་མ་སྦྱར་ན་གཏན་ལ་མི་ཕེབས་སོ།
[15516] །གྲངས་མང་བ་དང༌།
[15517] ཉུང་བ་དང༌།
[15518] འཕེལ་འགྲིབ་ཀྱིས་རླུང་མཁྲིས་བད་ཀན་གྱི་ནད་ངོས་གཟུང་བ་དང༌།
[15519] ནམ་གཞག་གཏན་གྱི་ཚེ་དང༌།
