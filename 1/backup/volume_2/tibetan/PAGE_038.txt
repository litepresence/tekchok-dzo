[1944] །ཕོ་མོ་གཉིས་ནི་གང་ཡང་རུང༌།
[1945] །ཁ་དོག་འདི་དང་ལྡན་པ་ནི།
[1946] །རྫོགས་པ་ཆེན་པོའི་སྣོད་དུ་གསུངས།
[1947] །ཡན་ལག་རགས་ཤིང་ཁ་དོག་ནས།
[1948] །སོ་ཉིད་ཐགས་བཟངས་དཀར་ལ་འདྲིལ།
[1949] །མིག་རྩ་ཡིད་ཙམ་དམར་བ་དང༌།
[1950] །སྐྲ་ཡི་མཚན་ཉིད་ཆེན་པོ་ཡང༌།
[1951] །གཡས་སུ་འཁྱིལ་ཞིང་ཁམ་ནག་མདོག
[1952] །ལུས་ལ་བཀུར་སྟི་ཆུང་བ་དང༌།
[1953] །ཕྱི་ཡི་སྤྱོད་ལམ་ཐ་མལ་མདོག
[1954] །ཚིག་རྣམས་ཤུགས་བྱུང་སྨྲ་བ་དང༌།
[1955] །ཡང་ན་ཚིག་གི་རྣམ་པ་ཀུན།
[1956] །གཅིག་གིས་གང་ལྟར་སྨྲས་པ་ལྟར།
[1957] །སྐྱེས་བུ་དེ་ཡང་དེ་ལྟར་སྨྲ།
[1958] །རྫོགས་ཆེན་གདམས་ངག་བསྟན་པར་གསུངས།
[1959] །དེ་རྣམས་གཅིག་ལ་ཚང་བ་ན།
[1960] །ཤན་པའི་རིགས་སམ་སྨད་འཚོང་རིགས།
[1961] །ཕྱགས་དར་ཁྲོད་པའི་རིགས་ཀྱང་རུང༌།
[1962] །རིགས་ངན་གང་དག་སུ་ཡིན་ཀྱང༌།
[1963] །གསང་བའི་མན་ངག་བཅུད་འདི་གཏད།
[1964] །རྒྱུ་ཆུང་ཡོ་བྱེད་མེད་པ་ན།
[1965] །རང་གི་ཡོ་བྱད་སྦྱར་ནས་ཀྱང༌།
[1966] །ལྷག་མ་མེད་པར་ཡོངས་སུ་སྦྱིན།
[1967] །གསང་བའི་བཅུད་གསུམ་མན་ངག་འདི།
[1968] །རིགས་དང་ཁ་དོག་བརྟག་ཐབས་རྣམས།
[1969] །ང་ནི་མྱ་ངན་འདས་འོག་ཏུ།
[1970] །གསང་བའི་མན་ངག་ཡང་བཅུད་འདི།
[1971] །ངེས་པའི་རྒྱལ་པོ་འདི་ལྟར་བསྟན།
[1972] །དེ་རྣམས་བསྟན་པའི་སྐྱེས་བུ་ལ།
[1973] །གོང་གི་གདམས་ངག་བསྟན་པར་བྱ།
[1974] །ཞེས་གསུངས་སོ།
[1975] །དོན་གཉིས་པ་ཇི་ལྟར་ཉམས་སུ་ལེན་པའི་ཚུལ་ངེས་པ་ལ་བསྒྲུབ་ལུགས་གཉིས་ཏེ།
[1976] སྤྲང་པོའི་ཚུལ་གྱིས་བསྒྲུབ་པ་དང༌།
[1977] རི་དྭགས་ཀྱི་ཚུལ་གྱིས་བསྒྲུབ་པའོ།
[1978] །དང་པོ་ནི།
[1979] རང་གི་ཡུལ་སྤངས་ནས།
[1980] གཞན་གྱི་ཡུལ་ལྗོངས་སུ་སོང་སྟེ།
[1981] སུས་ཀྱང་རང་གི་མིང་དང་བཟང་ངན་དང༌།
[1982] །ཅི་ཤེས་པ་དང་མི་ཤེས་པ་ལ་སོགས་པ་ངོ་མི་ཤེས་པར་བྱས་ཏེ།
[1983] ཐ་མལ་དུ་བརྫུས་ནས།
[1984] ཟས་གོས་སློང་མོས་འཚོ་ཞིང༌།
[1985] ས་ཕུག་དང་གྲོང་ལས་ཅུང་ཟད་ཕྱིན་པའམ།
[1986] དབེན་པའམ།
[1987] འགྲོན་ཁང་ལ་སོགས་པར་མངོན་མཚན་མེད་པས་བསྒྲུབ་སྟེ།
[1988] འདིའི་དུས་སུ་རང་གི་ཡོན་ཏན་མིང་ངོམ་པ་ནི་ཞེ་འདོད་མེད་པའི་རྟགས་ཡིན་ལ།
[1989] གཞན་གྱི་བླ་མ་དང་ཆོས་ལ་སོགས་པ་དྲི་བརྡ་མི་བྱེད་པ་ནི་རང་གི་མན་ངག་ལ་ཡིད་ཆེས་ཤིང་ཐེ་ཚོམ་མེད་པའི་རྟགས་ཡིན་ནོ།
[1990] །དེ་སྐད་དུ་ཡང༌།
