[8960] །སྣང་བའི་ཡུལ་དུ་མེད་པས་ན།
[8961] །རྡོ་རྗེ་ལུ་གུ་རྒྱུད་ཉིད་ལས།
[8962] །ཆ་ཤས་སྐུ་ནི་བསམ་མི་ཁྱབ།
[8963] །ཅེས་སོ།
[8964] །འདི་ནི་གོ་རྒྱུ་ཅིག་ཡོད་པས་བློ་གྲོས་ཅན་རྣམས་ཀྱིས་རིག་པར་བྱའོ།
[8965] །རང་བཞིན་དུ་རྟོགས་སླ་བས་ལུས་སྣང་ཞེས་བརྗོད་པ་ལ་འགལ་བ་མེད་པས།
[8966] ཕྱིའི་ནམ་མཁའ་ལ་དབྱིངས་རིག་དང་སྒྲོན་མ་སྣང་ཞེས་རྨོངས་པ་རྣམས་ལ་རྟོགས་སླ་བའི་ཕྱིར་བཤད་པར་བྱའོ།
[8967] །རང་གི་ནང་དུ་གོ་རྒྱུ་ནི་སྔ་མ་བཞིན་ནོ།
[8968] །ལྔ་པ་ཁྱད་པར་ངེས་ཚུལ་ནི།
[8969] རིག་པ་ལ་འགྱུར་བ་མེད་པའི་གནད་ཀྱིས།
[8970] དང་པོ་གཞི་ནས་འཕགས་ཏེ་འཁོར་བར་འཁྲུལ་ཡང་རིག་པ་ངན་དུ་སོང་བ་མེད།
[8971] དེ་ནས་བླ་མས་ངོ་སྤྲད་དེ་སྟོར་ནས་འཁྱམས་པ་དེ་ཕྱིས་རྩད་ཆོད་དུས་དགའ་བ་དང༌།
[8972] དང་པོ་སྟོར་དུས་འཁྲུལ་པའི་དུས་ཀྱི་རིག་པ་གཉིས་ལ་བཟང་ངན་ཁྱད་མེད།
[8973] མ་བསྒོམས་པའི་རིག་པ་དང༌།
[8974] བསྒོམས་དུས་ལ་ཁྱད་མེད་པ་དང༌།
[8975] འཁོར་དུས་དང༌།
[8976] འདས་དུས་ཀྱི་རིག་པ་ལ་བཟང་ངན་མེད་དེ།
[8977] གཉིད་དུ་སོང་དུས་དང༌།
[8978] སད་དུས་དང༌།
[8979] འགྲོ་ཁའི་རིག་པ་ལ་བཟང་ངན་ཁྱད་མེད་པ་བཞིན་ནོ།
[8980] །འོན་ཀྱང་སྣང་ཚུལ་བཟང་ངན་རྨི་ལམ་བདེ་སྡུག་ཙམ་ཡོད་པས།
[8981] དེ་ལ་ལྟོས་ནས་ལམ་བསྒོམ་པ་ཡིན་ལ།
[8982] ངོ་བོ་བཟང་དུ་འགྲོ་བར་རེ་བ་ནི་མ་ཡིན་ཏེ།
[8983] རིག་པ་ལ་འཕོ་འགྱུར་མེད་པའི་ཕྱིར་རོ།
[8984] །དེ་སྐད་དུའང་རང་ཤར་ལས།
[8985] ཨེ་མ་ཧོ།
[8986] །ནོར་བུ་རིན་ཆེན་བདུད་རྩི་འདི།
[8987] །ཐོག་མ་མེད་པའི་དུས་ཀྱི་ཚེ།
[8988] །ང་ཡི་ནོར་བུ་རིན་ཆེན་སྟོར།
[8989] །དེ་ཚེ་སྡུག་བསྔལ་ལོག་པར་འཁྱམས།
[8990] །ནོར་བུ་རིན་ཆེན་བདུད་རྩི་དེ།
[8991] །བཙལ་བས་རྙེད་པའི་གནས་མེད་དེ།
[8992] །རང་ལས་རང་བྱུང་རང་གིས་རྟོགས།
[8993] །ནོར་བུ་རིན་ཆེན་བདུད་རྩི་དེ།
[8994] །སྟོར་བ་མ་ཚོར་རྙེད་པ་ཚོར།
[8995] །སྟོར་བས་དངོས་པོ་མཚན་མ་མེད།
[8996] །རྙེད་པས་དགའ་བའི་མདངས་དང་ལྡན།
[8997] །རྙེད་པའི་དུས་དང་སྟོར་བའི་དུས།
[8998] །ཁྱད་མེད་གཅིག་གི་ངང་དུ་རྟོགས།
[8999] །ཞེས་སོ།
[9000] །དེའང་རང་རིག་གི་ངང་དུ་སྟོར་བ་དེ་རང་གི་ངང་ནས་རྙེད་ཀྱང༌།
[9001] རང་རིག་གཅིག་གི་འཁོར་ལོ་ལས།
