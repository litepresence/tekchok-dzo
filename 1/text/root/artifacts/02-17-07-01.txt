[3005] དང་པོ་ཆུ་དང་མེ་དང་ཁྱི་དང་དགྲ་ལ་སོགས་པའི་རྐྱེན་དྲག་པོས་འཇིགས་དངངས་སྐྱེས་པ་ལ་རྨི་ལམ་དུ་འདུག་སྙམ་དུ་ཟིན་ངེས་ཏེ་རྩུབ་ཟིན་ཞེས་བྱའོ།
[3006] །དེ་ནས་ཇི་ལྟར་རྨིས་པ་ཐམས་ཅད་ཟིན་པ་འོང་ངོ༌།
[3007] །ཟིན་དུས་སྤྲུལ་པ་ནི།
[3008] ཉིན་དུས་ཡིད་ཀྱི་ལྷ་སྐུ་དང་སྐྱེ་འགྲོ་དང་མཐོང་སྣང་མི་འདྲ་བ་དུ་མ་སྤྲུལ་ཞིང་གཅིག་ལས་མང་པོར་སྤྲུལ་པ་བྱས་པས།
[3009] རྨི་ལམ་དུ་སྤྲུལ་པ་བཞིན་མཐོང་བའོ།
[3010] །དེ་ནས་ཕན་ཚུན་ལྷ་ཀླུར་བསྒྱུར།
[3011] སྐྱེས་པ་བུད་མེད་དུ་བསྒྱུར།
[3012] གཅིག་མང་པོར་བསྒྱུར་མང་པོ་གཅིག་ཏུ་བསྒྱུར་བ་རྣམས་འདུན་པ་མ་ཡེངས་པར་ཉིན་དུས་སྦྱངས་པས་རྨི་ལམ་དུ་འབྱུང་ངེས་སོ།
[3013] །དེ་ནས་སྔར་མ་ཕྱིན་པའི་ཡུལ་དང་ཞིང་ཁམས་ལ་སོགས་པ་ཡིད་ལ་བྱས་པས་རྨི་ལམ་དུ་དེར་བགྲོད་པ་དང་ཆོས་ཐོས་པ་ལ་སོགས་པ་འབྱུང་ངོ༌།
[3014] །འདི་དག་འདུན་པ་དྲག་པོ་བྱས་ན་མྱུར་དུ་འབྱོངས་པ་ཡིན་ནོ།
[3015] །ཁ་ཅིག་ཏུ་རྨི་ལམ་བསྒོམ་པའི་དངོས་གཞིར་བཤད་ཀྱང༌།
[3016] སྐབས་འདིར་སྔོན་འགྲོར་འཐད་དེ་རྨི་ལམ་ཟིན་པ་དང་སྤྲུལ་བསྒྱུར་ཙམ་ཆོས་ཉིད་དོན་མ་ཡིན་ཞིང་བློའི་བདེན་ཞེན་ལས་མི་གྲོལ་བའི་ཕྱིར་རོ།
[3017] །སྔར་གྱི་ཕལ་ཆེར་གྱིས་འདི་བཤད་པ་མི་འདུག་ནའང་ཟིལ་གནོན་གྱི་བཤད་འགྲེལ་དང་རྨི་ལམ་གནད་ཀྱི་མན་ངག་ཏུ་གསུངས་པས་གནད་གལ་པོ་ཆེ་ཡིན་ལ།

[3018] རང་གིས་ཀྱང་ཉམས་སུ་བླངས་པས་ངེས་པ་ཡིན་ནོ།
[3019] །གཉིས་པ་དངོས་གཞི་ལ་གསུམ་སྟེ།
[3020] ཆོས་ཉིད་དོན་གྱི་གནད།

[3022] དམིགས་པའི་གནད།
[3023] རྩ་རླུང་ཆིངས་ཀྱི་གནད་དོ།
[3024] །དང་པོ་ལ་སྦྱང་བ།
[3025] བསྒྱུར་བ།
[3026] བཅད་པ་གསུམ་ལས།
[3027] ཤིན་ཏུ་དཀའ་བས་གསུམ་ཀ་བསྒོམ་ཞིང༌།
[3028] འབྲིང་གིས་གཉིས་སླ་བས་གང་རུང་རེ་རེས་ཆོག་པར་གསུངས་ན་ཡང༌།
[3029] འདི་ཁྲིད་དུ་བྱེད་དུས་ཐམས་ཅད་བསྒོམས་པས་ལག་ལེན་ལ་ཐམས་ཅད་སྦྱང་ངོ༌།
[3030] །དང་པོ་རྨི་ལམ་སྦྱང་བ་ནི་ཉིན་དུས་སྣང་བ་ཐམས་ཅད་མདང་རྨིས་པའི་རྨི་ལམ་དུ་གསལ་བའི་འདུན་པས་སྤྲུལ་བསྒྱུར་བྱས་ལ།
[3031] ཕྱི་བདེན་མེད་དུ་སྦྱང་དུས་སྦྱོང་བྱེད་རིག་པ་ལ་ཚུར་བལྟས་ཏེ་ཤར་ཤར་གཞི་མེད་རྩ་བ་དང་བྲལ་བའི་ངོ་བོ་རྗེན་པར་སལ་སལ་བསྒོམ།
[3032] ཐམས་ཅད་གཏད་མེད་བདེན་མེད་གཞི་མེད་ངོས་གཟུང་མེད་ལ་སྣང་བའི་རང་བཞིན་དུ་ངོས་བཟུང་ནས་རྩེ་གཅིག་ཏུ་བརྩོན་འགྲུས་ཐང་ལྷོད་མ་ཤོར་བར་བསྒོམས་པས།
[3033] སྣང་བ་ཙམ་དངོས་སུ་སྒྱུ་མ་རྨི་ལམ་དུ་འཆར་ཞིང་ལས་དང་བག་ཆགས་ཀྱི་འཁྲུལ་སྣང་དང་འཁྲུལ་འཛིན་ཐམས་ཅད་འཇིག་གོ
[3034] །རྨི་ལམ་དུའང་དེའི་ངང་མ་ཡེངས་པ་འབྱུང་སྟེ།
[3035] ཉིན་མཚན་ཆོས་ཉིད་ཀྱི་འཁོར་ལོ་ལས་མི་འདའ་བ་ཡིན་ནོ།
[3036] །རྨི་ལམ་བསྒྱུར་བ་ནི་སྣང་སྟོང་གཉིས་ལས།
[3037] སྣང་བ་སྒྱུར་བའི་གནད་ནི་མེ་ལོང་དུ་ལྷའི་སྐུ་བསྟན་པ་ལ་རྟེན་བྱས་ནས།
[3038] སྣང་བ་ཐམས་ཅད་ལྷའི་དཀྱིལ་འཁོར་དུ་བསྒྱུར་བ་དང༌།
[3039] སངས་རྒྱས་ཀྱི་ཞིང་དུ་བསྒྱུར་བ་དང༌།
[3040] སྒྱུ་མའི་དཔེ་བརྒྱད་དུ་བསྒྱུར་བ་རྣམས་འདུན་པ་དྲག་པོས་བྱའོ།
[3041] །སྟོང་པར་བསྒྱུར་བ་ནི་ནམ་མཁའ་ལ་སྤྲིན་དེངས་པའམ་དུ་བ་སངས་པ་ལྟར།
[3042] འཁྲུལ་སྣང་འདི་སྐད་ཅིག་གིས་སྟོང་པ་རྒྱ་ཕྱམ་དུ་ཕྱམ་ཕྱལ་བསྒོམ་པས།
[3043] རྨི་ལམ་དུའང་དེ་བཞིན་དུ་འགྱུར་བ་ཡིན་ནོ།
[3044] །རྨི་ལམ་བཅད་པ་ནི་ཇི་ལྟར་སྣང་དུས་འདི་ཉིད་ལ།
[3045] བར་སྣང་ཐོག་ཏུ་རིག་པ་འཛིན་པ་མི་འདོགས།
[3046] ཚུར་རིག་ཐོག་ཏུ་སྣང་བའི་གཟུགས་བརྙན་མི་འཛིན།
[3047] སྣང་སེམས་གཉིས་ལ་འབྲེལ་བའི་རྟོག་པས་མི་བསླད་པར།

[3048] རང་གསལ་གྱི་རིག་པ་ཁོང་གསལ་ཡངས་པར་གློད་ནས་གང་ཡང་ཡིད་ལ་མི་བྱེད་པའོ།
