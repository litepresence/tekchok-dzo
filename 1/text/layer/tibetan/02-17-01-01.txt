[22180] ད་ནི་ལམ་ངེས་པས་ཉམས་སུ་བླང་བའི་རིམ་པ་བཤད་པ་ལ།
[22181] དམིགས་པ་ཡུལ་གྱི་བློ་ཅན་རྣམས་ངེས་པར་འཇུག་པས་གོང་དུ་ཁྲིད་པའི་ཐབས་དམ་པ་དང༌།
[22182] རིག་པ་རང་སྣང་གི་བློ་ཅན་རྣམས་ཐད་དྲང་དུ་ཉམས་སུ་བླང་བའི་གནད་ངེས་པ་ལེགས་པར་བརྗོད་པས་ཐར་པ་ཆེན་པོའི་གྲོང་ཁྱེར་དུ་ཚེ་འདི་ལ་བགྲོད་པའི་ཐབས་མཆོག་ཏུ་གསང་བ་ལས།
[22183] ཐུན་མོང་དུ་དང་པོར་འཇུག་པའི་བློ་རིམ་ངེས་པ་ནས་རྩོམ་པར་སྟོན་པས།
[22184] གནས་དང་པོའི་དོན་ལ་ཆིངས་རྣམ་པ་གསུམ་སྟེ།
[22185] གང་གིས་ཉམས་སུ་ལེན་པར་བྱེད་པ་རྣལ་འབྱོར་པའི་དབྱེ་བ་དང༌།
[22186] ཇི་ལྟར་ཉམས་སུ་ལེན་པའི་ཚུལ་ངེས་པ་དང༌།
[22187] ཉམས་སུ་བླང་བའི་དོན་དངོས་བསྟན་པའོ།
[22188] །དང་པོ་ལ་གསུམ་སྟེ།
[22189] ཐུན་མོང་གི་རབ་ཏུ་དབྱེ་བ།
[22190] སོ་སོའི་དོན་བཤད་པ།
[22191] བླ་ན་མེད་པའི་གནས་ངེས་པའོ།
[22192] །དང་པོ་ནི།
[22193] རང་ཤར་ལས།
[22194] ཀྱེ་ཀྱེ་ཉེ་བའི་འཁོར་རྣམས་ཉོན།
[22195] །མཚན་མ་ཚིག་གི་རྣལ་འབྱོར་དང༌།
[22196] །རྟགས་ཙམ་འཛིན་པའི་རྣལ་འབྱོར་དང༌།
[22197] །རྗེས་སུ་རྣལ་འབྱོར་ཉིད་དང་ནི།
[22198] །ངང་ཉིད་ངེས་པའི་རྣམ་འབྱོར་དང༌།
[22199] །སྣང་བ་སེམས་ཀྱི་རྣལ་འབྱོར་དང༌།
[22200] །འཇུག་པ་ལས་ཀྱི་རྣལ་འབྱོར་དང༌།
[22201] །བྱ་བ་རྒྱུ་ཡི་རྣལ་འབྱོར་དང༌།
[22202] །བྱེད་པ་རྐྱེན་གྱི་རྣལ་འབྱོར་དང༌།
[22203] །མཐར་ཕྱིན་འབྲས་བུའི་རྣལ་འབྱོར་དང༌།
[22204] །ཡང་དག་དོན་གྱི་རྣལ་འབྱོར་དང༌།
[22205] །མཐར་ཕྱིན་གྲོལ་བའི་རྣལ་འབྱོར་དང༌།
[22206] །བྱས་པ་རྫོགས་པའི་རྣལ་འབྱོར་རོ།
[22207] །ཞེས་སོ།
[22208] །གཉིས་པ་སོ་སོའི་དོན་བཤད་པ་ལ་རྣལ་འབྱོར་པ་བཅུ་གཉིས་ལས།
[22209] དང་པོ་བརྒྱད་ཀྱིས་མི་གྲོལ་ཞིང༌།
[22210] ཕྱི་མ་བཞིས་གྲོལ་བའི་ཚུལ་ངེས་པར་བསྟན་པ་ནི།
[22211] ཡང་དེ་ཉིད་ལས།
[22212] མཚན་མ་ཚིག་གི་རྣལ་འབྱོར་ནི།
[22213] །ཚིག་ལ་ཞེན་ཅིང་ཚིག་ལ་སྤྱོད།
[22214] །མི་ཤེས་ཚིག་ལ་ལྟོས་གྱུར་ན།
[22215] །དེ་ཉིད་གྲོལ་བར་ངས་མ་བཤད།
[22216] །རྟགས་ལ་ལྟོས་པའི་རྣལ་འབྱོར་ནི།
[22217] །རང་ཉིད་རྟགས་ལ་ལྟོས་གྱུར་ནས།
[22218] །སྤྱོད་པས་གཞན་ལ་རྔན་སེམས་བྱེད།
[22219] །རང་གི་རྟོག་པ་ཉིད་ཀྱིས་བཅིངས།
[22220] །དེ་ཡང་གྲོལ་བར་ངས་མ་བཤད།
[22221] །རྗེས་སུ་འཇུག་པའི་རྣལ་འབྱོར་ནི།
[22222] །གང་དག་མོས་མའི་ས་ལ་གནས།
[22223] །མོས་པའི་སྤྱོད་པ་བྱེད་པ་སྟེ།

[22226] །ངང་ཉིད་ངེས་པའི་རྣལ་འབྱོར་ནི།
[22227] །རྣམ་པར་མི་རྟོག་ཆིག་ཚུགས་བསྒོམ།
[22228] །མི་རྟོག་པ་ཡི་ངང་ཉིད་ཅན།
[22229] །དེ་ཡང་གྲོལ་བར་ངས་མ་བཤད།
[22230] །སྣང་བ་སེམས་ཀྱི་རྣལ་འབྱོར་ནི།
[22231] །ཐམས་ཅད་ང་ཡི་སེམས་ལས་བྱུང༌།
[22232] །སེམས་ཉིད་རྟོགས་ནས་ཅི་བྱར་ཡོད།
[22233] །དེ་སྐད་ཀུན་ཏུ་རྗོད་བྱེད་པ།
[22234] །ང་ཡིས་གྲོལ་བར་མ་བཤད་དོ།
[22235] །འཇུག་པ་ལས་ཀྱི་རྣལ་འབྱོར་ནི།
[22236] །དོན་ཉིད་བྱ་བས་འགྲུབ་པར་འདོད།
[22237] །བྱ་བ་ཆེན་པོས་བཅིངས་པ་རྣམས།
[22238] །དང་པོ་སེམས་བསྐྱེད་བྱ་བ་བྱེད།
[22239] །དེ་ནས་བཟུང་ནས་བྱ་བ་རྣམས།
[22240] །ཐམས་ཅད་བྱེད་ཅིང་བསྒྲུབ་པའོ།
[22241] །དེ་ཡང་གྲོལ་བར་ངས་མ་བཤད།
[22242] །བྱ་བ་རྒྱུ་ཡི་རྣལ་འབྱོར་ནི།
[22243] །རྒྱུད་ཉིད་ལམ་དུ་བསླང་བ་སྟེ།
[22244] །རྒྱུ་ལ་རེ་ཞིང་རྒྱུ་ལ་ལྟོས།
[22245] །རྒྱུ་ཉིད་མངོན་པར་སྟོན་པ་སྟེ།
[22246] །དེ་ཡང་གྲོལ་བར་ངས་མ་བཤད།
[22247] །བྱེད་པ་རྐྱེན་གྱི་རྣལ་འབྱོར་ནི།
[22248] །སྤྱོད་པས་ཡང་དག་དོན་ཚོལ་བ།
[22249] །རིགས་མི་མཐུན་པའི་ཕྱོགས་འཛིན་པ།
[22250] །དེ་ཡང་གྲོལ་བར་ངས་མ་བཤད།
[22251] །མཐར་ཕྱིན་འབྲས་བུའི་རྣལ་འབྱོར་ནི།
[22252] །རྟོགས་པ་ལམ་དུ་ཁྱེར་བ་སྟེ།
[22253] །ས་དང་ལམ་ལ་མི་གནས་པར།
[22254] །རང་གི་དེ་ཉིད་མཐོང་ནས་ནི།
[22255] །ཡང་དག་གྲོལ་བ་ཆེན་པོར་འདོད།
[22256] །ཡང་དག་དོན་གྱི་རྣལ་འབྱོར་ནི།
[22257] །རང་གི་རྟོགས་པ་རྙེད་ནས་ཀྱང༌།
[22258] །བར་དོའི་སྣང་བ་དག་པ་ལ།
[22259] །རང་གི་ཡང་དག་ཉིད་མཐོང་ནས།
[22260] །དེ་ནས་མཐར་ཕྱིན་གྲོལ་བར་འདོད།
[22261] །མཐར་ཕྱིན་གྲོལ་བའི་རྣལ་འབྱོར་ནི།
[22262] །ངེས་པ་རྙེད་པའི་རྣལ་འབྱོར་ནི།
[22263] །ངེས་པ་རྙེད་པའི་ས་ཉིད་ལ།
[22264] །རང་གི་དེ་ཉིད་རྙེད་མཐོང་ནས།
[22265] །ཡང་དག་ས་ལས་འདའ་བར་འདོད།
[22266] །བྱས་པ་རྫོགས་པའི་རྣལ་འབྱོར་ནི།
[22267] །ཀ་དག་སངས་རྒྱས་ཆེན་པོ་ཡིན།
[22268] །དེ་ལས་གཤེགས་སུ་མེད་པར་གསུངས།
[22269] །རྣལ་འབྱོར་ཆེན་པོ་དེ་བཞི་ལ།
[22270] །ས་ལ་སྦྱང་དུ་མེད་པར་གསུངས།
[22271] །ལམ་ལ་བགྲོད་དུ་མེད་པར་གསུངས།
[22272] །ཞེས་སོ།
[22273] །དེའང་ལམ་བསྒོམ་ཚུལ་ཐ་དད་པས།
[22274] །འདིར་འོད་གསལ་མངོན་སུམ་བྱེད་པས་འོག་མ་ནས་གདོད་ཞུགས་ཏེ་ས་ལམ་སྦྱང་བགྲོད་མི་དགོས་པར་གསུངས་ཀྱང༌།
[22275] རང་ལམ་དུ་སྒོ་གསུམ་གྱི་དྲི་མ་དང་གདོས་བཅས་ཀྱི་ཕུང་པོ་དང༌།
[22276] འགྱུ་བའི་རྟོག་པ་དང༌།

[22278] དེ་དག་དག་དུས་འོག་མའི་ས་ལམ་ཞར་ལ་རྫོགས་པའི་རྟགས་ཤར་ཏེ།
