[22290] །རྣམ་པ་ལྔ་ཡི་ཁོངས་སུ་འདུས།
[22291] །དེ་ཡང་བསྟན་པ་འདི་ལྟ་སྟེ།
[22292] །རྩལ་ཆེན་རྫོགས་དང་ལུང་ཆེན་ཐོབ།
[22293] །དབུགས་ཆེན་འབྱིན་དང་ལུས་ཕྱེད་དང༌།
[22294] །བཀའ་སྡོད་ལས་ཀྱི་རྣལ་འབྱོར་རོ།
[22295] །ཞེས་སོ།
[22296] །རྣལ་འབྱོར་ཕྱི་མ་བཞི་ཡང་ལྔ་པོ་འདིའི་ཁོངས་སུ་འདུ་བར་བཤད་དེ།
[22297] དེ་ཉིད་ལས།
[22298] དེ་ཡང་རྣལ་འབྱོར་ལྔ་ལ་རྫོགས།
[22299] །ཞེས་སོ།
[22300] །རྣལ་འབྱོར་ལྔ་སོ་སོར་བཤད་པ་ནི།
[22301] ཡང་རང་ཤར་ལས།
[22302] ཁམས་གསུམ་སེམས་ཅན་ཐམས་ཅད་ནི།
[22303] བཀའ་སྡོད་ལས་ཀྱི་རྣལ་འབྱོར་ཡིན།
[22304] །བདེན་པ་མཐོང་བའི་གང་ཟག་དེ།
[22305] །འཁྲུལ་པའི་སྣང་བ་འགགས་པ་ནི།
[22306] །ལུས་སེམས་ཕྱེད་པའི་རྣལ་འབྱོར་ཡིན།
[22307] །ལྷུན་གྲུབ་སྣ་ཚོགས་སྣང་བ་ལ།
[22308] །ངེས་པ་རྙེད་པར་འགྱུར་བ་ནི།
[22309] །ལུང་བསྟན་ཐོབ་པའི་རྣལ་འབྱོར་ཡིན།
[22310] །དབང་པོ་ཐ་མའི་ཐ་མ་དེ།
[22311] །ཕྱོགས་བཅུར་སྤྲུལ་པའི་ཞིང་ཁམས་སུ།
[22312] །གནས་ནས་དེ་བཞིན་གཤེགས་པ་དེས།
[22313] །དབུགས་ཆེན་འབྱིན་པའི་རྣལ་འབྱོར་ཡིན།
[22314] །ཀ་དག་འགྱུར་མེད་ཆེན་པོ་ལ།
[22315] །བྱ་རྩོལ་ཐམས་ཅད་བྲལ་བ་དེ།
[22316] །རྩལ་ཆེན་རྫོགས་པའི་རྣལ་འབྱོར་ཡིན།
[22317] །ཅེས་སོ།
[22318] །གཉིས་པ་སྣོད་ལྡན་གྱི་གང་ཟག་བཤད་པ་ལ།
[22319] །སྤྱིར་རྫོགས་པ་ཆེན་པོའི་ཆོས་བསྟན་དུ་རུང་བའི་སྣོད་ནི་སྔར་བསྟན་པ་བཞིན།
[22320] བླ་མ་ལ་གུས་ཤིང་གཏོང་ཕོད་ཆེ་བ།
[22321] དད་པ་དང་ཤེས་རབ་དང་བརྩོན་འགྲུས་ཆེ་བ།
[22322] མི་དགེ་བ་ལ་འཛེམ་ཞིང་འཁོར་བ་ལ་སྐྱོ་བ།
[22323] ཁྱད་པར་མཆོག་གསང་རྫོགས་པ་ཆེན་པོ་ལ་མོས་ཤིང་སྤྱོད་པའི་སྐལ་པ་ཡོད་པ་སྟེ་བརྒྱད་ལྡན་གཅིག་ལ་བསྟན་ནོ།
[22324] །བྱེ་བྲག་ཏུ་ཆོས་དིའི་བསྟན་པ་རྫོགས་པར་སྦྱིན་ཞིང༌།
[22325] གང་གིས་འཛིན་པའི་སྐྱེས་བུ་རྒྱལ་མཚན་རྩེ་མོ་བསྟན་པའི་སྲོག་ཤིང་དུ་བྱོན་པ་ནི།
[22326] ཐལ་འགྱུར་ལས།
[22327] ལྟ་བ་བསྒོམ་པ་འཛིན་པ་ནི།
[22328] །དད་པ་རབ་དང་བརྩོན་འགྲུས་ལྡན།
[22329] །གཏོང་བར་ནུས་ཤིང་བླ་མར་གུས།
[22330] །བརྟུལ་ཞུགས་ཅན་ལ་སྡིག་ལ་འཛེམ།
[22331] །འགྱུར་མེད་རབ་ཏུ་བརྟན་པོས་འཛིན།
[22332] །ཡང་ན་ངང་རིང་ཤུགས་དལ་བ།
[22333] །ལྷུག་པ་གླེན་པའི་ལྷོད་ཡོད་པ།
[22334] །རྟོག་པ་ཆུང་ཞིང་རང་བཞིན་དལ།

[22336] །ཤེས་རབ་ལྡན་པས་འཛིན་པར་འགྱུར།
[22337] །ཅེས་པ་དང༌།

[22339] ཀྱེ་མ་ཉོན་ཅིག་ལྷ་ཡི་རིགས།
[22340] །ངས་བསྟན་ལེགས་པར་ཡིད་ལ་ཟུང༌།
[22341] །སྣང་བའི་དཀྱིལ་འཁོར་ཆེན་པོ་འདི།
[22342] །ཀུན་ལ་བསྟན་པར་མི་བྱ་སྟེ།

[22344] །གསང་སྔགས་གསང་བའི་དམ་པ་འདི།
[22345] །བསྟན་པའི་རིགས་ནི་འདི་ལྟ་སྟེ།
[22346] །གསང་བའི་གཉེར་གཏད་ཐུབ་པར་བྱེད།
[22347] །གསང་བའི་དོན་ལ་གུས་པ་དང༌།
[22348] །ཤེས་རབ་བློ་རྩལ་ཆེ་བ་དང༌།
[22349] །དམ་ཚིག་སྡོམ་པ་ལྡན་ལ་བསྟན།
[22350] །དེ་ནས་ཁྱད་པར་འཁོར་རྣམས་ཀྱིས།
[22351] །རྡོ་རྗེ་འཆང་ལ་ཡང་ཞུས་པ།
[22352] །ཀྱེ་ཀྱེ་བཅོམ་ལྡན་རྡོ་རྗེ་འཆང༌།
[22353] །རྫོགས་ཆེན་མན་ངག་གསང་དོན་འདི།
[22354] །རང་གར་བསྟན་པ་མ་ཡིན་ཏེ།
[22355] །ངེས་པའི་ལུང་རྒྱལ་ཁུ་བྱུག་མཆོག
[22356] །འདི་ཉིད་བསྟན་པའི་གང་ཟག་གི
[22357] །རིགས་དང་སྤྱོད་ལམ་ཆེན་པོ་དང༌།
[22358] །ཁ་དོག་མཚན་ཉིད་གསུམ་དུ་གསོལ།
[22359] །དེ་ནས་བཅོམ་ལྡན་རྡོ་རྗེ་འཆང༌།
[22360] །ཏིང་ངེ་འཛིན་ལས་བཞེངས་ནས་ནི།
[22361] །ཁྱད་པར་འཁོར་ལ་བཀའ་སྩལ་པ།
[22362] །ཀྱེ་མ་ཁྱད་པར་འཁོར་ཚོགས་ཉོན།
[22363] །ང་ཡིས་བསྟན་གྱིས་ལེགས་པར་ཉོན།
[22364] །རིགས་ཀྱི་ཁྱད་པར་འདི་ལྟ་སྟེ།
[22365] །རྒྱལ་རིགས་དང་ནི་བྲམ་ཟེའི་རིགས།
[22366] །དེ་བཞིན་རྗེའུ་རིགས་ཆེན་པོ་དང༌།
[22367] །ཁྱད་པར་ཆེན་པོའི་མི་རྣམས་ལ།
[22368] །གསང་སྔགས་ངེས་དོན་བསྟན་པར་གསུངས།

[22370] །ཕོ་མོ་གཉིས་ནི་གང་ཡང་རུང༌།
[22371] །ཁ་དོག་འདི་དང་ལྡན་པ་ནི།
[22372] །རྫོགས་པ་ཆེན་པོའི་སྣོད་དུ་གསུངས།
[22373] །ཡན་ལག་རགས་ཤིང་ཁ་དོག་ནས།
[22374] །སོ་ཉིད་ཐགས་བཟངས་དཀར་ལ་འདྲིལ།
[22375] །མིག་རྩ་ཡིད་ཙམ་དམར་བ་དང༌།
[22376] །སྐྲ་ཡི་མཚན་ཉིད་ཆེན་པོ་ཡང༌།
[22377] །གཡས་སུ་འཁྱིལ་ཞིང་ཁམ་ནག་མདོག
[22378] །ལུས་ལ་བཀུར་སྟི་ཆུང་བ་དང༌།
[22379] །ཕྱི་ཡི་སྤྱོད་ལམ་ཐ་མལ་མདོག
[22380] །ཚིག་རྣམས་ཤུགས་བྱུང་སྨྲ་བ་དང༌།
[22381] །ཡང་ན་ཚིག་གི་རྣམ་པ་ཀུན།
[22382] །གཅིག་གིས་གང་ལྟར་སྨྲས་པ་ལྟར།
[22383] །སྐྱེས་བུ་དེ་ཡང་དེ་ལྟར་སྨྲ།
[22384] །རྫོགས་ཆེན་གདམས་ངག་བསྟན་པར་གསུངས།
[22385] །དེ་རྣམས་གཅིག་ལ་ཚང་བ་ན།
[22386] །ཤན་པའི་རིགས་སམ་སྨད་འཚོང་རིགས།
[22387] །ཕྱགས་དར་ཁྲོད་པའི་རིགས་ཀྱང་རུང༌།
[22388] །རིགས་ངན་གང་དག་སུ་ཡིན་ཀྱང༌།
[22389] །གསང་བའི་མན་ངག་བཅུད་འདི་གཏད།
[22390] །རྒྱུ་ཆུང་ཡོ་བྱེད་མེད་པ་ན།
[22391] །རང་གི་ཡོ་བྱད་སྦྱར་ནས་ཀྱང༌།
[22392] །ལྷག་མ་མེད་པར་ཡོངས་སུ་སྦྱིན།
[22393] །གསང་བའི་བཅུད་གསུམ་མན་ངག་འདི།
[22394] །རིགས་དང་ཁ་དོག་བརྟག་ཐབས་རྣམས།
[22395] །ང་ནི་མྱ་ངན་འདས་འོག་ཏུ།
[22396] །གསང་བའི་མན་ངག་ཡང་བཅུད་འདི།
[22397] །ངེས་པའི་རྒྱལ་པོ་འདི་ལྟར་བསྟན།
[22398] །དེ་རྣམས་བསྟན་པའི་སྐྱེས་བུ་ལ།
[22399] །གོང་གི་གདམས་ངག་བསྟན་པར་བྱ།
[22400] །ཞེས་གསུངས་སོ།
[22401] །དོན་གཉིས་པ་ཇི་ལྟར་ཉམས་སུ་ལེན་པའི་ཚུལ་ངེས་པ་ལ་བསྒྲུབ་ལུགས་གཉིས་ཏེ།
[22402] སྤྲང་པོའི་ཚུལ་གྱིས་བསྒྲུབ་པ་དང༌།
[22403] རི་དྭགས་ཀྱི་ཚུལ་གྱིས་བསྒྲུབ་པའོ།
[22404] །དང་པོ་ནི།
[22405] རང་གི་ཡུལ་སྤངས་ནས།
[22406] གཞན་གྱི་ཡུལ་ལྗོངས་སུ་སོང་སྟེ།
[22407] སུས་ཀྱང་རང་གི་མིང་དང་བཟང་ངན་དང༌།
[22408] །ཅི་ཤེས་པ་དང་མི་ཤེས་པ་ལ་སོགས་པ་ངོ་མི་ཤེས་པར་བྱས་ཏེ།
[22409] ཐ་མལ་དུ་བརྫུས་ནས།
[22410] ཟས་གོས་སློང་མོས་འཚོ་ཞིང༌།
[22411] ས་ཕུག་དང་གྲོང་ལས་ཅུང་ཟད་ཕྱིན་པའམ།
[22412] དབེན་པའམ།
[22413] འགྲོན་ཁང་ལ་སོགས་པར་མངོན་མཚན་མེད་པས་བསྒྲུབ་སྟེ།
[22414] འདིའི་དུས་སུ་རང་གི་ཡོན་ཏན་མིང་ངོམ་པ་ནི་ཞེ་འདོད་མེད་པའི་རྟགས་ཡིན་ལ།
[22415] གཞན་གྱི་བླ་མ་དང་ཆོས་ལ་སོགས་པ་དྲི་བརྡ་མི་བྱེད་པ་ནི་རང་གི་མན་ངག་ལ་ཡིད་ཆེས་ཤིང་ཐེ་ཚོམ་མེད་པའི་རྟགས་ཡིན་ནོ།
[22416] །དེ་སྐད་དུ་ཡང༌།
[22417] མན་ངག་རིན་པོ་ཆེའི་འཁོར་ལོ་ལས།
[22418] དགྲ་གཉེན་སྤངས་པའི་ཕྱོགས་སུ།
[22419] །སྤྲང་པོ་ལྟ་བུའི་ཚུལ་གྱིས་བསྒྲུབ།
[22420] །ཅེས་སོ།
[22421] །གཉིས་པ་རི་དྭགས་ལྟར་བསྒྲུབ་པ་ནི་རབ་ཏུ་དབེན་བའི་རི་ངོགས་བྲག་གི་བྱ་སྐྱིབས་ཅན་ནམ་ནགས་ཁྲོད་དམ།
[22422] རི་རྩེའམ།
[22423] གངས་འདབས་སམ།
[22424] ལུང་སྟོང་ཆེན་པོར་སོང་ལ་བྱ་བ་གསུམ་སྤངས་ལ་བསྒྲུབ་བོ།
[22425] །དེའང་ལུས་ཀྱི་བྱ་བ་ཚོང་སོ་ནམ་མཁར་ལེན་སྒུལ་སྐྱོད།
[22426] །ཕྱག་བསྐོར་བ་ལ་སོགས་པ་དང༌།
[22427] ངག་གི་བརྗོད་པ་གང་ཅི་ཡང་རུང་བ་ཐམས་ཅད་དང༌།
[22428] སེམས་ཀྱི་འཕྲོ་འདུ་དྲན་བསམ་ཐ་དད་པ་རང་གི་གདམས་ངག་དང་འགལ་བ་རྣམས་ཐམས་ཅད་སྤངས་ལ།
[22429] མན་ངག་དམ་པ་གཅིག་པུ་ལ་རྩེ་གཅིག་ཏུ་མ་ཡེངས་པས་བསྒྲུབ་པར་བྱའོ།
[22430] །དགོས་པ་ནི་ལུས་དལ་བས་རྩ་དམ།
[22431] རྩ་དལ་བས་རླུང་སེམས་རང་བཞིན་གྱིས་དལ།
[22432] རླུང་སེམས་དལ་བས་རྣམ་རྟོག་ཤུགས་ཀྱིས་འགག་ཅིང༌།
[22433] མི་རྟོག་པའི་ཏིང་ངེ་འཛིན་རྒྱུད་ལ་སྐྱེའོ།
[22434] །ངག་དལ་བས་བརྗོད་དུ་མེད་པའི་དོན་རྒྱུད་ལ་སྐྱེ་སྟེ།
[22435] མ་བརྗོད་ན་བརྗོད་དུ་མེད་པའི་རང་བཞིན་ལ་གནས་པའི་ཕྱིར་རོ།
[22436] །སེམས་ཀྱི་དྲན་པས་འཕྲོ་འདུ་སྤངས་པས་ཏིང་ངེ་འཛིན་རང་ལུགས་སུ་སྐྱེ་བ་ཡིན་ནོ།
[22437] །དེ་ཡང་ཐལ་འགྱུར་ལས།
[22438] དད་ལྡན་འཁོར་བའི་ཡིད་བྲལ་བས།
[22439] །བླ་མས་མཆོད་དང་གཏོང་བར་བྱ།
[22440] །འཁོར་བའི་འབྲེལ་པ་ཀུན་སྤངས་ནས།
[22441] །གྲོགས་སྤངས་གཅིག་པུར་གནས་པར་བྱ།
[22442] །ཞེས་སོ།
[22443] །གསུམ་པ་ཉམས་སུ་བླངས་པའི་དོན་དངོས་བསྟན་པ་ལ་བཞི་སྟེ།
[22444] སྤྱོད་པ་ལ་རྒྱུན་དུ་བྱེད་པའི་རྣལ་འབྱོར།
[22445] སྒོམ་པ་ལ་རྒྱུན་དུ་བྱེད་པའི་རྣལ་འབྱོར།
[22446] ལྟ་བ་ལ་རྒྱུན་དུ་བྱེད་པའི་རྣལ་འབྱོར།
[22447] འབྲས་བུ་ལ་རྒྱུན་དུ་བྱེད་པའི་རྣལ་འབྱོར།
[22448] །ནོར་བུ་འཕྲ་བཀོད་ལས།
[22449] སྤྱོད་པ་རྒྱུན་གྱི་རྣལ་འབྱོར་དང༌།
[22450] །སྒོམ་པ་རྒྱུན་གྱི་རྣལ་འབྱོར་དང༌།
[22451] །ལྟ་བ་རྒྱུན་གྱི་རྣལ་འབྱོར་དང༌།
[22452] །འབྲས་བུ་རྒྱུན་གྱི་རྣལ་འབྱོར་རོ།
[22453] །ཞེས་སོ།
[22454] །དེ་ལྟར་བཞི་ལ་དོན་ཉི་ཤུ་ཉམས་སུ་ལེན་པར་བསྟན་པ་ནི།
[22455] རིན་པོ་ཆེ་སྤུངས་པའི་རྒྱུད་ལས།
[22456] དེ་རིམ་བྱེ་བྲག་འདི་ལྟ་བུས།
[22457] །ཁྲིགས་ཆགས་ཉིད་དུ་ཉམས་སུ་བླང༌།
[22458] །བདག་པོ་བཞི་ཡི་ཚུལ་གྱིས་ནི།
[22459] །དོན་གྱི་ངོ་བོ་བཞི་ཡང་གཟུང༌།
[22460] །དེ་ཡི་ཡུལ་རིག་རྣམ་བཞི་ཡིས།
[22461] །འདི་ལྟར་སྣང་བའི་སྐྱེ་རྒྱུན་གཅོད།
[22462] །བཞག་ཐབས་མན་ངག་རྣམ་བཞི་ཡིས།
[22463] །སྣང་སྲིད་ཆོས་རྣམས་རང་ཕྱེན་བྱ།
[22464] །དག་ལུགས་གྲོལ་བའི་གནད་བཞི་ཡིས།
[22465] །ཉོན་མོངས་རྣམས་ལ་རང་དབང་བསྒྱུར།
[22466] །དེ་ཡི་རང་གི་ཚད་འབེབས་པའི།
[22467] །དཔེ་བཞི་ཡིད་ཆེས་རང་ངོ་བསྟན།
[22468] །མི་འགྱུར་ལྟ་བའི་གདེང་ཆེན་བཞིས།
[22469] །མི་བཟློག་ཡེ་ཤེས་ཚད་ཀྱང་གཟུང༌།
[22470] །འདི་ལ་གོམས་པར་སུས་བྱས་པ།
[22471] །རང་བཞིན་རྫོགས་པ་ཆེན་པོ་ལ།
[22472] །འདི་ཡིས་རེག་གོས་ས་མ་ཡ།
[22473] །འདི་ལྟར་གང་ལ་སྣང་གྱུར་པ།
[22474] །ཡེ་ཤེས་ཀུན་ཀྱང་ཕུགས་སུ་ཟིན།
[22475] །གསང་ཆེན་ཕྱག་རྒྱ་ཆེན་པོ་འཐོབ།
[22476] །ཅེས་གསུངས་པའི་དོན་ཆ་ཤས་ཀྱི་དབྱེ་བས་ཉམས་སུ་བླང་བའི་རིམ་པ་ལས།
