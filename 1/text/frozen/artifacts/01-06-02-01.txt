[7215] །གལ་ཏེ་ཐ་མ་དག་བྱུང་ན།
[7216] །ལུས་ཀྱི་ཡོངས་སུ་ཚར་མ་ཆོད།
[7217] །ཕྱི་ནས་བསྐྱར་ལ་སྤྱོད་པ་བརྩམས།
[7218] །བསྐྱར་ཀྱང་སྡིག་པ་མ་དག་ན།
[7219] །ལྷ་ཡི་སྐུ་ནི་སྟོང་ཕྲག་བསྒོམ།
[7220] །དེ་ནས་བླ་མའི་མན་ངག་ནི།
[7221] །རིམ་པ་དག་ལས་བསླབ་པར་བྱའོ།
[7222] །ངག་གི་དུས་ནའང་དེ་བཞིན་དུ།
[7223] །རྣལ་འབྱོར་ལྡན་པས་ཤེས་པར་བྱ།
[7224] །ཐོག་མཐའ་མེད་ལ་བར་བྱུང་ན།
[7225] །ཡི་གེ་བརྒྱ་པ་འབུམ་ཕྲག་གཅིག
[7226] །འབད་དང་བཅས་ཏེ་བཟླས་པར་བྱའོ།
[7227] །གལ་ཏེ་ཐ་མར་བྱུང་བ་ན།
[7228] གསང་སྔགས་རྒྱུད་ནི་རང་ལོའི་གྲངས།
[7229] །བརྒྱ་དང་རྩ་བརྒྱད་བརྗོད་པར་བྱའོ།
[7230] །ཐོག་མཐའ་བར་མེད་དུས་རྣམས་སུ།
[7231] །རྒྱུད་ཀློག་པ་དང་སྔགས་བཟླས་དང༌།
[7232] །བླ་མ་ཡི་དམ་ལྷ་དག་གི

[7234] །ཨཱ་ལི་ཀཱ་ལི་ཡི་གེའི་སྒྲ།
[7235] །སོ་སོ་རྐྱང་པར་བརྗོད་པ་དང༌།
[7236] །ཧཱུྃ་དང་ཕཊ་དང་ཨོཾ་དང་ནི།
[7237] །སྦྲེལ་མ་སྣ་ཚོགས་བརྗོད་པ་ལ།
[7238] །ཟླ་བ་བདུན་དུ་འབད་པར་བྱའོ།
[7239] །དེ་ནས་གདམས་པའི་གནས་བྱའོ།
[7240] །ཡི་དམ་ལྷ་ཡི་ཏིང་ངེ་འཛིན།
[7241] །ཉི་མ་གསུམ་དུ་བསྒོམ་པར་བྱའོ།
[7242] །གལ་ཏེ་བར་དུ་བྱུང་བ་ན།
[7243] །སྔགས་ཀྱི་འོད་ཟེར་འཕྲོ་འདུ་ལས།
[7244] །ཉི་མ་བདུན་དུ་འབད་པར་བྱའོ།
[7245] །ཐ་མར་བྱུང་ན་ཡི་གེ་ཡི།
[7246] །འཕྲོ་འདུ་དག་ནི་བསྒོམ་པ་ལ།
[7247] །ཉི་མ་བཅུ་ཡི་བར་དུ་འབད།
[7248] །ཐོག་མ་བར་མཐའ་ཀུན་བྱུང་ན།
[7249] །རྫོགས་པའི་རིམ་པ་བསྒོམ་པའམ།
[7250] །ངག་ལས་བྱུང་བའི་བསམ་གཏན་ལ།
[7251] །ཟླ་བ་དྲུག་ཏུ་འབད་པར་བྱའོ།
[7252] །གལ་ཏེ་ལུས་ངག་སེམས་ཀྱི་ཡང༌།
[7253] །སྤྱོད་པ་སོ་སོའི་དུས་རྣམས་སུ།
[7254] །ནམ་མཁའ་སྔོ་བསངས་སྤྲིན་དང་བྲལ།
[7255] །རླུང་དང་སེར་བུའི་སྒྲ་མེད་པ།
[7256] །འཇམ་ཞིང་འཇམ་པ་དག་བྱུང་ན།
[7257] །སྡིག་པ་དག་ནི་སྦྱང་མི་དགོས།
[7258] །གལ་ཏེ་ཐོག་མ་ཡོད་པ་ལ།
[7259] །སོ་སོའི་ཐ་མར་དག་ན་ཡང༌།
[7260] །འདིས་ནི་སྡིག་པ་འབྱོངས་པས་ན།
[7261] །བཤགས་པའི་ཆོ་ག་བྱ་མི་དགོས།
[7262] །དེ་བཞིན་ལུས་ངག་སེམས་དག་གི

[7263] །སོ་སོའི་ངོས་གཟུང་མཁས་པས་བརྟག
[7264] །དེས་ནི་ཤིན་ཏུ་སྤྲོས་པ་མེད།

[7273] །ངོ་མཚར་ཚིག་རྣམས་བརྗོད་པ་ནི།
[7274] །དེ་བཞིན་དུ་ཡང་ངག་གིས་ཐོབ།
[7275] །སེམས་ལ་ཏིང་འཛིན་འབབ་པ་དང༌།

[7277] །སེམས་ཀྱིས་དབང་བསྐུར་ཐོབ་པའོ།
[7278] །གལ་ཏེ་ལུས་ཀྱིས་ཐོབ་ན་ནི།
[7279] །སྤྲུལ་པའི་སྐུ་ཉིད་འདི་ཡིས་ཐོབ།
[7280] །དེ་བཞིན་ངག་ལ་ལོངས་་སྐུའོ།
[7281] །སེམས་ནི་ཆོས་སྐུའི་ཞིང་དུ་འགྲོ།
[7282] །ཤིན་ཏུ་སྤྲོས་མེད་དུས་མཚན་བཤད།

[7287] །དུས་དང་མཚན་པ་ངེས་བསྟན་བྱ།
[7288] །ཕྱི་དུས་འབྲས་བུ་སྨིན་པའམ།
[7289] །འབྲུ་རྣམས་བསྡུ་བའི་དུས་ལ་གཏད།
[7290] །རང་སྣང་འབྲས་་བུར་གྲོལ་ཕྱིར་རོ།
[7291] །ནང་དུས་ཡར་ངོའི་ཚེས་གཅིག་གམ།
[7292] །ཡང་ན་ཟད་པའི་མཚམས་ལ་གཏད།
[7293] །ཐིག་ལེ་ཉག་གཅིག་རྟོགས་ནས་ནི།
[7294] །ལས་དང་བག་ཆགས་ཟད་ཕྱིར་རོ།
[7295] །སྐལ་ལྡན་གཅིག་ལས་མང་ན་ཡང༌།
[7296] །རྣལ་འབྱོར་པ་ཡིས་དོན་མི་འགྲུབ།
[7297] །དད་ལྡན་ཡིད་དུ་འོང་བའི་བུ།
[7298] །སྐལ་བ་ཡོད་པ་གཅིག་ལ་བསྐུར།

[7302] །སོ་སོའི་ཚིག་གིས་ངེས་གསལ་བས།
[7303] །རང་བཞིན་དག་པའི་གཞི་མཐོང་བའོ།
[7304] །འདི་ལ་མཚན་མ་ཅི་བྱུང་ཡང༌།
[7305] །སྡིག་པ་སྦྱང་བའི་ཆོ་ག་མེད།
[7306] །གོང་དུ་ཐད་ཀར་སྦྱང་བའི་ཕྱིར།
[7307] །གལ་ཏེ་སྤྲིན་མེད་གནམ་གདངས་ན།
[7308] །ཆོས་ཀྱི་སྐུ་ཡི་ཞིང་ལ་སྤྱོད།
[7309] །སྣ་ཚོགས་འོད་ཀྱི་གདངས་ཤར་ན།
[7310] །འདི་ནི་ལོངས་སྐུའི་ཞིང་དུའོ།
[7311] །སྣ་ཚོགས་སྒྲ་རྣམས་གྲག་པ་ན།
[7312] །སྤྲུལ་སྐུའི་ཞིང་དུ་དབུགས་འབྱིན་ནོ།

[7314] །འདི་དུས་ལུས་གནད་མཁས་པ་དང༌།

[7316] །ལོ་བདུན་གྱིས་ནི་ཟད་སར་ཕྱིན།

[7319] །ཆོས་ཉིད་ཟད་པའི་ས་ལ་ཕྱིན།

[7325] །བར་མ་ཆད་པར་རིམ་གྱིས་ཀྱང༌།
[7326] །དབང་དག་རྫོགས་པར་བྱེད་པའི་ལུགས།

[7682] །འགྲོ་མེད་འོང་མེད་འགྱུར་བ་མེད།

[8031] །ཡུམ་ནི་མཁའ་གསང་བདེ་བ་ཆེན་པོའི་དབང་དང༌།
[8032] གསང་བ་དངས་མ་རྣམས་གཉིས་ཀྱི་དབང་ངོ༌།
[8033] །དེ་དག་ཀྱང༌།
[8034] །རང་ཤར་ལས།
[8035] མ་རྟོགས་རྒྱུ་མཚོན་དབང་བསྐུར་ནི།
[8036] །མཚན་མའི་རྫས་ལ་བརྟེན་པ་དང༌།
[8037] །རྡུལ་ཚོན་དཀྱིལ་འཁོར་བརྟེན་པ་དང༌།
[8038] །ཡུམ་གྱི་བྷ་གས་དབང་བསྐུར་རོ།
[8039] །མཚན་མ་རྫས་ལ་རྣམ་གསུམ་སྟེ།
[8040] །བཀྲ་ཤིས་རྫས་བརྒྱད་བསྐུར་ཐབས་དང༌།
[8041] །རིན་ཆེན་བུམ་པའི་དབང་བསྐུར་དང༌།
[8042] །ཤ་ཁྲག་གཏོར་མའི་བསྐུར་ཐབས་སོ།
[8043] །དཀྱིལ་འཁོར་བསྐུར་ཐབས་རྣམ་གསུམ་སྟེ།
[8044] །འཁོར་གྱི་དཀྱིལ་འཁོར་བསྐུར་ཐབས་དང༌།
[8045] །ཙ་ཀ་ལི་ཡི་བསྐུར་ཐབས་དང༌།
[8046] །དབུས་ཀྱི་ཕོ་བྲང་བསྐུར་ཐབས་སོ།
[8047] །ཡུམ་ལ་རྣམ་པ་གསུམ་དུ་གསུངས།
[8048] །ཕྱག་རྒྱའི་སྒོ་ནས་དབང་བསྐུར་དང༌།
[8049] །བྷ་ག་དང་ནི་རྡོ་རྗེ་ཡི།
[8050] །དབང་བསྐུར་དམ་པ་མཆོག་ཡིན་ནོ།
[8051] །བྱང་སེམས་དཀར་དམར་བསྐུར་ཐབས་སོ།
[8052] །མ་རྟོགས་རྒྱུ་མཚོན་དབང་བསྐུར་རོ།
[8053] །ཞེས་པ་དང༌།
[8054] ཡང༌།
[8055] མ་རྟོགས་པ་ཡི་ཚིག་སྒོ་ནས།
[8056] །རྫས་རེས་དོན་རེ་བསྟན་པར་བྱ།
[8057] །བསྐྱེད་པ་ཞི་བའི་དབང་དང་ནི།
[8058] །རྫོགས་པ་ཁྲོ་བོའི་དབང་བསྐུར་རོ།
[8059] །གསང་མཚན་རྡོ་རྗེར་དབང་ཡང་བསྐུར།
[8060] །བཀའ་རྟགས་པོ་ཏིའི་དབང་བསྐུར་ཞིང༌།
[8061] །དེ་ལ་སོགས་པའི་རྫས་རྣམས་ཀུན།
[8062] །རིམ་པ་བཞིན་དུ་བསྐུར་བར་བྱ།
[8063] །ཞེས་སོ།
[8064] །གཉིས་པ་མཚན་མེད་ཡེ་ཤེས་ཀྱི་དབང་ནི།

[8065] ཕྱི་སྣང་བ་སེམས་ཀྱི་རོལ་པར་བསྐུར་བ།
[8066] ནང་སྒོམ་པ་སེམས་ཀྱི་བཞག་ཐབས་སུ་བསྐུར་བ།
[8067] གསང་བ་རིག་པ་ཡེ་ཤེས་ཀྱི་རང་ངོར་བསྐུར་བའོ།
[8068] །རིག་པ་ལའང་ངོ་སྤྲོད་ཉི་ཤུ་་རྩ་གཅིག་གིས་དབང་པོ་དང་སྦྱར་ཏེ་དཔེ་དོན་རྟགས་སུ་གསུམ་དུ་བསྐུར་བ་དང༌།
[8069] རིག་པ་རྐྱང་པ་འོད་ལྔ་རང་གསལ་དུ་བསྐུར་བ་དང༌།
[8070] ཤར་གྲོལ་རིས་མེད་རྫོགས་པ་ཆེན་པོ་སྤང་བླང་བྱ་རྩོལ་ལས་འདས་པར་བསྐུར་བ་སྟེ་གསུམ་དུ་གསུངས་པ་ཡང༌།
[8071] །དེ་ཉིད་ལས།
[8072] རྟོགས་པའི་ཡེ་ཤེས་དབང་བསྐུར་ནི།
[8073] །རྣམ་པ་གསུམ་དུ་ཤེས་པར་བྱ།
[8074] །སེམས་ཅན་སྣང་བ་ཕྱི་ལྟར་བསྐུར།
[8075] །སྒོམ་པ་སེམས་འཛིན་ནང་ལྟར་བསྐུར།
[8076] །ངོ་སྤྲོད་ཉི་ཤུ་རྩ་གཅིག་གིས།
[8077] །གསང་བའི་དབང་དུ་བསྐུར་བར་བྱ།
[8078] །ངོ་སྤྲོད་ལ་ཡང་རྣམ་གསུམ་སྟེ།
[8079] །དཔེ་དོན་རྟགས་གསུམ་ཚོགས་པ་ཡིས།
[8080] །མཚོན་པའི་དབང་ཆེན་བསྐུར་བར་བྱ།
[8081] །རིག་པའི་ཡེ་ཤེས་ངོ་སྤྲོད་བྱ།
[8082] །གསང་བའི་ཡེ་ཤེས་དབང་བསྐུར་རོ།
[8083] །རིག་པ་རྐྱང་པའི་ངོ་སྤྲོད་ཀྱིས།
[8084] །གཟུགས་སྐུའི་ཞིང་ཁམས་དབང་བསྐུར་བྱ།
[8085] །རྟོགས་པའི་ཡེ་ཤེས་དབང་མཆོག་ནི།
[8086] །ཅིར་སྣང་རིག་པར་ཤེས་པའོ།
[8087] །རིག་པ་དགག་སྒྲུབ་མེད་པ་ཡི།
[8088] །དབང་རྣམས་དུས་གཅིག་རྫོགས་པའོ།
[8089] །དབང་རྣམས་གྲོལ་བྱེད་གདམས་ངག་གོ

[8091] །དབང་ནི་རང་ལ་ཡོད་པའོ།
[8092] །དེ་ཡང་དོན་ལ་བཀྲོལ་ཏེ་གསུངས།
[8093] །གཅིག་བཀྲོལ་གཉིས་བཀྲོལ་ཐམས་ཅད་བཀྲོལ།
[8094] །གྲོལ་བ་རང་བྱུང་གཞན་ལས་མེད།
[8095] །རང་བྱུང་དབང་ཆེན་དེ་བཞིན་ནོ།
[8096] །དོན་གྱི་དབང་མཆོག་དཀྱིལ་འཁོར་ལ།
[8097] །གཅིག་རྫོགས་གཉིས་རྫོགས་ཐམས་ཅད་རྫོགས།
[8098] །དོན་དབང་ཡེ་ཤེས་མཚོན་པ་ལ།
[8099] །གཅིག་མེད་གཉིས་མེད་ཐམས་ཅད་མེད།
[8100] །རྡོ་རྗེའི་དོན་རྟོགས་དབང་ཆེན་ལ།
[8101] །གཅིག་བྱུང་གཉིས་བྱུང་ཐམས་ཅད་བྱུང༌།
[8102] །སྟོང་པ་ཡེ་ཤེས་དབང་ཆེན་ལ།
[8103] །གཅིག་སྐྱེ་གཉིས་སྐྱེ་ཐམས་ཅད་སྐྱེ།
[8104] །རྟོགས་ན་དབང་མཆོག་དེ་ལྟར་བསྐུར།

[8106] །དེ་ཡང་རང་ལ་ཡེ་ནས་ཡོད་པའི་དབང་བླ་མས་ངོ་སྤྲོད་པ་ལ་བསྐུར་བའི་ཐ་སྙད་དུ་བརྗོད་པ་ཡིན་ནོ།

[8108] བླ་མ་གྲུབ་པ་ཐོབ་པ་རྣམས་ཀྱིས་སིལ་བུ་དུམ་བུ་གང་རུང་རེ་རེས་ཀྱང་སློབ་མ་ལ་དབང་རྫོགས་པ་འཐོབ་པས།
