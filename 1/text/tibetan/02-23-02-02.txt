[12452] ཁམས་གསུམ་འཁོར་བའི་ཁང་པ་ན་དུག་ལྔ་རང་ཚན་ལ་སྤྱོད་པའི་འགྲོ་བ་ཀུན་ཡང་ཉལ་ཡང་ལངས་པའི་གཉིད་དང་རྨི་ལམ་ལ་ཐོག་མཐའ་དང་སངས་པའི་དུས་མེད་པ་འདི་བསམས་ལ།
[12453] ད་རེས་འཁོར་བའི་མུན་ཁང་ནས་ཐར་པའི་ཐབས་ལ་འབད་པ་འདོན་དགོས་སོ།
[12454] །ལས་དང་ཉོན་མོངས་དབང་གྱུར་འཁོར་བ་པ།
[12455] །མི་རྟག་མངོན་དུ་གྱུར་ཀྱང་བདེན་པར་འཛིན།
[12456] །སྐྱེ་ཤིའི་ཆུ་བོར་ལྷུང་ཡང་ཐར་མི་ཤེས།
[12457] །སྣ་ཚོགས་སྡུག་བསྔལ་མྱོང་ཡང་བྲེད་དུས་མེད།
[12458] །ཡང་དག་བཤེས་བྲལ་ལམ་སྟོན་མགོན་མེད་ཅིང༌།
[12459] །ཆོས་ཀྱི་མིག་ཟུམ་མ་རིག་མུན་པར་འཁྱམས།
[12460] །འདི་དག་བསམས་ནས་རྙེད་དཀའི་དལ་བ་ལ།
[12461] །སྤྲོ་དགའི་སེམས་བསྐྱེད་དམ་ཆོས་བསྒྲུབ་པར་འོས།
[12462] །ཡང་དག་ལམ་སྟོན་དེད་དཔོན་མཆོག་གྱུར་པ།
[12463] །མཚན་ལྡན་བླ་མ་རྣམས་དང་མཇལ་གྱུར་ཅིང༌།
[12464] །ཟབ་མོའི་ཆོས་མཆོག་བདུད་རྩི་ཐོབ་དུས་འདིར།
[12465] །བརྩོན་པས་བསྒྲུབས་ཏེ་སྲིད་གསུམ་རྒྱ་མཚོ་ཡི།
[12466] །ཕ་རོལ་མཆོག་ཏུ་རྒལ་ལ་བརྩོན་འཚལ་ལོ།
[12467] །ཚེ་ནི་མི་རྟག་ནུབ་རིའི་གྲིབ་མ་བཞིན།
[12468] །གཡང་ཁའི་ལྗོན་ཤིང་རླུང་གིས་བསྐྱོད་འདྲ་ཞིང༌།
[12469] །སྤྲིན་དང་གློག་བཞི་རྩྭ་རྩེའི་ཟིལ་པ་བཞིན།
[12470] །མྱུར་བ་ཉིད་དུ་འཆི་བར་བསམ་འཚལ་ལོ།
[12471] །ད་ལྟ་འཆི་ཞེས་སྙིང་ནས་མ་དྲན་ན།
[12472] །ལེ་ལོས་གཡེངས་པས་འཆི་དུས་ཕན་མི་ཐོགས།
[12473] །དེ་བས་ད་ལྟ་ཐོག་ཏུ་བབ་པ་བཞིན།
[12474] །འཇིགས་སྔངས་སྐྲག་པས་སྙིང་པོའི་དོན་ལ་སྦྱོངས།
[12475] །དེང་ནས་བདག་ཀྱང་ནགས་ཚལ་མེ་ཏོག་ཅན།
[12476] །སྐྱེ་བོས་སྟོང་ཞིང་འབབ་ཆུའི་སྒྲ་གྲག་ལ།
[12477] །དབེན་ཞིང་ཡིད་འོང་བྲག་རིའི་དྲུང་འདུག་སྟེ།
[12478] །དུས་བཞིར་སྙིང་པོ་མེད་ཅེས་སེམས་པ་ཡི།
[12479] །དགེ་བ་རྒྱན་རྒྱུར་འཇིག་རྟེན་མིས་བསྐྲུན་པ།
[12480] །ཀུན་ལས་ཆེས་མཆོག་བདུད་རྩི་ཐོབ་གྱུར་ཅིང༌།
[12481] །མི་རྟག་སྟོང་པའི་སྒྲ་ཆེན་སྒྲོགས་པར་ཤོག

[12482] །སྨན་ཤིང་མེ་ཏོག་ཤང་ཤོང་མཛེས་པ་དེར།
[12483] །ཆོས་ཚུལ་ཞི་བ་སྤྲིན་གྱི་བླ་རེ་ཅན།
[12484] །རིག་འཛིན་རྒྱལ་བའི་གསེར་གྱི་གློག་འགྱུ་བས།
[12485] །ཚེ་གཅིག་འོད་གསལ་སྙིང་པོ་འགྲུབ་པར་ཤོག
[12486] །ཐར་པའི་ཁང་བཟངས་རིན་ཆེན་ཕོ་བྲང་ནས།
[12487] །ཆོས་ཚུལ་ཞི་བ་རྔ་སྒྲའི་དབྱངས་གྲག་པས།
[12488] །མ་ལུས་འགྲོ་ཀུན་མ་རིག་གཉིད་སངས་ཏེ།
[12489] །གདོད་མའི་ས་ལ་མངོན་པར་བྱང་ཆུབ་ཤོག
[12490] །སྐྱོ་བས་སེམས་པའི་རང་བཞིན་གསལ་བ་ཅན།
[12491] །ཞི་བའི་ནགས་སུ་གཅིག་པུ་འགྲོ་འདོད་ནས།
[12492] །སྙིང་ནས་སྨོན་པའི་ཚིག་གི་དབྱངས་སྨྲས་པ།
[12493] །མོས་དགེ་དག་པས་མྱུར་དུ་འགྲུབ་པར་ཤོག
[12494] །ཅེས་བྱ་བ་ནི་བར་སྐབས་ཀྱི་ཚིགས་སུ་བཅད་པ་དག་གོ
[12495] །གཉིས་པ་འཆི་བའི་རྟགས་བརྟག་པ་ནི།

[12498] །དུར་ཁྲོད་དབེན་པའི་ཕྱོགས་སོང་ལ།
[12499] །བླ་མ་མཆོད་ཅིང་གཏང་བར་བྱ།
[12500] །གཞན་ཡང་བཟའ་དང་བཏུང་བ་ཡིས།
[12501] །འགྲོ་བ་གཞན་ཡང་མགུ་བྱས་ཏེ།
[12502] །དེ་ནས་ཕྱི་ནང་གསང་བ་ཡི།
[12503] །འཆི་བའི་རྟགས་ཀྱང་བརྟག་པར་བྱ།

[12505] ཐ་མལ་པ་ནམ་འཆི་མི་ཤེས་པ་ཀུན་ཀྱང༌།
[12506] མོ་དང་པྲ་ལ་སོགས་པས་རྟོག་པ་ཕྱི་རོལ་པ་ལ་སོགས་པ་འཇིག་རྟེན་པ་ལས་ཁྱད་པར་དུ་འཕགས་པ།
[12507] རྣལ་འབྱོར་པས་དུས་ངེས་པར་གཟུང་བ་ནི།
[12508] མི་མེད་པའི་ལུང་སྟོང་ངམ།
[12509] དགོན་པའམ།
[12510] དུར་ཁྲོད་དམ།
[12511] ནགས་ཁྲོད་དམ།
[12512] རི་བོའི་ལྗོངས་ལ་སོགས་པར་སོང་ལ།
[12513] ཚེ་གཟུགས་བརྟག་པ་ལ་སོགས་པ་རྣམས་བྱ་སྟེ།

[12518] །འཆི་བའི་དུས་ཚོད་གཟུང་བ་གཅེས།
