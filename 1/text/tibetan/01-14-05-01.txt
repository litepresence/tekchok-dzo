[18643] སྣང་བ་བཞིའི་ཆོས་ཉིད་གོང་འཕེལ་དུ་ཐོབ་པ་སྟེ།

[18681] །ཞེས་སོ།

[18848] །ཐིག་ལེ་སྟོང་པའི་སྒྲོན་མ་ནི་དཔེར་ན་རྨ་བྱའི་མདོངས་ལྟ་བུའོ།
[18849] །དབྱིངས་རིག་པའི་སྒྲོན་མ་ནི་དཔེར་ན་ཡི་གེ་ན་རོ་ལྟ་བུའོ།

[18852] །འགྲོ་ས་ནི་རིན་པོ་ཆེའི་ལུས་ཏེ་དཔེར་ན་ཁ་བྲག་གི་ལམ་ལྟ་བུའོ།

[18874] །གཉིས་པ་ཡེ་ཤེས་ཀྱི་བླང་བྱའི་གནས་བྱེ་བྲག་ཏུ་བཤད་པ་ལ་གསུམ་ཏེ་གནས་པའི་རྟེན་རྩ་སྦྱོང་བ་འཆར་བའི་སྒོ་སྒྲོན་མ་གཟུང་བ།

[18875] སྣང་བའི་གདངས་འོད་གསལ་བསྟན་པའོ།
[18876] །དང་པོ་ནི་ལུས་ལ་ཡེ་ཤེས་གནས་པས་གནད་གཉིས་ཀྱིས་ལམ་དུ་གཞུག་པ་སྟེ།
[18877] དེའང་སྤྱིའི་གནད་རླུང་སེམས་གཅུད་ལ་བོར་ཞིང་སྒྲོན་མ་ཨར་ལ་གཏད་པས་དབྱིངས་རིག་གི་འཕེལ་ཟད་པར་བྱ་བ་ནི་ཐལ་འགྱུར་ལས།

[18882] །ལུས་དང་བསྟུན་པས་སྐུ་རྣམས་དང༌།
[18883] །ཡེ་ཤེས་འོད་ངད་ཁ་དོག་རྟགས།

[18885] །གདངས་དང་བཅས་ཏེ་ཁྱབ་པའོ།
[18886] །ཤེས་རབ་ཉིད་དང་སྒྲོན་མའི་ལས།
[18887] །དོན་དམ་ཀུན་རྫོབ་རྟགས་ཕྱེ་བའོ།
[18888] །སྨིན་བྱེད་ཐིག་ལེ་སོ་སོའི་གནད།
[18889] །འགྱུར་བྱེད་རྟོག་པ་རགས་འཛིན་པ།
[18890] །ཡེ་ཤེས་སྣང་བ་བཞི་གཉིས་ཆ།
[18891] །ལས་འབྱུང་འཇུག་པའི་དབྱེ་བ་ལས།
[18892] །ཙིཏྟ་རིན་ཆེན་གཞལ་ཡས་སུ།
[18893] །རིན་ཆེན་འདུས་པ་ཟུར་བརྒྱད་སྒོ།
[18894] །ཡེ་ཤེས་ལྔ་དང་སྐུ་ལྔ་སྟེ།
[18895] །རླུང་དང་ཤེས་པའི་རྩལ་ཡང་ལྔ༌།
[18896] །ཐམས་ཅད་ལྔ་ལྔའི་རང་བཞིན་ནོ།
[18897] །སྐུ་ཡི་གནད་ནི་གདངས་ཀྱིས་སོ།
[18898] །ཡེ་ཤེས་འདུག་སྟངས་དག་གིས་སོ།
[18899] །ཐིག་ལེ་བསྐྱིལ་ཞིང་དཀྲུགས་པས་འགྲུབ།
[18900] །འོད་ནི་ཡས་ཤིང་སྡུད་པའོ།
[18901] །འབྲེལ་ལྡན་བསྡོམས་པས་རླུང་གི་གནད།
[18902] །འཇུག་པའི་ལས་ནི་མཐའ་ཡིས་དབྱེ།
[18903] །དེ་ཡི་འགྱུ་བྱེད་ལས་རླུང་ཟད།
[18904] །འཁྲུལ་པའི་སྡེབས་དང་སྦྱར་བའོ།
[18905] །རིག་པའི་གནད་ནི་དབྱུང་བ་དང༌།
[18906] །བསྐྱིལ་ཞིང་མཁའ་ལ་གཏད་པའོ།
[18907] །དབྱིངས་ཀྱི་གནད་ནི་བསྡུ་བ་དང༌།
[18908] །དགུག་དང་ཁམས་ཀྱི་བྱེར་ཡས་སྦྱར།
[18909] །སྣང་བའི་གནད་ནི་འཕེལ་དང་ཟད།
[18910] །སྣ་ཚོགས་རང་སར་གྲོལ་བའོ།
[18911] །ཞེས་སོ།
[18912] །བྱེ་བྲག་གི་གནད་འཁོར་ལོ་བཞི་དང་གཙོ་མོ་གསུམ་གྱི་གནད་གཅུན་པས་རླུང་སེམས་ཡེ་ཤེས་ཀྱི་དྲ་བར་གཞུག་པ་ནི་དེ་ཉིད་ལས།
[18913] རིག་པ་ཡེ་ཤེས་ལམ་ཉིད་ནི།
[18914] །རང་གི་རྩ་ཡི་འཁོར་ལོ་ལ།
[18915] །སྐུ་གསུང་རང་རྫོགས་གནས་པ་ལས།
[18916] །རོ་མ་རྐྱང་མ་ཀུན་འདར་མ།
[18917] །མ་གསུམ་དག་གིས་ལུས་བཟུང་སྟེ།
[18918] །དབུས་མས་འཁོར་ལོ་བཞི་རྟེན་བྱས།
[18919] །འདི་དག་གཡས་གཡོན་དབུས་གནས་པས།
[18920] །ཀུན་རྫོབ་དོན་དམ་དབྱེ་བ་ཡིས།
[18921] །ལུས་དང་ཡན་ལག་སྐྱེད་པ་དང༌།
[18922] །སྒྲོན་མ་ཉིད་ཀྱང་སྨིན་པར་བྱེད།
[18923] །རོ་མས་རོ་རྣམས་འཁོར་ལོ་ལ།
[18924] །རླུང་གིས་བསྐུལ་ཞིང་ཐིག་ལེ་འཕེན།
[18925] །རོ་ནི་རྩ་ལ་བརྟེན་པ་ཡིས།
[18926] །བདེ་བས་ལུས་སེམས་རྒྱས་པར་བྱེད།
[18927] །མས་ནི་དངས་མ་སྡུད་པ་དང༌།
[18928] །གོང་འོག་གནས་གྱི་གཞི་མ་བྱེད།
[18929] །འདི་ལ་བསྐྱིལ་ཞིང་གཏེམས་པ་གནད།
[18930] །འདི་ཡི་ཡན་ལག་དྲུག་པོ་ལ།
[18931] །གཏར་ཞིང་བསྲེགས་ན་ཡེ་ཤེས་འགག

[18932] །དྲོད་ཐོབ་འདོད་ན་མཉེ་བ་གནད།
[18933] །རླུང་དང་ཐིག་ལེ་དགུག་པར་བྱའོ།
[18934] །རྐྱང་མས་སྐྱེད་བྱེད་འཁོར་ལོ་ལ།
[18935] །རླུང་གིས་འཕེན་སྡུད་ཐིག་ལེ་འཇུག
[18936] །རྐྱང་ཞེས་བྱ་བ་ཟླ་མེད་པས།
[18937] །ཉག་གཅིག་དགོངས་པ་སྟོན་པར་བྱེད།
[18938] །མས་ནི་བརྙིངས་མ་སེལ་བ་དང༌།
[18939] །ངོ་བོའི་གཟི་མདངས་བསྐྱེད་པའོ།
[18940] །འདི་ལ་གསེང་ཞིང་མཉེ་བ་གནད།
[18941] །འདི་ལ་ཡལ་ག་མི་སྣང་བས།
[18942] །སྨིན་སོ་བཞི་ལ་ཐིག་ལེ་གདབ།
[18943] །ཟས་ལ་དབང་ཐོབ་འདོད་པས་ནི།
[18944] །རླུང་ནི་བསྟིམ་ཞིང་ཆུ་མི་འདོར།
[18945] །དམ་ཚིག་གིས་ནི་རྩ་བརྟག་གོ
[18946] །ཀུན་འདར་མས་ནི་དྲན་པ་དང༌།
[18947] །རྩེ་མོའི་འཁོར་ལོ་བསྡུས་པས་སྨིན།
[18948] །ལས་རླུང་བཟློག་ཅིང་ཡེ་ཤེས་འཛིན།
[18949] །ཀུན་ནི་སྤྲུལ་སྐུའི་ཡོན་ཏན་སྟེར།
[18950] །འདར་བས་ལོངས་སྐུའི་ཡོན་ཏན་ནོ།
[18951] །མས་ནི་ཆོས་སྐུ་དངོས་སུ་སྟོན།
[18952] །འདི་ཡི་ཡན་ལག་བཅུ་གཉིས་ལ།
[18953] །གསེང་ཞིང་འདྲིལ་བ་གནད་ཡིན་ནོ།
[18954] །ཚེ་ཉིད་སྤེལ་ན་བྱུག་པ་བསྟེན།
[18955] །ཐིག་ལེ་ལུས་དང་སེམས་ཀྱིས་བྱ།
[18956] །ཡེ་ཤེས་ལམ་ནི་འདི་དག་ལས།
[18957] །ཐིག་ལེ་སྐུ་དང་བཅས་པར་འགྱུར།
[18958] །རླུང་ནི་འགྲོ་འོང་ལས་ཀུན་བྱེད།
[18959] །སེམས་ཀྱི་བདག་པོ་ཡིད་ཀྱིས་བཟུང༌།
[18960] །ལམ་གྱིས་བཟུང་དང་དགུག་པའོ།
[18961] །ཞེས་སོ།
[18962] །གཉིས་པ་འཆར་བའི་སྒོ་སྒྲོན་མ་བཟུང་བ་ལ་སྒོ་དང་སྒྲོན་མ་གཉིས་ལས།
[18963] སྒོའི་གནད་ནི་སྙིང་ནས་ཀླད་པར་འབྲེལ་བའི་རྩ་སྐྱོན་ཡོན་དབང་པོ་དངས་བྱེད་པ་ལ་ཡལ་ག་ལྔ་ཡོད་པའི་མིག་ལ་བ་མེན་རྭ་འདྲའི་རྩའི་དཀྱིལ་ལས་འོད་རྩ་ནས་ཡེ་ཤེས་དངོས་སུ་འབྱུང་བས་མི་འགུལ་བར་ནམ་མཁའི་མཐོངས་སུ་བལྟ་བ་ནི་ཡང་དག་པའི་གནད་དེ་ཐལ་འགྱུར་ལས།
[18964] །ཡེ་ཤེས་འཆར་བའི་སྒོ་ཉིད་ནི།
[18965] །ལུས་བཅུད་དངས་མ་ཀུན་འདུས་པའི།
[18966] །ཙཀྵུ་ཞེས་པའི་སྒོ་ལས་འཐོན།
[18967] །གཉིས་ཀྱི་ཆ་འཛིན་སྣང་བ་སྟོན།
[18968] །སྐྱེད་བྱེད་རྩ་ལས་འབྲེལ་འབྱུང་བའོ།
[18969] །རྩེ་མོ་ལས་ནི་གཅིག་གྱུར་པ།

[19070] །དང་པོ་ནི་དབང་པོ་ལ་སྣང་ཞིང་དགག་སྒྲུབ་མེད་པ་སྟེ་དབྱེ་ན་དབང་པོ་དང༌།
[19071] ཕུང་པོ་དང༌།
[19072] ཡུལ་དང་ཡུལ་ཅན་རྣམ་པར་དག་པའི་སྐུ་དང་གསུམ་སྟེ།
[19073] དེ་ཉིད་ལས།
[19074] རིག་པ་ཡུལ་ལ་སྣང་བའི་སྐུ་ནི་དབང་པོ་རྣམས་རྣམ་པར་དག་པའི་སྐུ་ཆེན་པོ་དང༌།
[19075] ཕུང་པོ་རྣམས་རྣམ་པར་དག་པའི་སྐུ་ཆེན་པོ་དང༌།
[19076] ཁམས་དང་སྐྱེ་མཆེད་དང་ཡུལ་དང་ཡུལ་ཅན་དང་ཉོན་མོངས་པ་རྣམས་རྣམ་པར་དག་པའི་སྐུ་ཆེན་པོའོ་ཞེས་སོ།
[19077] །གཉིས་པ་དངས་མ་འདུས་པའི་སྐུ་ནི་གོམས་པ་སྟོབས་སུ་གྱུར་པ་རྣམས་ལ་སྣང་བ་སྟེ།

[19078] དབྱེ་ན་ལམ་ཕྱིན་ཅི་མ་ལོག་པ་རང་གི་མཚན་ཉིད་འཛིན་པའི་སྐུ་དང༌།
[19079] འོད་དང་སྐུའི་མཚན་ཉིད་ཁ་དོག་མི་ཕྱེད་པ་ཡེ་ཇི་བཞིན་པའི་སྐུ་གཉིས་ལས་དང་པོ་ནི།
[19080] །ངོ་བོ་རང་བཞིན་ཐུགས་རྗེ་གསུམ་གྱིས་ལམ་གྱི་མཚན་ཉིད་འཛིན་པས་འབྲས་བུ་ཚད་ལ་ཕེབས་པར་བྱེད་པས་རོལ་པ་ཆེན་པོའི་སྐུ་དང༌།
[19081] སྤྱི་ཕུད་ཆེན་པོའི་སྐུ་དང༌།
[19082] རིགས་རྗེས་སུ་འཛིན་པའི་སྐུ་ཞེས་གྲགས་ཏེ་ཉི་ཟླ་ཁ་སྦྱོར་ལས།
[19083] རང་གི་མཚན་ཉིད་འཛིན་པའི་སྐུ་ནི་འདི་ལྟ་སྟེ།
[19084] རྣམ་པར་མི་འབྱེད་པ་རོལ་པ་ཆེན་པོའི་སྐུ་དང༌།
[19085] དམ་པ་སྤྱི་ཕུད་ཆེན་པོའི་སྐུ་དང༌།
[19086] རིག་རྗེས་སུ་འཛིན་པའི་སྐུའོ་ཞེས་སོ།
[19087] །གཉིས་པ་ཡེ་ཇི་བཞིན་པའི་སྐུ་ནི་གཞི་དང་ལམ་དང་འབྲས་བུ་གསུམ་སྟེ།
[19088] གཞི་རང་བྱུང་གི་ཡེ་ཤེས་ནས་གནས་པའི་སྐུ་དང༌།
[19089] ལམ་འོད་གདངས་དབྱིངས་སྐུ་ཡེ་ཐོག་ལས་ཕྱིར་སྣང་ཙམ་ལས་མི་འགྱུར་བའི་སྐུ་དང༌།
[19090] འབྲས་བུ་སྐུ་གསུམ་ལྷུན་གྲུབ་དབྱིངས་ཟད་སར་ཕྱིན་པའི་སྐུ་གསུམ་སྟེ་ཡང་དེ་ཉིད་ལས།
[19091] ཡེ་ཇི་བཞིན་པའི་སྐུ་ནི་འདི་ལྟ་སྟེ།
[19092] ཡེ་གནས་དང་པོའི་སྐུ་དང༌།
[19093] ཡེ་ཐོག་མི་འགྱུར་བའི་སྐུ་དང༌།
[19094] འབྲས་བུ་རང་རྫོགས་པའི་སྐུའོ།
[19095] །འབྲས་བུ་རང་རྫོགས་པའི་སྐུ་ནི་གསུམ་སྟེ།
[19096] ཆོས་ཀྱི་སྐུ་སྤྲོས་པ་མེད་པ་ཆེན་པོ་དང༌།
[19097] ལོངས་སྤྱོད་རྫོགས་པའི་སྐུ་འཛིན་པ་མེད་པ་ཆེན་པོ་དང༌།
[19098] སྤྲུལ་པའི་སྐུ་བདག་མེད་པ་ཆེན་པོའོ་ཞེས་སོ།
[19099] །གཉིས་པ་ཁྱད་པར་གྱི་དབྱེ་བ་རྒྱས་པར་བཤད་པ་ལ་གཉིས་ཏེ་ངོ་བོ་མདོར་བསྟན་པ་དང༌།
[19100] རྣམ་པར་རྒྱས་པར་བཤད་པའོ།
[19101] །དང་པོ་ནི་རང་ཤར་ལས།
[19102] དེ་ཡང་སྐུ་གང་ཞེ་ན་རྩ་བ་རིག་པའི་སྐུ་དང༌།
[19103] ཡངས་པ་ཆོས་ཉིད་ཀྱི་སྐུ་དང༌།
[19104] གདལ་བ་དབྱིངས་ཀྱི་སྐུ་དང༌།
[19105] འགྱུར་བ་མེད་པ་ངོ་བོ་ཉིད་ཀྱི་སྐུ་དང༌།
[19106] ཡང་དག་བདེ་བའི་སྐུ་དང༌།
[19107] ལོག་པ་སེམས་ཀྱི་སྐུ་དང༌།
[19108] མཐོང་བ་རང་གྲོལ་གྱི་སྐུ་དང༌།
[19109] གཅིག་པ་ཐིག་ལེའི་སྐུ་དང༌།
[19110] རྒྱ་མ་ཆད་པ་ནམ་མཁའི་སྐུ་དང༌།
[19111] ཕྱོགས་སུ་མ་ལྷུང་བ་ཉི་ཟླའི་སྐུ་དང༌།

[19112] ཐ་མི་དད་པ་ཅིག་ཤོས་ཀྱི་སྐུ་དང༌།
[19113] ཞེན་པ་མེད་པ་རང་གྲོལ་གྱི་སྐུའོ་ཞེས་སོ།
[19114] །གཉིས་པ་ནི།
[19115] སྐུ་བཅུ་གསུམ་པོ་དེ་དག་གི་རྣམ་པ་གང་ཡིན་ཞེ་ན་བཤད་པ་ནི་ཡང་དེ་ཉིད་ལས།
[19116] དེ་ཡང་འདི་ལྟ་སྟེ།
[19117] རྩ་བ་རིག་པའི་སྐུ་ནི་རྡོ་རྗེ་ལུ་གུ་རྒྱུད་ཀྱི་སྐུ་ཡིན་ཏེ་རང་གི་མིག་གི་དབང་པོ་མ་འགགས་པ་ལ་གསལ་ལོ།
[19118] །དེ་ནི་རྩ་བ་རིག་པའི་སྐུ་ཞེས་བྱའོ།
[19119] །ཡངས་པ་ཆོས་ཉིད་ཀྱི་སྐུ་ནི་དང་པོའི་གཞི་ཀ་དག་ཆེན་པོའི་དེ་ལས་བསམ་པ་མེད།
[19120] མ་རིག་པ་མེད།
[19121] སེམས་མེད།
[19122] ཡིད་མེད།
[19123] འཛིན་པ་མེད་དོ།
[19124] །དེ་ཡང་འདི་ལྟ་སྟེ།
[19125] རྒྱ་མ་ཆད་པའི་ཡེ་ཤེས།
[19126] ལྷུན་གྱིས་གྲུབ་པའི་སངས་རྒྱས།
[19127] སྤྲོས་པ་དང་བྲལ་བའི་ཆོས་ཉིད།
[19128] མཐའ་དང་བྲལ་བའི་རིག་པ།
[19129] དག་པ་ཆེན་པོའི་སྣང་བ།
[19130] ཕྱོགས་ལྷུང་མེད་པའི་ལྟ་བར་བཞུགས་སོ།
[19131] །དེ་ཡང་ཡངས་པ།
[19132] གསལ་བ།
[19133] མི་འགྱུར་བ།
[19134] ཡེ་ཤེས་མི་འགག་པ།
[19135] ཐིག་ལེ་སྣ་ཚོགས་སུ་འཕྲོ་བ།
[19136] མི་འགྱུར་བའི་ཆོས་ཉིད།
[19137] ཕྱི་རྒྱ་མ་རལ་བ།
[19138] ནང་ཡེ་ཤེས་ཀྱི་འོད་ལྔ་མ་འགགས་པར་གསལ་བ།
[19139] སྐུ་དང་ཡེ་ཤེས་ཀྱི་ངོ་བོར་བཞུགས་སོ།
[19140] །དེ་ནི་ཡངས་པ་ཆོས་ཉིད་ཀྱི་སྐུ་ཞེས་བྱའོ།
[19141] །གདལ་བ་དབྱིངས་ཀྱི་སྐུ་ནི་དབྱིངས་རྣམ་པར་དག་པའི་སྒྲོན་མ་ཡིན་ཏེ།
[19142] དབྱིངས་དེ་ཡང་རྒྱ་མ་ཆད་པ།
[19143] ཕྱོགས་སུ་མ་ལྷུང་བ།
[19144] འགྲིབ་པ་མེད་པ།
[19145] འཕེལ་བ་མེད་པ།
[19146] སྟོངས་པ་མེད་པ།
[19147] གང་བ་མེད་པ།
[19148] གསལ་ཆ་དང་བཅས་པ།
[19149] སྐུ་མང་པོ་འཆར་བའི་གནས།
[19150] ཡངས་པ་ཀུན་གྱི་བདག་ཉིད།
[19151] ཤིན་ཏུ་ཆེ་བས་གོང་ན་མེད་པ།
[19152] ལྟ་བ་མཐོང་བའི་སངས་རྒྱས།
[19153] ངེས་པ་རྙེད་པའི་ས་ཞེས་བྱའོ།
[19154] །དེ་ནི་གདལ་བ་དབྱིངས་ཀྱི་སྐུ་ཞེས་བྱའོ།
[19155] །མི་འགྱུར་བ་རྡོ་རྗེའི་སྐུ་ནི་སངས་རྒྱས་མེད་པའི་འབྲས་བུ།
[19156] སེམས་ཅན་མེད་པའི་འཁོར་བ།
[19157] སྲོག་བཅད་པའི་སངས་རྒྱས།
[19158] བྱས་པ་མེད་པའི་བྱུང་བ།
[19159] མཐོང་མ་མྱོང་བའི་ཡུལ།
[19160] ཤེས་མ་མྱོང་བའི་གཏམ།
[19161] ལེན་མ་མྱོང་བའི་ལུས།

[19283] གཞི་གནས་ཀྱི་ཡེ་ཤེས་ནི་གསུམ་སྟེ་ངོ་བོ་ཀ་དག་གི་ཡེ་ཤེས་དང།
[19284] རང་བཞིན་ལྷུན་གྲུབ་ཀྱི་ཡེ་ཤེས་དང༌།
[19285] ཐུགས་རྗེ་རང་སྣང་གི་ཡེ་ཤེས་སོ་ཞེས་སོ།
[19286] །གཉིས་པ་མཚན་ཉིད་འཛིན་པའི་ཡེ་ཤེས་ལྔ་ནི།
[19287] སྐུ་ལ་དབྱིངས་ཡོད་པས་དེའི་རྩལ་ལས་ཤར་བ་་རང་ཤར་ལས།
[19288] སྐུ་མི་འགྱུར་བ་ལས་ཡེ་ཤེས་འགག་པ་མེད་པ་འབྱུང་ངོ༌།
[19289] །དེ་ཡང་འདི་ལྟ་སྟེ།
[19290] སྐུ་ལ་དབྱིངས་ཡོད་པའི་ཕྱིར།
[19291] ཆོས་ཀྱི་དབྱིངས་ཀྱི་ཡེ་ཤེས་དང༌།
[19292] དབྱིངས་ལ་གསལ་ཆ་ཡོད་པའི་ཕྱིར།
[19293] མེ་ལོང་ལྟ་བུའི་ཡེ་ཤེས་སུ་བྱུང༌།
[19294] །གསལ་བ་ལ་གཉིས་མེད་ཡོད་པའི་ཕྱིར།
[19295] །མཉམ་པ་ཉིད་ཀྱི་ཡེ་ཤེས་བྱུང༌།
[19296] གཉིས་མེད་ལ་མཐོང་བ་ཡོད་པའི་ཕྱིར།
[19297] །སོ་སོར་རྟོག་པའི་ཡེ་ཤེས་བྱུང༌།
[19298] མཐོང་བ་ལ་ཚད་ཡོད་པའི་ཕྱིར།
[19299] བྱ་བ་གྲུབ་པའི་ཡེ་ཤེས་བྱུང་ངོ་ཞེས་སོ།
[19300] །གསུམ་པ་འོད་ནི་ཡེ་ཤེས་ཀྱི་རང་གདངས་ཏེ།
[19301] ནང་གསལ་ཕྲ་བ་གཏིང་གསལ་ཁ་དོག་ལས་འདས་པ་དང༌།
[19302] ཕྱིར་གསལ་རགས་པ་ཁ་དོག་འཛིན་པའི་འོད་གཉིས་ལས།
