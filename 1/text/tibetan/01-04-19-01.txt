[3994] དང་པོ་འཁྲུལ་མ་མྱོང་བས་མ་རིག་པའི་རྒྱུད་དག་ལ།
[3995] ད་ལྟར་འཁྲུལ་པ་ལ་མི་གནས་པས་ལས་དང་ཉོན་མོངས་པའི་གཞི་སྟོངས་ཤིང༌།
[3996] ཕྱིས་འཁྲུལ་མི་སྲིད་པས་ལས་དགེ་སྡིག་གི་ཐག་ཆོད་པའོ།

[3999] མཚན་ཉིད་ལ་འཁྲུལ་རྒྱུ་མེད་པས་ཁམས་གསུམ་དོང་སྤྲུགས་པའོ།

[4002] རང་བཞིན་རྩིས་ལས་འདས་པས་འཁྲུལ་མ་འཁྲུལ་ཙམ་དུ་ཡང་གྲུབ་པ་མེད་པ་དང༌།
[4003] མཚན་ཉིད་ལ་གཉིས་མ་གྲུབ་པས་འཁྲུལ་མ་འཁྲུལ་གྱི་ཟླ་དང་བྲལ་བའོ།
[4004] །གསུམ་པ་ཐིག་ལེ་རང་གནས་ལ་ཕབ་པའི་ངོ་བོ་ཉག་གཅིག་སྤྲོས་པ་དང་བྲལ་བས་དུ་མ་ཆ་ཤས་མེད་པ་དང༌།

[4005] རང་བཞིན་ལ་དུ་མ་མེད་པས་ཐ་སྙད་སྤྲོས་པའི་ཚིག་ལས་འདས་པའི་ཆོས་ཉིད་ཀུན་ཏུ་ཁྱབ་པའོ།

[4012] མཚན་ཉིད་ལ་དཀྲིར་མེད་པས་ཡིད་ཆེས་ཤིང་གྲོལ་བའི་གདེང་དང་བཅས་པའོ།

[4017] རང་བཞིན།
[4018] འོད་གསལ་ལ་དྲི་མ་མེད་པས་ཡེ་ཤེས་ལྔའི་ཚན་ཉིད་ཤེས།
[4019] མཚན་ཉིད་རྟོག་དཔྱོད་ལས་འདས་པ་སྐུ་གསུམ་ཡེ་ཤེས་ལྔ་ལྡན་གྱི་དགོངས་པ་ལ་གནས་པའོ།
[4020] །དབྱེ་ན་བཞི་སྟེ།

[4022] ནང་གི་སྐོར།
[4023] གསང་བའི་སྐོར།
[4024] གསང་བ་བླ་ན་མེད་པའི་སྐོར་རོ།
[4025] །ཕྱི་སྐོར་གྱི་ངོ་བོ།
[4026] ཉོན་མོངས་པ་སྤང་དུ་མེད་པས་དུག་ལྔ་ལམ་ཁྱེར་ལ་གནས།
[4027] རང་བཞིན་ལ་རྩོལ་སྒྲུབ་མེད་པས་ཅིར་སྣང་ཆོས་ཉིད་དུ་འཆར།
[4028] མཚན་ཉིད་ལ་ཆ་ཤས་མེད་པས་སྟོང་ཉིད་གང་དུའང་མ་ཕྱེ་བའོ།
[4029] །ནང་སྐོར་གྱི་ངོ་བོ་གཟུགས་ཅན་མ་ཡིན་པས་མཚན་མ་མེད་པའི་ཆོས་ཉིད།
[4030] རང་བཞིན་འགྲོ་འོང་མེད་པས་དུས་ཐམས་ཅད་དུ་རྟག་པར་བཞུགས་པའི་ཡེ་ཤེས།
[4031] མཚན་ཉིད་ཟུག་པའི་ངོས་ནས་རྩ་བ་ལྟར་བྱུང་ལ།
[4032] འདྲིལ་བའི་ངོས་ནས་སྡོང་པོ་ལྟར་འབྱུང་ཞིང་གསལ་བའི་ངོས་ནས་མེ་ཏོག་ལྟར་བྱུང་ལ།
[4033] སྨིན་པའི་ངོས་ནས་འབྲས་བུ་ལྟར་བྱུང་བའོ།
[4034] །གསང་སྐོར་གྱི་ངོ་བོ།
[4035] ངོ་སྤྲད་པ་དང༌།
[4036] རྟོགས་པ་ཐོབ་པ་དུས་མཉམ་པས་ཐོས་བསམ་སྒོམ་གསུམ་ལ་རག་མ་ལུས།
[4037] རང་བཞིན་དབུགས་ཆད་པ་དང་སངས་རྒྱས་པ་དུས་མཉམ་པས་རྩོལ་སྒྲུབ་དང་གོམས་སྟོབས་ལ་རག་མ་ལུས།
[4038] མཚན་ཉིད་སངས་རྒྱས་པ་དང་ཐུགས་རྗེ་འབྱུང་བ་དུས་མཉམ་པས་ཚོགས་དང་བསོད་ནམས་ལ་རག་མ་ལུས་པའོ།
[4039] །གསང་བ་བླ་ན་མེད་པའི་སྐོར་གྱི་ངོ་བོ།

[4040] ཚིག་ལ་མ་བརྟེན་པས་ཤེས་རབ་ལ་རག་མ་ལུས།
[4041] རང་བཞིན།
[4042] མངོན་སུམ་དུ་མཐོང་བས་ཡིད་དཔྱོད་འཛིན་པའི་ལྟ་བ་ལ་མི་གནས།
[4043] མཚན་ཉིད་སྣང་བ་བཞི་ཚད་ལ་ཕེབས་པས་འབྲས་བུ་སྐུ་གསུམ་ཡེ་ཤེས་ལྔ་ལ་ལྟོས་མི་འཆའ་བའོ།
[4044] །དེ་ལས་འདི་ཉིད་གསང་བ་བླ་ན་མེད་པ་སྟེ།
[4045] མན་ཆད་ཀྱིས་ཆོས་ཉིད་ཡིད་དཔྱོད་དུ་གདེང་འཆའ་བ་ལས།
[4046] འདིར་མངོན་སུམ་དབང་པོའི་ཡུལ་དུ་བླ་མའི་མཐུས་བལྟར་ཡོད་པས་ཁྱད་པར་དུ་འཕགས་སོ།
[4047] །དེ་ལྟར་བསྟན་པའི་སྐོར་ཐམས་ཅད་མདོ་གཅིག་ཏུ་དྲིལ་ནས་དབྱེ་བ་དང་བསྡུ་བའི་རིམ་པ་ངེས་པ་ནི།
[4048] རྫོགས་པ་ཆེན་པོ་དེ་དག་ལ་ཀློང་ན་དགུ
[4049] བམ་པོ་ན་ཉི་ཁྲི་དང་ཆིག་སྟོང༌།
[4050] སྡེ་ན་གསུམ།
[4051] ལེའུ་ན་སྟོང་ཕྲག་སུམ་ཅུ་རྩ་ལྔ།
[4052] འཕྱོང་ཆེན་པོ་བརྒྱད།
[4053] གཟེར་བུ་བརྒྱ་དང་བརྒྱད་ཅུ།
[4054] གགས་ན་སྟོང་དང་ལྔ་བརྒྱ།
[4055] ཤན་འབྱེད་ཁྲིག་ཕྲག་བདུན།
[4056] ལ་བཟླ་བ་འབུམ་ཕྲག་ཕྱེད་དང་གཉིས།
[4057] གནད་སྟོང་ཕྲག་ལྔ།
[4058] ཤླཽ་ཀ་འབུམ་ཕྲག་དྲུག་ཅུ་རྩ་བཞི།
[4059] མིང་དུ་ཐོགས་པའི་རྒྱུད་ཁྲི་ཕྲག་གཉིས་ཀྱི་བདག་ཉིད་དུ་བཞུགས་པ་རྣམས།
[4060] ཐལ་འགྱུར་ལས།
[4061] རང་བཞིན་རྫོགས་པ་ཆེན་པོ་ལས།
[4062] ཚིག་གི་བྱེ་བྲག་རྣམས་འཕྲོས་པ།
[4063] །སྡེ་གསུམ་ལས་ནི་ཀློང་དགུར་འགྱུར།
[4064] །ཚིག་སྡུས་པ་ལས་ཤླཽ་ཀ
[4065] འབུམ་ཕྲག་ཅུ་རྩ་བཞི་ལས།
[4066] །ལེའུ་སྟོང་ཕྲག་སུམ་ཅུ་ལྔ།
[4067] བམ་པོ་ཉི་ཁྲི་སྟོང་ཕྲག་གཅིག
[4068] །བརྗོད་བྱ་ནམ་མཁའི་མཐའ་དང་མཉམ།
[4069] །གདུལ་བྱའི་སེམས་ཅན་རྟོག་པ་གཅོད།
[4070] །འདི་ལས་བརྗོད་བྱ་ཉེར་བསྡུས་པ།
[4071] །གཟེར་བུ་བརྒྱ་དང་བརྒྱད་ཅུ་ལ།
[4072] །གགས་ནི་སྟོང་ཕྲག་ཕྱེད་དང་གཉིས།
[4073] །གནད་ཆེན་འདུས་པ་སྟོང་ཕྲག་གསུམ།
[4074] །གོལ་སྒྲིབ་ཁྲི་ཕྲག་བཞི་བཅུ་ལ།
[4075] །མིང་ཐོགས་རྒྱུད་ནི་ཁྲི་ཕྲག་གཉིས།
[4076] །ཚིག་བརྗོད་རིམ་པ་གསལ་བཀོད་དོ།

[4102] །དེ་ཡང་མིང་དུ་ཐོགས་པའི་རྒྱུད་སྟོང་ཕྲག་གསུམ་དུ་བསྡུས་སོ།
