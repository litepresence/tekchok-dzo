[4519] སྦྲུལ་གྱི་མདུད་པ་ལྟར་གཞུང་ཚིག་རང་གྲོལ་དུ་སྟོན་པ་སྦྲུལ་མདུད་ལྟ་བུའི་རྒྱུད་རིག་པ་རང་གྲོལ།
[4520] ཁྱུང་ཆེན་པོ་ནམ་མཁའ་ལ་རྩོལ་སྒྲུབ་མེད་པར་འགྲོ་བ་ལྟ་བུ་ལྟ་བའི་རྒྱུད།
[4521] ཀུན་ཏུ་བཟང་པོ་ཐུགས་ཀྱི་མེ་ལོང་གི་རྒྱུད།
[4522] དབང་དང་བྱིན་རླབས་ཆུ་བོའི་རྒྱུན་བཞིན་པས་ཆུ་བོ་ལྟ་བུའི་རྒྱུད་རྫོགས་པ་རང་བྱུང༌།
[4523] རྒྱལ་པོའི་གདུང་མི་འགྱུར་བས་རྒྱལ་ས་ཟིན་པ་ལྟ་བུའི་རྒྱུད་སྐུ་གདུང་འབར་བ།
[4524] ཕྱུག་པོའི་བང་མཛོད་གང་འདོད་འབྱུང་བ་ལྟ་བུའི་རྒྱུད་རིན་པོ་ཆེ་སྤུངས་པ།
[4525] སྤུ་གྲི་ངར་མས་གང་ལ་རེག་པ་གཅོད་པ་ལྟར་འགལ་བྱེད་ཆམ་ལ་འབེབས་པས་སྤུ་གྲི་ལྟ་བུའི་རྒྱུད་ནག་མོ་ཁྲོས་མ་དང་བཅོ་བརྒྱད་དོ།
[4526] །དེའང་ནག་མོ་ཁྲོས་མ་ནི་བསྟན་པ་བསྲུང་བའི་ཕྱིར་རྒྱུད་ནས་འཕྲོས་པ་ཡིན་ལ།
[4527] གཞན་བཅུ་དྲུག་ཐལ་འགྱུར་གྱི་རྩ་བ་ལས་འཕྲོས་པ་ཡིན་ཏེ།
[4528] སྒྲ་ཐལ་འགྱུར་ཆེན་པོའི་རྒྱུ་ལས།
[4529] འོད་ཀྱི་ཡང་ཞུན་བཅུ་བདུན་བཤད།
[4530] །རང་ཤར་བ་དང་རང་གྲོལ་དང༌།
[4531] །རང་བྱུང་ཉིད་དང་རྩལ་རྫོགས་དང༌།
[4532] །མཛེས་ལྡན་དག་དང་འཕྲ་བཀོད་དང༌།
[4533] །མུ་ཏིག་ཕྲེང་དང་ཡི་གེ་མེད།
[4534] །དབྱིངས་དང་ཐུགས་ཀྱི་མེ་ལོང་དང༌།
[4535] །ཉི་ཟླ་ཁ་སྦྱོར་ངོ་སྤྲོད་སྤྲས།
[4536] །སྐུ་གདུང་བར་དང་རིན་ཆེན་སྤུངས།
[4537] །སྒྲོན་མ་འབར་དང་ཀློང་ཆེན་དྲུག
[4538] །དེ་ལྟར་བཅུ་དྲུག་རིམ་པ་རྣམས།
[4539] །ཐལ་འགྱུར་ལས་ནི་ངེས་འཕྲོས་ནས།
[4540] །གདུལ་ཡུལ་སེམས་ཅན་རྣམས་ལ་སྣང༌།
[4541] །ཞེས་སོ།
[4542] །དེ་དག་གི་གནས་ངེས་པ་ནི།
[4543] མུ་ཏིག་ཕྲེང་བ་ལས།
[4544] ཚིག་རྒྱུད་རིམ་པས་གསལ་བར་དབྱེ།
[4545] །སྤྲོས་པ་གཅོད་ཕྱིར་འཁོར་བ་ཡི།
[4546] །དོན་རྣམས་རྒྱུད་ཀྱིས་གྲོལ་བར་འགྱུར།
[4547] །རྒྱ་མཚོ་དང་ནི་ཉི་མ་དང༌།
[4548] །སེང་གེ་དང་ནི་རི་རྒྱལ་དང༌།
[4549] །འཁོར་ལོ་དང་ནི་ལྡེ་མིག་དང༌།
[4550] །རང་གྲི་དང་ནི་གསལ་ཤིང་དང༌།
[4551] །གསེར་ཞུན་དང་ནི་མ་བུ་འབྲེལ།
[4552] །མེ་ལོང་དང་ནི་མུ་ཏིག་རྒྱུས།
[4553] །སྦྲུལ་མདུད་དང་ནི་ཁྱུང་ཆེན་དང༌།
[4554] །ཆུ་རྒྱུན་དང་ནི་སྤུ་གྲི་དང༌།
[4555] །རྒྱལ་པོ་དང་ནི་བང་མཛོད་དང༌།
[4556] །དེ་ལྟར་རྣམ་པར་ཕྱེ་བས་ནི།
[4557] །ཚིག་རྒྱུད་གསལ་བར་ཕྱེ་བའོ།

[4559] །གཉིས་པ་དགོས་ཆེད་རྒྱས་པར་བཤད་པ་ལ་སྡེ་ཚན་བཅུ་གཅིག་ལས།
