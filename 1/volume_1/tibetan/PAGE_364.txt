[15099] །བཅུ་ན་ས་ཡི་སྐྱེད་བྱེད་ཤ།
[15100] །བཅུ་ན་ནམ་མཁའི་སྤོར་བྱེད་སེམས།
[15101] །ཤེས་རབ་རྩ་ནི་ཉི་ཤུ་བརྒྱད།
[15102] །དགུ་ན་འབྱེད་བྱེད་པྲཛྙཱ་སྟེ།
[15103] །དགུ་ན་སྡུད་བྱེད་དངས་མའོ།
[15104] །ལྔ་ན་སྒྲོལ་བྱེད་ཤེས་རབ་པོ།
[15105] །ལྔ་ན་སྐྱོད་པར་བྱེད་པའོ།
[15106] །སེམས་ཀྱི་རྩ་ནི་བཅུ་རྣམས་ན།
[15107] །རེ་རེ་དག་ནི་གནས་པའོ།
[15108] །གཅིག་ན་གཉིས་སུ་གནས་པའོ།
[15109] །མ་རིག་རྩ་ནི་རྣམ་པ་བཅུ།
[15110] །གཉིས་ན་རྩ་བའི་མ་རིག་གནས།
[15111] །གཉིས་ན་འཁྲུལ་པའི་མ་རིག་གནས།
[15112] །གཉིས་ན་འཁྲུལ་བཞིའི་མ་རིག་གནས།
[15113] །གཉིས་ན་འཛིན་པ་རྟོག་པའི་གནས།
[15114] །ལྷག་མ་དག་ན་རེ་རེའོ།
[15115] །ཡིད་ནི་རྣམ་པ་བཅུ་གཉིས་ཏེ།
[15116] །ཡིད་རེ་རྩ་ནི་གཉིས་གཉིས་སོ།
[15117] །ཉོན་མོངས་རྩ་ནི་ཉི་ཤུ་བརྒྱད།
[15118] །བཅུ་ན་ཞེ་སྡང་གནས་པའོ།
[15119] །ལྔ་ན་དེ་བཞིན་འདོད་ཆགས་གནས།
[15120] །ལྔ་ན་གཏི་མུག་རྣམ་པར་གནས།
[15121] །བཞི་ན་ཕྲག་དོག་དེ་བཞིན་གནས།
[15122] །བཞི་ནང་ང་རྒྱལ་གནས་པའོ།
[15123] །དེ་ལྟར་རྩ་དང་རླུང་རྣམས་ནི།
[15124] །རིམ་པར་གནས་ཤིང་འཆར་བའོ།
[15125] །ཞེས་སོ།
[15126] །དེ་དག་གི་རྣམ་གྲངས་དང་བྱེད་ལས་བཤད་ན།
[15127] དེ་ཉིད་ལས།
[15128] སྤྱི་བོ་བདེ་ཆེན་འཁོར་ལོ་ལ།
[15129] །རིག་དང་ཡེ་ཤེས་རླུང་དང་ལས།
[15130] །འདུ་དང་ཁྲག་དང་ཤེས་པ་སྟེ།
[15131] །གནས་བདུན་པོ་ལ་རེ་རེ་ཞིང༌།
[15132] །རྩ་ནི་ལྔ་བཅུ་ལྔ་བཅུའོ།
[15133] །རྩ་ཡི་རྣམ་པ་དེ་དག་ནི།
[15134] །ལྔ་བཅུ་སྟེང་དུ་འགྲོ་བ་སྟེ།
[15135] །ཤེས་པའི་དངས་མ་སྡུད་པར་བྱེད།
[15136] །ལྔ་བཅུ་གཡོན་དུ་འགྲོ་བ་སྟེ།
[15137] །སེམས་ཀྱི་དངས་མ་སྡུད་པར་བྱེད།
[15138] །ལྔ་བཅུ་རྩིབས་ལ་གནས་པ་སྟེ།
[15139] །ཡུལ་གྱི་དངས་མ་འཛིན་པར་བྱེད།
[15140] །ལྔ་བཅུ་དབུས་ན་གནས་པ་སྟེ།
[15141] །རང་རང་ཤེས་པའི་ཆ་འཛིན་ནོ།
[15142] །ལྔ་བཅུ་ཀུན་ཏུ་ཁྱབ་པ་སྟེ།
[15143] །དབུགས་དང་དངས་མ་ཡིད་ཀྱི་རྟ།
[15144] །ལུས་དང་དབང་པོར་བཅས་བཅུའོ།
[15145] །སྒོ་ལ་མི་དབྱུང་ཉི་ཤུའོ།
[15146] །དབུགས་ཀྱི་རྟ་བྱེད་སུམ་ཅུའོ།
[15147] །འཁྲུག་མ་དང་ནི་ལས་རླུང་རྟ།
[15148] །ལས་ཀྱི་འབྱུང་བ་འཁྲུགས་པ་ཡི།
[15149] །རྩ་ནི་རྣམ་པ་བཞི་ཡོད་དེ།
[15150] །གྱེན་དུ་སྐྱུག་དང་ཐུར་དུ་འཁྲུ།
[15151] །བར་དུ་འཁྲུག་པར་བྱེད་པ་ཡི།
[15152] །མ་བསྒྲུངས་པ་ཡི་རྩ་ཆེན་བཞི།
[15153] །ཁྲག་དང་ཆུ་སེར་དངས་མ་ལས།
[15154] །ལུས་དང་ཡན་ལག་ན་གྱུར་ནས།
[15155] །སྙིང་ཡང་འདར་ཞིང་ལག་པ་འདར།
[15156] །མཆི་མ་འཛག་ཅིང་མགོ་བོ་འཁོར།
[15157] །ཡན་ལག་རྣམས་ཀྱང་སྦྲིད་པ་འབྱུང༌།
[15158] །དེས་ནི་ལུས་ཟུངས་འཆོར་བར་བྱེད།
[15159] །གཞི་ལ་གནས་པ་དྲུག་ཡིན་ཏེ།
[15160] །གནས་དང་སྟོང་དང་འབྱུང་བ་དང༌།
[15161] །ལས་ནི་སོ་སོར་བྱེད་པའོ།
[15162] །ཐིག་ལེ་སྡུད་དང་དངས་མ་སྤྲོ།
[15163] །བཞི་བཅུ་སྦྲ་ཡི་རེ་ཐག་བཞིན།
[15164] །སྦྲིང་དེ་བཞིན་ལ་གནས་པའོ།
[15165] །དེ་བཞིན་རྩ་ཡི་རྣམ་པ་རྣམས།
[15166] །རྒྱུ་ནི་སྟོང་རྒྱུ་འཇུ་བ་ཆེ།
[15167] །ལུས་ཀྱི་དྲོད་དང་ལས་གང་ཀུན།
[15168] །འདི་རྫོགས་གཅིག་ལ་འབྲས་བུ་ནི།
[15169] །སྣ་ཚོགས་སྡུད་ཅིང་བསྐྱོད་པའོ།
[15170] །ཤེས་པའི་དངས་མ་རེ་རེ་ལས།
[15171] །ཡེ་ཤེས་རླུང་ནི་གཉིས་གཉིས་སོ།
[15172] །འགྱུ་བྱེད་རླུང་ནི་བདུན་ཅུ་གཉིས།
[15173] །དབུགས་ནི་ཉི་ཁྲི་ཆིག་སྟོང་ལས།
[15174] །རྩ་ནི་འབུམ་ཁྲི་བཞི་རྣམས་དང༌།
[15175] །བདུན་སྟོང་དག་ནི་རྒྱུ་བར་བྱེད།
[15176] །སྤྱི་གཙུག་བདེ་འཁོར་རྩ་དེ་རྣམས།
[15177] །འཁྲིགས་པའི་ཚེ་དང་འཁྲུགས་པའི་ཚེ།
[15178] །གནས་པའི་ཚེ་དང་འགྱུ་བའི་ཚེ།
[15179] །སྟོངས་པ་དང་ནི་འཇུག་པའོ།
[15180] །འདུས་པས་ཡལ་བའི་ངང་དུ་ཐིམ།
[15181] །ཞེས་སོ།
[15182] །གཉིས་པ་དེ་ལས་གྱེས་པ་ཕྱི་རྩའི་འདབ་མ་དང་ཡན་ལག་ནི།
