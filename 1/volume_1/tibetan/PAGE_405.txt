[16952] །ཁ་དགོ་གསལ་བའི་འོད་ལྔར་ཡང༌།
[16953] །འདི་ཉིད་རབ་ཏུ་གསལ་འགྱུར་ཞིང༌།
[16954] །རང་གི་མདུན་གྱི་ཡར་མཚམས་སུ།

[16957] །དེ་ཡི་ཚད་དུ་དེ་འགྱུར་རོ།
[16958] །ཞེས་པ་དང༌།
[16959] རང་ཤར་ལས།
[16960] དེ་ནས་སྣང་བ་ཤར་བ་དང༌།
[16961] །རང་གི་ཤེས་རབ་ཆེན་པོ་ཡིས།
[16962] །བརྟགས་ན་ཡེ་ཤེས་སྣང་བ་ལ།
[16963] །སྐུ་དང་ཡེ་ཤེས་ཕྱོགས་མེད་ཤར།
[16964] །དེ་ཡང་འཆར་ལུགས་འདི་ལྟ་སྟེ།
[16965] །དང་པོ་མེ་ལོང་དཀྱིལ་འཁོར་ཙམ།
[16966] །དེ་ལས་སྣང་བ་དུམ་བུར་འཆར།།
[16967] །དེ་ལས་ཡུལ་གྲུ་ཙམ་དུ་འཆར།
[16968] །དེ་ལས་ནམ་མཁའ་གང་བར་འཆར།
[16969] །ཅི་སྣང་འོད་ཀྱི་དཀྱིལ་འཁོར་དུ།
[16970] །ཤར་ནས་གཟུགས་སྐུ་མཐོང་བའི་ཚེ།
[16971] །སྐུ་ཡི་དཀྱིལ་འཁོར་ཐམས་ཅད་ནི།
[16972] །རང་ལས་བྱུང་ཞིང་རང་ལས་སྤྲུལ།
[16973] །ནམ་མཁའ་གང་བའི་སྣང་བ་རྣམས།
[16974] །མཐོང་བའི་ཆོས་ཉིད་མ་འགགས་པའི།
[16975] །ཡེ་ཤེས་མཁའ་ལ་གཤེགས་པའི་རྟགས།
[16976] །རང་བཞིན་ཉིད་ཀྱི་སྣང་བ་ལ།
[16977] །ངོ་བོ་ཉིད་ཀྱི་སྐུ་ཤར་བ།
[16978] །ཤར་བ་དེ་ཉིད་རང་ཉིད་ཡིན།
[16979] །སངས་རྒྱས་དགོངས་པ་ཚད་མའི་ཆོས།
[16980] །ཡང་དག་དོན་ལ་དེ་བཞིན་གཤེགས།
[16981] །དེ་ནི་གོམས་པའི་ཚད་དུ་བསྟན།
[16982] །ཅེས་སོ།
[16983] །བརྒྱད་པ་གནས་ནི།
[16984] གཞིའི་དབྱིངས་ཙིཏྟ།
[16985] ལམ་གྱི་དབྱིངས་འབྲི་གུ་ཏ།
[16986] འབྲས་བུ་རང་ཤར་གྱི་དབྱིངས་ཀྱི་གནས་ནམ་མཁའ་སྤྲིན་བྲལ་ཏེ།
[16987] སྒྲོན་མ་འབར་བ་ལས།
[16988] མིག་ནི་འབྱུང་བའི་རྐྱེན་མེད་ལ།
[16989] །གནས་པས་རང་གི་རིག་པ་ཉིད།
[16990] །ལུ་གུ་རྒྱུད་དུ་སྣང་བ་སྟེ།
[16991] །སྤྲིན་བྲལ་མཁའ་ལ་འཛིན་པར་བྱེད།
[16992] །རང་ཤེས་ནང་དུ་གསལ་བའི་ཚེ།
[16993] །ཙིཏྟའི་དཀྱིལ་དུ་གནས་པ་ཡིན།
[16994] །ཕྱིར་འཕྲོས་སྤྲིན་བྲལ་ནམ་མཁའ་འོ།
[16995] །མིག་གི་རྣམ་ཤེས་དག་པ་ལ།
[16996] །གསལ་བྱེད་འབར་བའི་འོད་ཟེར་ལྔ།
[16997] །མི་འགྱུར་ཁྱབ་པའི་ཚུལ་དུ་ཡང༌།
[16998] །མཐིང་གའི་རང་གདངས་ཆེན་པོར་གནས།
[16999] །ཞེས་སོ།
[17000] །དགུ་པ་སྒོ་ནི།
[17001] མིག་ནས་འཆར་ཏེ།
[17002] དེ་ཉིད་ལས།
[17003] འདི་ཡི་འཆར་སྒོ་མིག་ནས་ཡང༌།
[17004] །འགྱུར་མེད་ཁྱབ་པའི་ཚུལ་དུ་སྣང༌།
[17005] །འདི་ཡི་སྣང་བ་རང་དག་པས།
[17006] །དབྱིངས་དང་རིག་པའི་ཟུང་འཇུག་པར།
