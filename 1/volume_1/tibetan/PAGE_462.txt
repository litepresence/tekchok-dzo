[19640] །རང་བཞིན་ཉིད་དང་ངོ་བོ་ཉིད།
[19641] །གཉིས་མེད་དུ་འབྲལ་མེད་པར་སྣང༌།
[19642] །འདི་དུས་ཟད་པའི་མཚན་ཉིད་ནི།
[19643] །བྱ་བྲལ་ནམ་མཁའ་ཇི་བཞིན་ནོ།
[19644] །དེ་ལ་ཅི་ཡིས་མཐོང་བ་ནི།
[19645] །གོམས་དང་གནད་ཀྱིས་མཐོང་བ་སྟེ།
[19646] །གོམས་པ་སྔོན་དུ་འགྲོ་བ་ནས།
[19647] །བརྩམས་ཏེ་ལུས་ངག་གནད་གཟིར་བས།
[19648] །རང་སྣང་དག་པའི་གཟུགས་མཐོང་ནས།
[19649] །འཁྲུལ་པ་ཐམས་ཅད་ནུབ་པའོ།
[19650] །ཞེས་སོ།
[19651] །དེ་ལྟར་ཉམས་སུ་ལེན་པའི་གནད་ཀྱི་ལམ་སྒྲོན་མ་བཞིར་འདུས་པས།
[19652] སྔར་ཡང་རྒྱས་པར་བསྟན་ཟིན་མོད་ཀྱི།
[19653] འདིར་སྐབས་བསྡུ་བས་ཤིན་ཏུ་གསལ་བར་དབྱེ་ན།
[19654] མཚན་ཉིད།
[19655] ཡུལ་རྟེན།
[19656] ཁྱད་པར་དང་བཞི་ལས།
[19657] དང་པོ་མཚན་ཉིད་ལ་གཉིས་ལས།
[19658] སྤྱིའི་མཚན་ཉིད་ཡེ་ཤེས་ཀྱི་རང་སྣང་སྟོན་ཞིང་འོད་གསལ་མ་བུ་སྤྲོད་པས་འཁོར་འདས་གཉིས་ཀྱི་མཚམས་འབྱེད་པ་སྟེ།
[19659] ཐལ་འགྱུར་ལས།
[19660] སྒྲོན་མའི་མཚན་ཉིད་འདི་ལྟ་སྟེ།
[19661] །སྤྱིར་སྣང་བ་སྟོན་པ་དང༌།
[19662] །ཡེ་ཤེས་མ་བུར་སྦྲེལ་བར་བྱེད།
[19663] །འཁོར་འདས་གཉིས་ཀྱི་མཚམས་སྦྱོར་བའོ།
[19664] །རང་གདངས་གསལ་བས་འོད་ལུས་ལ།
[19665] །རྟོག་བྲལ་ཡེ་ཤེས་སྡུད་པར་བྱེད།
[19666] །རྟོག་བཅས་ཡུལ་གྱི་སྣང་བ་ལ།
[19667] །མི་རྟོག་ཡེ་ཤེས་གཞི་མ་འཛིན།
[19668] །ཅེས་སོ།
[19669] །བྱེ་བྲག་གི་མཚན་ཉིད་ལ་བཞིའི་དང་པོ་རྒྱང་ཞགས་ནི་འབྱུང་བཞིའི་དངས་མ་འདུས་པ་གང་ཞིག་ཡུལ་རྣམ་པ་གཉིས་ཀྱི་ཆ་འཛིན་པས་སྣང་བ་མཆེད་པའི་སྒོར་གྱུར་བ་སྟེ།
[19670] ཐལ་འགྱུར་ལས།
[19671] རྒྱང་ཞགས་སྒྲོན་མའི་མཚན་ཉིད་ནི།
[19672] །ཁྱབ་པས་འཇུག་ལ་འཇུག་པས་ཁྱབ།
[19673] །འབྱུང་བ་བཞི་བདག་བྱེད་པར་གནས།
[19674] །འཛིན་འཇུག་མཆེད་དང་བྱེད་པའོ།
[19675] །ཞེས་སོ།
[19676] །ཤེས་རབ་རང་བྱུང་གི་མཚན་ཉིད་ཕྱིའི་གདངས་ལས་ནང་ཤར་བློའི་རྟོགས་པས་ཆོས་ཉིད་དག་པའི་དགོངས་པ་ལ་གནས་པས་འཁོར་འདས་ལ་བཟླ་བ་སྟེ།
[19677] དེ་ཉིད་ལས།
[19678] ཤེས་རབ་དག་གི་མཚན་ཉིད་ནི།
[19679] །གསལ་བ་ཚིག་འཇུག་གཞི་མ་སྡུད།
[19680] །ཉོན་མོངས་ལས་དང་བག་ཆགས་བསྲེག
[19681] །སྨིན་བྱེད་འཁོར་འདས་སྲོག་གཅོད་པའོ།
[19682] །ཞེས་སོ།
[19683] །དེའང་གཞི་སྣང་དབང་པོ་ལ་ཤར་བ་དེ་གནས་སྐབས་སུ་སྐུ་དང་ཡེ་ཤེས།
