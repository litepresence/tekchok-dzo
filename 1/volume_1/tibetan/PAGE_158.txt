[6197] སྒྲ་ཐལ་འགྱུར་ལས།
[6198] བསྟན་པ་འཇིག་པའི་ཉེ་རྒྱུན་ནི།
[6199] །དེ་དག་སོ་སོའི་གནས་དག་ཏུ།
[6200] །གང་ཟག་ལོག་སྲེད་ཅན་ཞེས་བྱ།
[6201] །རེ་རེ་དག་གིས་བསྟན་པ་ཀུན།
[6202] །འཇིག་པར་བྱེད་པའི་ཉེ་རྒྱུ་ཡིན།
[6203] །འདི་ཉིད་དང་པོ་མུན་པའི་དུས།
[6204] །འཛམ་བུ་གླིང་གི་ནུབ་ཕྱོགས་སུ།
[6205] །རྒྱལ་པོ་ལག་པ་སོར་མོ་དྲུག
[6206] །མདུན་ན་སོ་ལྷག་སྔོན་པོ་ཡོད།
[6207] །ལག་པ་གཡས་པའི་གྲུ་མོ་ལ།
[6208] །སྨེ་བ་ནག་པོ་སྡིག་པའི་དབྱིབས།
[6209] །རབ་ཏུ་འཇིགས་པ་ཡོད་པ་ཡིས།
[6210] །སངས་རྒྱས་བསྟན་པ་མུན་པར་འགྱུར།
[6211] །ལོ་ནི་དྲུག་ཅུའི་བར་ལ་འགྱུར།
[6212] །དེ་ནས་དཔལ་འཛིན་དང་པོས་འཛིན།
[6213] །སྟོང་དང་སུམ་བརྒྱ་འདས་འོག་ཏུ།
[6214] །ཡང་ནི་འཛམ་གླིང་བྱང་གི་ངོས།
[6215] །གཟི་བརྗིད་ལྡན་པའི་རི་ཡི་རྩར།
[6216] །དགེ་སྦྱོང་གཟུགས་ལ་བྱམས་སེམས་བྲལ།
[6217] །གཡས་པ་ལ་ནི་རྣ་བ་གཉིས།
[6218] །མིག་ནི་ཟུར་གསུམ་མས་ཡར་འགེབས།
[6219] །འདི་དང་མུ་སྟེགས་སྡེ་རྩོད་ནས།
[6220] །ལོ་ནི་དྲུག་ཅུ་མུན་པར་འགྱུར།
[6221] །དེ་འོག་བདེ་བའི་ལྷ་ཞེས་བྱ།
[6222] །ལོ་བརྒྱའི་བར་ལ་འཕེལ་བར་བྱེད།
[6223] །དེ་འདས་འཛམ་གླིང་བྱང་ཤར་དུ།
[6224] །ཆུ་བོ་རབ་མེད་འབབ་པའི་རྩར།
[6225] །རྒྱལ་པོའི་ལུས་ལ་བསོད་ནམས་ལྡན།
[6226] །མིག་ནི་གཅིག་ལས་མེད་པ་ལ།
[6227] །ཡན་ལག་རགས་ཤིང་ལུས་ཤེད་ཅན།
[6228] །འཕོང་དང་སྒྱུ་རྩལ་རྣམས་ཀྱིས་བརྒྱན།
[6229] །འདིས་ནི་ཆོས་བྱེད་བསྣུབས་ནས་ནི།
[6230] །ལོ་ནི་དྲུག་ཅུ་བསྟན་པ་ཉིད།
[6231] །འདིར་ནི་ནུབ་ནས་གཞན་དུ་འགྱུར།
[6232] །དེ་འདས་རང་སངས་རྒྱས་ཀྱི་གཟུགས།
[6233] །རྡོ་རྗེ་གདན་ནས་ནུབ་ཏུ་འབྱུང༌།
[6234] །དེས་ནི་བསྟན་པ་དམ་པ་ཉིད།
[6235] །ལོ་ནི་བརྒྱ་དང་ཉི་ཤུར་གནས།
[6236] །དེ་འདས་འོག་ཏུ་འཛམ་གླིང་དབུས།
[6237] །གསེར་གྱི་གྲོང་ཁྱེར་ཞེས་བྱ་བར།
[6238] །གནས་བརྟན་ལུས་ལ་ལྕེ་རྩེ་གཉིས།
[6239] །བོངས་ཆུང་ཡན་ལག་ཐུང་བ་ཞིག
[6240] །སྡེ་པ་རྩོད་ཅིང་མཚོན་ཆ་འཛིན།
[6241] །འདི་བྱུང་བས་ནི་མུན་པར་ཡང༌།
[6242] །ལོ་ནི་དྲུག་ཅུའི་བར་ལ་འགྱུར།
[6243] །དེ་འདས་འོག་ཏུ་འཛམ་གླིང་ཤར།
[6244] །ཆུ་བོ་དག་ནི་ལོ་ཀི་ཏ།
[6245] །གཡས་ཀྱི་ཕྱོགས་ཀྱི་འགྲམ་དག་ཏུ།
[6246] །དཔལ་གྱི་སེང་གེ་ཞེས་བྱ་བས།
[6247] །ལོ་ནི་སྟོང་དང་བཞི་བརྒྱར་གནས།
[6248] །དེ་ནས་ཡང་ནི་འཛམ་གླིང་དབུས།
[6249] །ཆུ་བོ་ཆེན་པོ་གངྒའི་འགྲམ།
[6250] །རྒྱལ་པོའི་བུ་ལ་དཔུང་པ་ནི།
[6251] །མར་གད་མདོག་དང་མཚུངས་པ་ལ།
[6252] །རབ་ཏུ་བྱུང་ལ་སོར་བཞིའི་ཚད།
[6253] །འདི་བྱུང་བས་ནི་མུན་པར་ཡང༌།
[6254] །ལོ་ནི་དྲུག་ཅུའི་བར་ལ་འགྱུར།
[6255] །དེ་འདས་འོག་ཏུ་ཀ་མ་ལ།
[6256] །ཤཱི་ལ་ཞེས་པའི་གཙུག་ལག་ཁང༌།
[6257] །དེ་ཡི་བདག་པོ་གཞོན་ནུའི་མགོན།
[6258] །འདིས་ནི་ལོང་སྟོང་བར་དུ་གནས།
[6259] །དེ་འོག་འཛམ་བུ་གླིང་གི་ལྷོ།
[6260] །ཚངས་པའི་གླིང་ཞེས་བྱ་བའི་གྲོང༌།
[6261] །དམངས་ཀྱི་རིགས་ལ་སེང་གེའི་མགོ
