[17009] །སྣང་བ་མིག་དང་རང་འབྲེལ་བས།
[17010] །དེ་ཉིད་རྐྱེན་བྲལ་ནམ་མཁར་འཆར།
[17011] །དེ་ལྟར་འཆར་སྒོ་མ་འགགས་པའོ།
[17012] །ཞེས་སོ།
[17013] །བཅུ་པ་དབྱིངས་ཀྱི་སྒྲོན་མ་དངོས་ནི།
[17014] མིག་གི་ཟུར་གཉིས་ལས་ཤར་ཏེ་མཐིང་ག་ཁྱབ་པ་ཆེན་པོའི་འོད་མཐའ་མེད་པ་སྟེ།
[17015] ནང་དུ་རིག་པ་འོད་དེའི་ཁང་པ་ན་བཞུགས་ལ།
[17016] ཕྱིར་ལུ་གུ་རྒྱུད་ཐིག་ལེ་སྟོང་པ་དང་བཅས་པ་དེའི་ར་བ་ན་བཞུགས་ཏེ།
[17017] འོད་ལྔའི་སྣང་བ་དང་མུ་ཁྱུད་ཀྱི་ར་བ་གཉིས་དབྱིངས་དངོས་ཡིན་ནོ།
[17018] །དེ་ཡང་དངོས་པོ་རགས་པར་མེད་པས་གཟུང་ཡུལ་དུ་མ་གྲུབ།
[17019] ཆད་པ་ཕྱང་ཆད་པ་ཡིན་པས་རང་གསལ་འཇའ་ཚོན་ལྟ་བུ།
[17020] འབྱུང་བའི་གནོད་པས་མི་ཤིགས་པས་འདུས་བྱས་འཁོར་བ་མ་ཡིན་པ།
[17021] གནད་དང་ལྡན་པས་མཐོང་ནས་གོམས་ན་ཡེ་སངས་ལ།

[17024] འདི་ལྟར་གནད་གསུམ་མན་ངག་དང༌།
[17025] སུ་ལ་འཛོམ་པའི་སྐྱེས་བུ་ནི།
[17026] །དབྱིངས་ལ་གོམས་ཏེ་མངོན་དུའོ།
[17027] །སྲིད་པ་ཡི་ནི་ཡར་མཚམས་ན།
[17028] །འདི་ཉིད་གཉིས་འཇུག་ཟུར་ལས་ནི།
[17029] །ཐབས་དང་ཤེས་རབ་རང་བཞིན་ལས།
[17030] །གཉིས་ནི་འཁྲིལ་ཏེ་གཅིག་པར་ཡང༌།
[17031] །སོར་གཅིག་གིས་ནི་མ་རིག་པར།
[17032] །ཉིན་དུས་འབྱུང་བའི་རྐྱེན་ལྡན་པས།
[17033] །སྣང་བར་འགྱུར་ཏེ་མཚན་མོ་ཡང༌།
[17034] །མཐིང་ག་རང་ངོ་ཉིད་དང་འབྱར།
[17035] །དབྱིངས་ཀྱི་རང་བཞིན་འོད་ཀྱི་ཁྱིམ།
[17036] །འགྱུར་མེད་ཡེ་གདངས་ཆེན་པོར་ནི།
[17037] །སངས་རྒྱས་སེམས་ཅན་ཐམས་ཅད་ལ།
[17038] །སུས་ཀྱང་བྱས་པ་མེད་པར་ནི།
[17039] །ཡེ་ནས་གཉིས་མེད་མཉམ་པར་གནས།
[17040] །འདི་ནི་ཁྱབ་ཆེ་ཕྲ་བ་ལས།
[17041] །རང་གི་ཤེས་པའི་ཆར་ཤར་ནས།
[17042] །མངོན་སུམ་ཉིད་དེ་རབ་ཏུ་གསང༌།
[17043] །གསང་བ་ཆེན་པོ་འདི་ཉིད་ལ།
[17044] །གང་སུ་སྤྱོད་པར་འདོད་པ་དེས།
[17045] །ཡ་སངས་ཉིད་ལ་ཡང་སངས་རྒྱས།

[17047] །བཅུ་གཅིག་པ་འདིའི་མཐུན་དཔེ་ནི།
[17048] ཡི་གེ་ན་རོ་བཞིན་དུ་གཤོག་པ་གཉིས་བསྡོམས་པས་ན་རོ་གཅིག་འགྲུབ་པ་ལྟར།
