[7058] མཁའ་འགྲོ་མ་རབ་ཏུ་མི་བསྲུན་པ་ཡོད་པའི་གནས་དུར་ཁྲོད་ནི་མཆོག་ཏུ་ཤིས་སོ།
[7059] །དེ་སྐད་དུ་ཡང༌།
[7060] རྫོགས་པ་རང་བྱུང་ལས།
[7061] དབང་བསྐུར་དམ་པ་འདི་ཉིད་བསྟན།
[7062] །གྲོང་རྡལ་ཆེན་པོའི་དབུས་དག་ཏུ།
[7063] །རྣལ་འབྱོར་པ་ཡིས་དོན་མི་འགྲུབ།
[7064] །གྲོང་དང་གྲོང་ཁྱེར་ཟད་མཐའ་དང༌།
[7065] །ནགས་སམ་རི་བྲག་ལུང་སྟོང་དུ།
[7066] །རྣལ་འབྱོར་དབང་གི་དོན་ལ་འཇུག
[7067] །ཅེས་པ་དང༌།
[7068] ཡང་ཤིན་ཏུ་དབེན་ཞིང་ཉམས་དགའ་བར།
[7069] །གནས་ཀྱི་མཚན་ཉིད་བརྟག་པར་བྱ།
[7070] །གྲོང་གདལ་ཟད་པའི་མཐའ་དག་དང༌།
[7071] །རྩི་ཞིང་ནགས་ཚལ་སྨིན་པའི་སར།
[7072] །མེ་ཏོག་སྣ་ཚོགས་འཁྲུངས་པ་དང༌།
[7073] །ནུབ་ཕྱོགས་མཐོ་ལ་ཤར་ཉིད་གདངས།
[7074] །བྱང་ཉིད་སྐྱོབ་པ་འདྲ་བ་དང༌།
[7075] །ལྷོ་ཕྱོགས་གདལ་བ་འདྲ་བའི་གནས།
[7076] །བདག་ཉིད་མངོན་པའི་ཕྱོགས་དག་ནས།
[7077] །གྲོག་པོ་སལ་སུལ་མེད་པ་དང༌།
[7078] །མདུན་ན་གཟུགས་པོ་མེད་པ་དང༌།
[7079] །སེམས་ཅན་འདྲ་བའི་གཟུགས་མེད་དང༌།
[7080] །རྒྱབ་ནི་བྲག་ལ་གཏད་པ་དང༌།
[7081] །ལོགས་ལ་ཤིང་ཐོག་སྣ་ཚོགས་འབར།
[7082] །མདུན་ན་ན་དང་ཆུ་གཙང་འབབ།
[7083] །ཡང་ན་ཚེ་ཟན་ལྟ་བུའི་གནས།
[7084] །རི་རྒྱུད་རྣམས་ཀྱང་ཡོངས་མ་ཆད།
[7085] །བྱ་སྐད་སྣ་ཚོགས་སྒྲོགས་པའོ།
[7086] །ཞེས་སོ།
[7087] །བཞི་པ་ནམ་གྱི་ཚེ་བསྐུར་བའི་དུས་ལ་གཉིས་ཏེ།
[7088] སྤྱིར་བསྟན་པ་དང༌།
[7089] བྱེ་བྲག་ཏུ་བཤད་པའོ།
[7090] །དང་པོ་ནི་བླ་མས་གནང་ནས་རྟེན་འབྲེལ་ཚོགས་པ་ནི་ཤིས་པའི་དུས་ཡིན་ལ།
[7091] འོན་ཀྱང་ཁྱད་པར་གྱི་དུས་ནི།
[7092] སྟོན་བྱང་ཆུབ་ཀྱི་འབྲས་བུ་སྨིན་པ་དང༌།
[7093] དཔྱིད་ཐར་པའི་ས་བོན་འདེབས་པ་ལ་མཚོན་པའོ།
[7094] །ལོ་ནི་སྤྲེའུ་ཤིས་སོ།
[7095] །ཡར་ངོའི་ཚེས་བརྒྱད་བཟང་ལ།
[7096] སྐར་མ་ནི་རྒྱལ་མཆོག་ལ།
[7097] གཟའ་ཕུར་བུ་བཟང་ངོ༌།
[7098] །གཉིས་པ་ལ་བཞི་ལས།
[7099] དང་པོ་སྤྲོས་བཅས་ཀྱི་དུས་དང་མཚན་མ་ནི།
[7100] དབང་བསྐུར་བ་ལ་རིམ་པར་འཇུག་པའི་དུས་གནད་བསྟན་པ་ལས།
[7101] སྤྲོས་བཅས་དག་གི་དབང་བསྐུར་ནི།
[7102] །བྱི་བ་འབྲུག་དང་རྟ་དང་ཕག
[7103] །ཁྱད་པར་ལོ་ནི་བཞི་དག་ལ།
[7104] །དབང་བསྐུར་ཆོ་ག་འགྲུབ་པ་ཡིན།
[7105] །ཟླ་བ་ར་བ་བཞི་དག་ལ།
[7106] །དུས་རྫོགས་འབྲས་བུ་འབྱིན་ཕྱིར་འགྲུབ།
[7107] །ཚེས་གྲངས་ཡར་གྱི་གཅིག་ནས་ནི།
[7108] །ཟླ་བ་རྒྱས་པའི་མཚམས་སུའོ།
[7109] །བསྐུར་དུས་ཉིན་ཕྱེད་ནམ་གུང་ངོ༌།
[7110] །རྣལ་འབྱོར་ལུས་ཀྱི་བྱེ་བྲག་གིས།
[7111] །སོ་སོའི་འབྱུང་བའི་ལོ་ལ་སྦྱར།
[7112] །དབང་བསྐུར་ཆོ་ག་ཤིས་པར་འགྱུར།
[7113] །འདི་དུས་སློབ་མ་དམ་ཚིག་ཅན།
[7114] །འབྱུང་བའི་ལུས་རྣམས་མཐུན་པ་ནི།
[7115] །གྲངས་ལ་ཚད་དུ་གཟུང་དུ་མེད།
[7116] །བརྒྱའམ་ལྔ་བཅུ་ལ་སོགས་པའི།
[7117] །འབྱུང་ལས་མཐུན་ན་འགལ་བ་མེད།
[7118] །གལ་ཏེ་ཆོ་ག་ལ་ཞུགས་ཚེ།
[7119] །མར་མེ་གདངས་དང་མི་ལྡན་ན།
[7120] །དམ་ཚིག་བསྐང་བའི་ཆོ་ག་བྱ།
[7121] །དམེ་བ་མེད་ན་དངོས་གྲུབ་ནི།
[7122] །ཡང་ནས་ཡང་དུ་བླང་བར་བྱའོ།
[7123] །དོན་མེད་ཤི་ན་བར་ཆད་འབྱུང༌།
[7124] །དབང་རྫོགས་འཛབ་ལ་དྲག་ཏུ་འབད།
[7125] །དེ་ཚེ་འདུས་པའི་མཆེད་ལྕམ་སུ།
[7126] །ལུས་ལ་ནད་ཉིད་བྱུང་བ་ན།
[7127] །ཕྱག་དང་བསྐོར་བ་དག་བྱའོ།
[7128] །ངག་ལ་ཀླབ་ཅོལ་སུ་ལ་བྱུང༌།
[7129] །བསམ་ཏེ་ཡི་གེ་བརྒྱ་པ་གདོན།
[7130] །སེམས་ནི་ལྟོགས་སམ་དྲན་ཉམས་ན།
[7131] །བསྐྱེད་དང་རྫོགས་པའི་རིམ་ལ་འཇུག
