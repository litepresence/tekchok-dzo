[7132] །གལ་ཏེ་བྱང་བའི་རྟགས་བྱུང་ན།
[7133] །གདོད་ཀྱིས་དབང་བསྐུར་ཆོ་ག་འབད།
[7134] །གལ་ཏེ་སྒྲུབ་པའི་རྫས་འཕེལ་ན།
[7135] །དངོས་གྲུབ་རབ་མཆོག་ཐོབ་པས་ན།
[7136] །མཉམ་པའི་དོན་ལ་མོས་པར་བྱ།
[7137] །དབང་བསྐུར་དག་ལ་ཞུགས་གྱུར་ནས།
[7138] །རྟག་ཏུ་མཚན་མ་བཟང་མཐོང་ན།
[7139] །ཐུན་མོངས་དངོས་གྲུབ་དེ་ཡིས་འགྲུབ།
[7140] །ཉམས་དང་ཤེས་པར་ལྡན་ན་ཡང༌།
[7141] །འདིས་ཀྱང་དགའ་བདེ་ཉིད་ལ་སྤྱོད།
[7142] །བླ་མ་ཉིད་ལ་མཆོག་ཏུ་ཡང༌།
[7143] །མོས་པ་བཟང་པོ་སུ་སྐྱེས་པ།
[7144] །འདིས་ཀྱང་ཕྱག་རྒྱ་མཆོག་ལ་སྦྱོར།
[7145] །བདག་ཉིད་ལུས་ངག་བདེ་མཐོང་ན།
[7146] །འདིས་ནི་སངས་རྒྱས་ཞིང་དག་གོ
[7147] །སྤྲོས་བཅས་དབང་གི་དུས་དང་ཡང༌།
[7148] །མཚན་མ་དག་ནི་ལེགས་བཤད་དོ།
[7149] །ཞེས་སོ།
[7150] །གཉིས་པ་སྤྲོས་མེད་ཀྱི་དུས་དང་མཚན་མ་ནི།
[7151] ཡང་དེ་ཉིད་ལས།
[7152] ད་ནི་སྤྲོས་པ་མེད་པ་ཡི།
[7153] །དུས་དང་མཚན་མ་ཉེར་བསྟན་བྱ།
[7154] །འདི་ལ་ལོ་ཡི་བྱེ་བྲག་མེད།
[7155] །གང་ལ་ཡང་ནི་ཤེས་པའོ།
[7156] །འདི་དུས་དབྱར་དང་དགུན་ལ་བཟང༌།
[7157] །ཚེས་གྲངས་ཡར་གྱི་ངོ་ལའོ།
[7158] །བསྐུར་དུས་ཐོ་རེངས་སྒྲ་སངས་ལའོ།
[7159] །འདི་ལ་དད་ལྡན་ལྔ་དང་དྲུག
[7160] །བདུན་ལས་མང་ན་དོན་མི་འགྲུབ།
[7161] །དེ་ལྟར་དད་དང་ལྡན་པའི་བུ།
[7162] །སེམས་ཀྱི་ལས་སུ་དགེ་ལྡན་ལ།
[7163] །ངག་གིས་བླ་མའི་བསྟོད་ལྡན་པ།
[7164] །དེ་ལྟ་བུ་ལ་དབང་བསྐུར་ཚེ།
[7165] །ལུས་ནི་བདེ་བ་མ་ཡིན་ན།
[7166] །ལུས་སྡིག་ཕྱི་ནས་བཤགས་པ་བྱ།
[7167] །དེ་བཞིན་ངག་དང་སེམས་ལ་ཡང༌།
[7168] །སྡིག་པ་བཤགས་པའི་ལས་བརྩམ་མོ།
[7169] །སྤྲོས་པ་མེད་པའི་དབང་བསྐུར་གྱི།
[7170] །དུས་དང་མཚན་མ་བཤད་པའོ།
[7171] །ཞེས་སོ།
[7172] །གསུམ་པ་ཤིན་ཏུ་སྤྲོས་མེད་ཀྱི་དུས་དང་མཚན་མ་ཡང༌།
[7173] །དེ་ཉིད་ལས།
[7174] ད་ནི་ཤིན་ཏུ་སྤྲོས་པ་མེད།
[7175] །མཆོག་ཏུ་གསང་བའི་དབང་བསྐུར་གྱི།
[7176] །དུས་དང་མཚན་མ་བཤད་པར་བྱའོ།
[7177] །དེ་ཡང་ལོ་ཡི་བྱེ་བྲག་མིན།
[7178] །རང་འཐད་ཉིད་དུ་བྱས་པས་འགྲུབ།
[7179] །ཕྱི་དུས་ས་བོན་འཇུག་པའི་དུས།
[7180] །བྱང་ཆུབ་ས་བོན་བརྟེན་པའི་ཕྱིར།
[7181] །འདི་ཡི་དུས་སུ་དབང་བསྐུར་ན།
[7182] །རང་བྱུང་ས་བོན་འཕེལ་བར་བཤད།
[7183] །འབྲུ་དང་འབྲས་བུ་སྡུད་དུས་སུ།
[7184] །ཤིན་ཏུ་སྤྲོས་མེད་དབང་བསྐུར་ན།
[7185] །ཡེ་ཤེས་འབྲས་བུ་རྫོགས་པའི་ཕྱིར།
[7186] །སྐུ་གསུམ་ས་ལ་ངེས་པར་འགྲོ
