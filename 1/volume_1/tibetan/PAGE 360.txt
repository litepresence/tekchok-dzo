[14862] །གིས་ནས་ནམ་མཁའི་དངས་མ་སེམས།
[14863] །རེ་ཞིག་དེ་ནས་འགྱུར་བ་ཙམ།
[14864] །ཁྱབ་པ་ཀུན་ལ་ཁྱབ་པའོ།
[14865] །སྟོང་པའི་ཐིག་ལེ་བརྒྱད་ཡིན་ཏེ།
[14866] །གཉིས་ནི་འགྱུ་བའི་ཐིག་ལེའོ།
[14867] །ལྷག་མ་སྟོང་འགྱུར་ཐིག་ལེའོ།
[14868] །འདུ་བའི་རྩ་ནི་བཅུ་བཞི་ལས།
[14869] །བཞི་ནི་བད་ཀན་སེམས་ཀྱི་རྩ།
[14870] །བཞི་ནི་གཏི་མུག་མཁྲིས་པའི་རྩ།
[14871] །བཞི་ནི་ཞེ་སྡང་རླུང་གི་རྩ།
[14872] །གཉིས་ནི་འདོད་ཆགས་ས་ཡི་རྩ།
[14873] །བཅུ་བཞི་ཉོན་མོངས་ལམ་ཡིན་པས།
[14874] །བཞི་ནི་མ་རིག་དངོས་པོའོ།
[14875] །ལྔ་ནི་རྟོག་བཅས་ཡིད་ཡིན་ནོ།
[14876] །ལྔ་ནི་དུག་གསུམ་ལམ་དུ་བཤད།
[14877] །དེ་དག་གནས་ནི་རེ་རེ་ལ།
[14878] །ལས་དང་རླུང་ནི་རེ་རེའོ།
[14879] །རྩ་དེ་དག་གི་འཕོ་འཇུག་དང་བྱེད་པ་ནི།
[14880] ཡང་དེ་ལས།
[14881] མགྲིན་པ་ལོངས་སྤྱོད་འཁོར་ལོའི་རྩ།
[14882] །བཞི་ནི་སྟེང་དུ་འགྲོ་བ་སྟེ།
[14883] །སྟོང་པའི་རྩལ་ནི་སྦྱོང་བར་བྱེད།
[14884] །གཉིས་ནི་ཀུན་ཏུ་ཁྱབ་པ་སྟེ།
[14885] །རིག་པའི་རང་རྩལ་སྦྱོང་བར་བྱེད།
[14886] །གཉིས་ནི་འོག་ཏུ་འགྲོ་བ་སྟེ།
[14887] །འབྱུང་བའི་དངས་སྙིགས་འཁྲུག་པར་བྱེད།
[14888] །བཞི་ནི་གཡས་སུ་འགྲོ་བ་སྟེ།
[14889] །ཡེ་ཤེས་རྩལ་ནི་སྦྱོང་བར་བྱེད།
[14890] །བཞི་ནི་གཡོན་དུ་འགྲོ་བ་སྟེ།
[14891] །ལས་ཀྱི་སྟོབས་ནི་སྐྱེད་པར་བྱེད།
[14892] །དེ་དག་རྣམས་ཀྱི་མཚན་ཉིད་ནི།
[14893] །ཕྱོགས་ཀྱི་ཆ་དག་རེ་རེ་ལ།
[14894] །འགྱུ་བའི་རླུང་ནི་གཉིས་གཉིས་སོ།
[14895] །ལྷག་མ་འགྱུ་བའི་དངས་མའོ།
[14896] །གཅིག་ན་གཉིས་ཀ་རྒྱུ་བའོ།
[14897] །དེ་དག་རྣམས་ཀྱིས་ཤེས་པའི་ཆ།
[14898] །ཐིམ་དང་ཟློག་དང་གནས་པའོ།
[14899] །ལས་དང་འབྲས་བུ་རྫོགས་པའོ།
[14900] །ཤེས་པའི་དངས་མ་རེ་རེ་ལ།
[14901] །འགྱུ་བའི་རྩལ་ནི་དྲུག་དྲུག་གོ
[14902] །ཐིམ་པའི་སྒོ་ནི་གཉིས་གཉིས་ཏེ།
[14903] །འཇུག་པའི་ས་ཡང་གསུམ་གསུམ་མོ།
[14904] །ཕྲ་དང་ཕྱེ་དང་ཕྱོགས་ལ་གནས།
[14905] །འགྱུ་དང་སྟོང་དང་མཐར་ཕྱིན་ནོ།
[14906] །འཛིན་པའི་དངས་མ་ལྔ་ཡི་ཆ།
[14907] །ཤེས་པས་བཟུང་བ་མ་ཡིན་ཏེ།
[14908] །གཟུགས་ལ་གཉིས་གཉིས་ཟུང་འཇུག་གོ
[14909] །འདུས་པས་ཐིག་ལེ་སྟོང་པ་སྟེ།
[14910] །གྱེས་པས་ས་ལ་རྣམ་པར་གྲོལ།
[14911] །བསྟན་པའི་ལྷ་མོ་མི་གཡོ་ཞིང༌།
[14912] །ལག་ན་རླུང་གི་བ་དན་འཕྱར།
[14913] །ཤུགས་འཆང་ལྷ་མོ་གསལ་འཛིན་མ།
[14914] །ལག་ན་རླུང་གི་བུམ་པ་ཐོགས།
[14915] །འབྱུང་བར་བྱེད་ཅིང་གནས་པའོ།
[14916] །ལྷ་མོ་ཆེན་པོ་མི་གཡོ་མ།
[14917] །ལག་ན་རླུང་གི་ཞགས་པ་འཕེན།
[14918] །གསལ་བྱེད་ལྷ་མོ་སྐྱོང་བྱེད་མ།
[14919] །ལག་ན་རླུང་གི་སྒྲོམ་བུ་ཐོགས།
[14920] །འཕྲོ་ཞིང་འཕེན་པར་བྱེད་པའོ།
[14921] །ལྷ་མོ་ཆེན་མོ་འདེགས་བྱེད་མ།
[14922] །ལག་ན་རླུང་གི་གྲུ་བཞུད་མ།
[14923] །རླུང་གི་ལྷ་མོ་དེ་དག་ནི།
[14924] །སྤྲོས་དང་སྤྲོས་པའི་ལས་བྱེད་མ།
[14925] །བསྡུ་དང་འཛིན་དང་སྤྲོ་བའོ།
[14926] །ཞེས་སོ།
