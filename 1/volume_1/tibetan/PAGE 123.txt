[4597] །བརྒྱད་པ་རྣལ་འབྱོར་གྲོལ་བའི་རྒྱུད་ནི།
[4598] སྒྲོན་མ་འབར་བ་སྟེ།
[4599] རྒྱལ་པོ་ཕོ་བྲང་གི་རྩེར་བཞུགས་ན་འདོད་དགུ་འགྲུབ་པ་ལྟར།
[4600] འདི་ཤེས་པས་དོན་རྣམས་འགྲུབ་པའི་དགོས་པ་ཡོད་དོ།
[4601] །དགུ་པ་སྙིང་དང་འདྲ་བ་ནི།
[4602] ཀློང་དྲུག་པ་སྟེ།
[4603] སྒོ་བསྡམས་པས་དགྲ་རྐུན་གྱིས་གླགས་མི་མཐོང་བ་ལྟར།
[4604] ལྟ་བ་རྟོགས་པས་གོལ་སྒྲིབ་དང་བར་ཆོད་མེད་པའི་དགོས་པ་ཡོད་དོ།
[4605] །བཅུ་པ་གསང་བ་རང་གཞུང་ནི།
[4606] ཐ་མ་ལྕགས་རིས་བསྐོར་བ་ལྟར་དོགས་པ་ཐམས་ཅད་གཅོད་པ་སྒྲ་ཐལ་འགྱུར་རོ།
[4607] །བཅུ་གཅིག་པ་མཚོན་ཆ་ལྟ་བུ་ནི།
[4608] ནག་མོ་ཁྲོས་མ་སྟེ།
[4609] སྒོ་སྲུང་བཟངས་བས་གནོད་བྱེད་ནང་དུ་མི་གཏོང་བ་ལྟར།
[4610] །འགལ་བྱེད་ཆམ་ལ་འབེབས་ཤིང་བར་ཆད་ཐམས་ཅད་བསྲུང་བའོ།

[4612] དེ་དག་ཀྱང་མུ་ཏིག་ཕྲེང་བ་ལས།
[4613] རྡོ་རྗེ་འཛིན་པ་ངེས་ཟུང་ཅིག
[4614] །རྒྱུད་ཀྱི་སྙིང་པོ་གསུམ་ཤེས་ན།
[4615] །རྒྱལ་བློན་འབངས་གསུམ་འཛོམ་པ་འདྲ།
[4616] །རྒྱུད་ཀྱི་དངས་མ་གསུམ་ཤེས་ན།
[4617] །རི་གསུམ་རྩེར་མོར་ཕྱིན་པ་འདྲ།
[4618] །རྒྱུད་ཀྱི་མེ་ཏོག་གསུམ་ཤེས་ན།
[4619] །མཁའ་ལ་ཉི་མ་གསུམ་ཤར་འདྲ།
[4620] །རྒྱུད་ཀུན་སྡོམ་གཅིག་ཤེས་པ་ནི།
[4621] །མཁར་གྱི་འགྲམ་རྡོ་ལེགས་བཙུད་འདྲ།
[4622] །སྣང་བ་གཏིང་རྫོགས་རྒྱུད་ཤེས་ན།
[4623] །རྩེ་མོར་འབྱོངས་པའི་མཁར་དང་འདྲ།
[4624] །ཡེ་ཤེས་གཡུལ་ལྡོག་རྒྱུད་ཤེས་ན།
[4625] །ཕྱོགས་བཞིར་སྐར་ཁུང་བཏོད་པ་འདྲ།
[4626] །ཡན་ལག་རྒྱུད་ཆེན་གཉིས་ཤེས་ན།
[4627] །ཕྱོགས་བཞིར་བ་གམ་བཏགས་པ་འདྲ།
[4628] །རྣལ་འབྱོར་གྲོལ་བའི་རྒྱུད་ཤེས་ན།
[4629] །རྒྱལ་པོ་མཁར་གྱི་རྩེར་བཞག་འདྲ།
[4630] །སྙིང་དང་འདྲ་བའི་རྒྱུད་ཤེས་ན།
[4631] །དེ་ལ་སྒོ་མོ་དམ་དཀྲིགས་འདྲ།
[4632] །གསང་བ་རང་གཞུང་རྒྱུད་ཤེས་ན།
[4633] །ཐ་མ་ལྕགས་རིས་བསྐོར་བ་འདྲ།
[4634] །མཚོན་ཆ་ལྟ་བུའི་རྒྱུད་ཤེས་ན།
[4635] །གོ་ཞུབ་ལྡན་པས་སྒོ་བསྲུངས་འདྲ།
