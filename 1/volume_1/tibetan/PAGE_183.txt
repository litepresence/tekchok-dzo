[7395] དབང་རྣམས་པར་ཕྱེ་བའི་འཕྲ་ཏིག་ལས།
[7396] དགའ་སྟོན་ཆེན་པོས་མཆོད་པ་ཡིས།
[7397] །བླ་མ་དང་ནི་བླ་མའི་འཁོར།
[7398] །ཡུམ་དང་བཅས་པ་ཐམས་ཅད་ལ།
[7399] །སུ་ཡིས་སྟོན་མོ་བྱས་ན་ནི།
[7400] །བསྐལ་པ་འབུམ་ཕྲག་བསམ་ཡས་སུ།
[7401] །སངས་རྒྱས་ཐམས་ཅད་མཉེས་བྱས་ཀྱང༌།
[7402] །བསོད་ནམས་འདི་ཡི་ཆར་མི་ཕོད།
[7403] །གལ་ཏེ་བླ་མའི་བ་སྤུའི་ཁུང༌།
[7404] །ཙནྡན་གྱིས་ནི་རེག་སྤྱོད་ན།
[7405] །འདི་ལ་བསོད་ནམས་་མཐའ་ཡས།
[7406] །དེ་བཞིན་རང་གི་རྡོ་རྗེའི་སྤུན།
[7407] །གཅིག་དང་གཉིས་དང་གསུམ་དག་གམ།
[7408] །ཡང་ན་བརྒྱ་སྟོང་གྲངས་མེད་ལ།
[7409] །ཚོགས་ཀྱི་དགའ་སྟོན་སུས་བྱས་པ།
[7410] །གལ་ཏེ་དབང་བསྐུར་དུས་མིན་པར།
[7411] །རང་གི་བླ་མ་མཆོད་པ་ནི།
[7412] །བརྒྱ་ཕྲག་གྲངས་མེད་བྱས་པ་བས།
[7413] །འདི་ལ་གཅིག་བྱས་བསོད་ནམས་ནི།
[7414] །བློ་ཡིས་དཔག་པར་ནུས་མ་ཡིན།
[7415] །འབྲས་ཆེན་བཟའ་བཏུང་སྟོན་མོ་ཡིས།
[7416] །རྣལ་འབྱོར་པ་རྣམས་མགུ་བྱས་ན།
[7417] །ལེགས་པ་ཐམས་ཅད་འགྲུབ་པ་དང༌།
[7418] །ཉེས་པ་ལས་ནི་གྲོལ་བར་འགྱུར།
[7419] །གལ་ཏེ་རྡོ་རྗེའི་སྲིང་མོའམ།
[7420] །འོན་ཏེ་བླ་མའི་སློབ་མ་ནི།
[7421] །རྣལ་འབྱོར་མ་ཡི་ཚོགས་རྣམས་ལ།
[7422] །ཚོགས་ཀྱི་སྟོན་མོས་མགུ་བྱས་ན།
[7423] །མ་མོ་བརྒྱད་དམ་ཕག་མོ་ལ།
[7424] །ཚོགས་ཀྱི་མཆོད་པ་བརྒྱ་ཕྲག་དགུ
[7425] །དུས་མ་ཡིན་པར་བྱས་པ་བས།
[7426] །འདི་དུས་རྣལ་འབྱོར་མ་གཅིག་ལ།
[7427] །བཟའ་བའམ་ཡང་ན་བཏུང་བའམ།
[7428] །གལ་ཏེ་སྐོམ་ཀྱང་ཉུང་བྱིན་ན།
[7429] །བསོད་ནམས་འདི་ནི་མཉམ་པ་མིན།
[7430] །དེས་ན་མ་མོའི་གྲོང་ཁྱེར་དུ།
[7431] །ཕྱིན་ཏེ་རྣལ་འབྱོར་མ་མཉེས་བྱའོ།
[7432] །རྣལ་འཕྱོར་ཕ་དང་མ་གཉིས་ནི།
[7433] །མཉམ་ཞིང་འདྲེས་པ་མཆོད་བྱས་ནས།
[7434] །འདི་ལ་དངོས་གྲུབ་འདི་ལྟ་བུ།
[7435] །ཡ་མཚན་མཆོག་དང་ལྡན་པ་འབྱུང༌།
[7436] །འཇིག་རྟེན་འདི་ལ་འདོད་དོན་དང༌།
[7437] །ཕྱི་མ་ཡེ་ཤེས་དབྱིངས་སྤྱོད་ཕྱིར།
[7438] །མཁའ་སྤྱོད་དག་གི་དངོས་གྲུབ་ནི།
[7439] །སྣ་ཚོགས་བདག་པོ་སུས་བྱས་པ།
[7440] །འཁོར་དང་བཅས་ཏེ་ནམ་མཁའ་ལ།
[7441] །སྤྱོད་པ་དག་ཏུ་ནུས་པ་ཡིན།
[7442] །ཅེས་སོ།
[7443] །དེའང་དགའ་སྟོན་རྒྱ་ཆེར་བྱས་ན།
[7444] ཚེ་འདི་ཉིད་ལ་དངོས་གྲུབ་སྣ་རེ་ཅིས་ཀྱང་ཐོབ་སྟེ་འདུ་བ་ཁྱད་པར་ཅན་གྱི་དུས་ཡིན་པའི་ཕྱིར་རོ།
[7445] །དངོས་གྲུབ་ཀྱི་མཚན་མ་བཤད་པ་ནི།
